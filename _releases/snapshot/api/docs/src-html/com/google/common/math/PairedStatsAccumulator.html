017package com.google.common.math;
018
019import static com.google.common.base.Preconditions.checkState;
020import static com.google.common.primitives.Doubles.isFinite;
021import static java.lang.Double.NaN;
022import static java.lang.Double.isNaN;
023
024import com.google.common.annotations.Beta;
025
026/**
027 * A mutable object which accumulates paired double values (e.g. points on a plane) and tracks
028 * some basic statistics over all the values added so far. This class is not thread safe.
029 *
030 * @author Pete Gillin
031 * @since 20.0
032 */
033@Beta
034public final class PairedStatsAccumulator {
035
036  // These fields must satisfy the requirements of PairedStats' constructor as well as those of the
037  // stat methods of this class.
038  private final StatsAccumulator xStats = new StatsAccumulator();
039  private final StatsAccumulator yStats = new StatsAccumulator();
040  private double sumOfProductsOfDeltas = 0.0;
041
042  /**
043   * Adds the given pair of values to the dataset.
044   */
045  public void add(double x, double y) {
046    // We extend the recursive expression for the one-variable case at Art of Computer Programming
047    // vol. 2, Knuth, 4.2.2, (16) to the two-variable case. We have two value series x_i and y_i.
048    // We define the arithmetic means X_n = 1/n \sum_{i=1}^n x_i, and Y_n = 1/n \sum_{i=1}^n y_i.
049    // We also define the sum of the products of the differences from the means
050    //           C_n = \sum_{i=1}^n x_i y_i - n X_n Y_n
051    // for all n >= 1. Then for all n > 1:
052    //       C_{n-1} = \sum_{i=1}^{n-1} x_i y_i - (n-1) X_{n-1} Y_{n-1}
053    // C_n - C_{n-1} = x_n y_n - n X_n Y_n + (n-1) X_{n-1} Y_{n-1}
054    //               = x_n y_n - X_n [ y_n + (n-1) Y_{n-1} ] + [ n X_n - x_n ] Y_{n-1}
055    //               = x_n y_n - X_n y_n - x_n Y_{n-1} + X_n Y_{n-1}
056    //               = (x_n - X_n) (y_n - Y_{n-1})
057    xStats.add(x);
058    if (isFinite(x) && isFinite(y)) {
059      if (xStats.count() > 1) {
060        sumOfProductsOfDeltas += (x - xStats.mean()) * (y - yStats.mean());
061      }
062    } else {
063      sumOfProductsOfDeltas = NaN;
064    }
065    yStats.add(y);
066  }
067
068  /**
069   * Returns an immutable snapshot of the current statistics.
070   */
071  public PairedStats snapshot() {
072    return new PairedStats(xStats.snapshot(), yStats.snapshot(), sumOfProductsOfDeltas);
073  }
074
075  /**
076   * Returns the number of pairs in the dataset.
077   */
078  public long count() {
079    return xStats.count();
080  }
081
082  /**
083   * Returns an immutable snapshot of the statistics on the {@code x} values alone.
084   */
085  public Stats xStats() {
086    return xStats.snapshot();
087  }
088
089  /**
090   * Returns an immutable snapshot of the statistics on the {@code y} values alone.
091   */
092  public Stats yStats() {
093    return yStats.snapshot();
094  }
095
096  /**
097   * Returns the population covariance of the values. The count must be non-zero.
098   *
099   * <p>This is guaranteed to return zero if the dataset contains a single pair of finite values.
100   * It is not guaranteed to return zero when the dataset consists of the same pair of values
101   * multiple times, due to numerical errors.
102   *
103   * <h3>Non-finite values</h3>
104   *
105   * <p>If the dataset contains any non-finite values ({@link Double#POSITIVE_INFINITY},
106   * {@link Double#NEGATIVE_INFINITY}, or {@link Double#NaN}) then the result is {@link Double#NaN}.
107   *
108   * @throws IllegalStateException if the dataset is empty
109   */
110  public double populationCovariance() {
111    checkState(count() != 0);
112    return sumOfProductsOfDeltas / count();
113  }
114
115  /**
116   * Returns the sample covariance of the values. The count must be greater than one.
117   *
118   * <p>This is not guaranteed to return zero when the dataset consists of the same pair of values
119   * multiple times, due to numerical errors.
120   *
121   * <h3>Non-finite values</h3>
122   *
123   * <p>If the dataset contains any non-finite values ({@link Double#POSITIVE_INFINITY},
124   * {@link Double#NEGATIVE_INFINITY}, or {@link Double#NaN}) then the result is {@link Double#NaN}.
125   *
126   * @throws IllegalStateException if the dataset is empty or contains a single pair of values
127   */
128  public final double sampleCovariance() {
129    checkState(count() > 1);
130    return sumOfProductsOfDeltas / (count() - 1);
131  }
132
133  /**
134   * Returns the <a href="http://mathworld.wolfram.com/CorrelationCoefficient.html">Pearson's or
135   * product-moment correlation coefficient</a> of the values. The count must greater than one, and
136   * the {@code x} and {@code y} values must both have non-zero population variance (i.e.
137   * {@code xStats().populationVariance() > 0.0 && yStats().populationVariance() > 0.0}). The result
138   * is not guaranteed to be exactly +/-1 even when the data are perfectly (anti-)correlated, due to
139   * numerical errors. However, it is guaranteed to be in the inclusive range [-1, +1].
140   *
141   * <h3>Non-finite values</h3>
142   *
143   * <p>If the dataset contains any non-finite values ({@link Double#POSITIVE_INFINITY},
144   * {@link Double#NEGATIVE_INFINITY}, or {@link Double#NaN}) then the result is {@link Double#NaN}.
145   *
146   * @throws IllegalStateException if the dataset is empty or contains a single pair of values, or
147   *     either the {@code x} and {@code y} dataset has zero population variance
148   */
149  public final double pearsonsCorrelationCoefficient() {
150    checkState(count() > 1);
151    if (isNaN(sumOfProductsOfDeltas)) {
152      return NaN;
153    }
154    double xSumOfSquaresOfDeltas = xStats.sumOfSquaresOfDeltas();
155    double ySumOfSquaresOfDeltas = yStats.sumOfSquaresOfDeltas();
156    checkState(xSumOfSquaresOfDeltas > 0.0);
157    checkState(ySumOfSquaresOfDeltas > 0.0);
158    // The product of two positive numbers can be zero if the multiplication underflowed. We
159    // force a positive value by effectively rounding up to MIN_VALUE.
160    double productOfSumsOfSquaresOfDeltas =
161        ensurePositive(xSumOfSquaresOfDeltas * ySumOfSquaresOfDeltas);
162    return ensureInUnitRange(sumOfProductsOfDeltas / Math.sqrt(productOfSumsOfSquaresOfDeltas));
163  }
164
165  /**
166   * Returns a linear transformation giving the best fit to the data according to
167   * <a href="http://mathworld.wolfram.com/LeastSquaresFitting.html">Ordinary Least Squares linear
168   * regression</a> of {@code y} as a function of {@code x}. The count must be greater than one, and
169   * either the {@code x} or {@code y} data must have a non-zero population variance (i.e.
170   * {@code xStats().populationVariance() > 0.0 || yStats().populationVariance() > 0.0}). The result
171   * is guaranteed to be horizontal if there is variance in the {@code x} data but not the {@code y}
172   * data, and vertical if there is variance in the {@code y} data but not the {@code x} data.
173   *
174   * <p>This fit minimizes the root-mean-square error in {@code y} as a function of {@code x}. This
175   * error is defined as the square root of the mean of the squares of the differences between the
176   * actual {@code y} values of the data and the values predicted by the fit for the {@code x}
177   * values (i.e. it is the square root of the mean of the squares of the vertical distances between
178   * the data points and the best fit line). For this fit, this error is a fraction
179   * {@code sqrt(1 - R*R)} of the population standard deviation of {@code y}, where {@code R} is the
180   * Pearson's correlation coefficient (as given by {@link #pearsonsCorrelationCoefficient()}).
181   *
182   * <p>The corresponding root-mean-square error in {@code x} as a function of {@code y} is a
183   * fraction {@code sqrt(1/(R*R) - 1)} of the population standard deviation of {@code x}. This fit
184   * does not normally minimize that error: to do that, you should swap the roles of {@code x} and
185   * {@code y}.
186   *
187   * <h3>Non-finite values</h3>
188   *
189   * <p>If the dataset contains any non-finite values ({@link Double#POSITIVE_INFINITY},
190   * {@link Double#NEGATIVE_INFINITY}, or {@link Double#NaN}) then the result is
191   * {@link LinearTransformation#forNaN()}.
192   *
193   * @throws IllegalStateException if the dataset is empty or contains a single pair of values, or
194   *     both the {@code x} and {@code y} dataset have zero population variance
195   */
196  public final LinearTransformation leastSquaresFit() {
197    checkState(count() > 1);
198    if (isNaN(sumOfProductsOfDeltas)) {
199      return LinearTransformation.forNaN();
200    }
201    double xSumOfSquaresOfDeltas = xStats.sumOfSquaresOfDeltas();
202    if (xSumOfSquaresOfDeltas > 0.0) {
203      if (yStats.sumOfSquaresOfDeltas() > 0.0) {
204        return LinearTransformation.mapping(xStats.mean(), yStats.mean())
205            .withSlope(sumOfProductsOfDeltas / xSumOfSquaresOfDeltas);
206      } else {
207        return LinearTransformation.horizontal(yStats.mean());
208      }
209    } else {
210      checkState(yStats.sumOfSquaresOfDeltas() > 0.0);
211      return LinearTransformation.vertical(xStats.mean());
212    }
213  }
214
215  private double ensurePositive(double value) {
216    if (value > 0.0) {
217      return value;
218    } else {
219      return Double.MIN_VALUE;
220    }
221  }
222
223  private static double ensureInUnitRange(double value) {
224    if (value >= 1.0) {
225      return 1.0;
226    }
227    if (value <= -1.0) {
228      return -1.0;
229    }
230    return value;
231  }
232}
