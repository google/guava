015package com.google.common.util.concurrent;
016
017import com.google.common.annotations.Beta;
018import com.google.common.annotations.GwtIncompatible;
019
020import java.util.concurrent.CancellationException;
021import java.util.concurrent.ExecutionException;
022import java.util.concurrent.TimeUnit;
023import java.util.concurrent.TimeoutException;
024
025/**
026 * A delegating wrapper around a {@link ListenableFuture} that adds support for the {@link
027 * #checkedGet()} and {@link #checkedGet(long, TimeUnit)} methods.
028 *
029 * @author Sven Mawson
030 * @since 1.0
031 */
032@Beta
033@GwtIncompatible
034public abstract class AbstractCheckedFuture<V, X extends Exception>
035    extends ForwardingListenableFuture.SimpleForwardingListenableFuture<V>
036    implements CheckedFuture<V, X> {
037  /**
038   * Constructs an {@code AbstractCheckedFuture} that wraps a delegate.
039   */
040  protected AbstractCheckedFuture(ListenableFuture<V> delegate) {
041    super(delegate);
042  }
043
044  /**
045   * Translates from an {@link InterruptedException}, {@link CancellationException} or {@link
046   * ExecutionException} thrown by {@code get} to an exception of type {@code X} to be thrown by
047   * {@code checkedGet}. Subclasses must implement this method.
048   *
049   * <p>If {@code e} is an {@code InterruptedException}, the calling {@code checkedGet} method has
050   * already restored the interrupt after catching the exception. If an implementation of {@link
051   * #mapException(Exception)} wishes to swallow the interrupt, it can do so by calling {@link
052   * Thread#interrupted()}.
053   *
054   * <p>Subclasses may choose to throw, rather than return, a subclass of {@code RuntimeException}
055   * to allow creating a CheckedFuture that throws both checked and unchecked exceptions.
056   */
057  protected abstract X mapException(Exception e);
058
059  /**
060   * {@inheritDoc}
061   *
062   * <p>This implementation calls {@link #get()} and maps that method's standard exceptions to
063   * instances of type {@code X} using {@link #mapException}.
064   *
065   * <p>In addition, if {@code get} throws an {@link InterruptedException}, this implementation will
066   * set the current thread's interrupt status before calling {@code mapException}.
067   *
068   * @throws X if {@link #get()} throws an {@link InterruptedException}, {@link
069   *     CancellationException}, or {@link ExecutionException}
070   */
071  @Override
072  public V checkedGet() throws X {
073    try {
074      return get();
075    } catch (InterruptedException e) {
076      Thread.currentThread().interrupt();
077      throw mapException(e);
078    } catch (CancellationException e) {
079      throw mapException(e);
080    } catch (ExecutionException e) {
081      throw mapException(e);
082    }
083  }
084
085  /**
086   * {@inheritDoc}
087   *
088   * <p>This implementation calls {@link #get(long, TimeUnit)} and maps that method's standard
089   * exceptions (excluding {@link TimeoutException}, which is propagated) to instances of type
090   * {@code X} using {@link #mapException}.
091   *
092   * <p>In addition, if {@code get} throws an {@link InterruptedException}, this implementation will
093   * set the current thread's interrupt status before calling {@code mapException}.
094   *
095   * @throws X if {@link #get()} throws an {@link InterruptedException}, {@link
096   *     CancellationException}, or {@link ExecutionException}
097   */
098  @Override
099  public V checkedGet(long timeout, TimeUnit unit) throws TimeoutException, X {
100    try {
101      return get(timeout, unit);
102    } catch (InterruptedException e) {
103      Thread.currentThread().interrupt();
104      throw mapException(e);
105    } catch (CancellationException e) {
106      throw mapException(e);
107    } catch (ExecutionException e) {
108      throw mapException(e);
109    }
110  }
111}
