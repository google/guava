039/**
040 * An object of this class encapsulates type mappings from type variables. Mappings are established
041 * with {@link #where} and types are resolved using {@link #resolveType}.
042 *
043 * <p>Note that usually type mappings are already implied by the static type hierarchy (for example,
044 * the {@code E} type variable declared by class {@code List} naturally maps to {@code String} in
045 * the context of {@code class MyStringList implements List<String>}). In such case, prefer to use
046 * {@link TypeToken#resolveType} since it's simpler and more type safe. This class should only be
047 * used when the type mapping isn't implied by the static type hierarchy, but provided through other
048 * means such as an annotation or external configuration file.
049 *
050 * @author Ben Yu
051 * @since 15.0
052 */
053public final class TypeResolver {
054
055  private final TypeTable typeTable;
056
057  public TypeResolver() {
058    this.typeTable = new TypeTable();
059  }
060
061  private TypeResolver(TypeTable typeTable) {
062    this.typeTable = typeTable;
063  }
064
065  /**
066   * Returns a resolver that resolves types "covariantly".
067   *
068   * <p>For example, when resolving {@code List<T>} in the context of {@code ArrayList<?>}, {@code
069   * <T>} is covariantly resolved to {@code <?>} such that return type of {@code List::get} is
070   * {@code <?>}.
071   */
072  static TypeResolver covariantly(Type contextType) {
073    return new TypeResolver().where(TypeMappingIntrospector.getTypeMappings(contextType));
074  }
075
076  /**
077   * Returns a resolver that resolves types "invariantly".
078   *
079   * <p>For example, when resolving {@code List<T>} in the context of {@code ArrayList<?>}, {@code
080   * <T>} cannot be invariantly resolved to {@code <?>} because otherwise the parameter type of
081   * {@code List::set} will be {@code <?>} and it'll falsely say any object can be passed into
082   * {@code ArrayList<?>::set}.
083   *
084   * <p>Instead, {@code <?>} will be resolved to a capture in the form of a type variable {@code
085   * <capture-of-? extends Object>}, effectively preventing {@code set} from accepting any type.
086   */
087  static TypeResolver invariantly(Type contextType) {
088    Type invariantContext = WildcardCapturer.INSTANCE.capture(contextType);
089    return new TypeResolver().where(TypeMappingIntrospector.getTypeMappings(invariantContext));
090  }
091
092  /**
093   * Returns a new {@code TypeResolver} with type variables in {@code formal} mapping to types in
094   * {@code actual}.
095   *
096   * <p>For example, if {@code formal} is a {@code TypeVariable T}, and {@code actual} is {@code
097   * String.class}, then {@code new TypeResolver().where(formal, actual)} will {@linkplain
098   * #resolveType resolve} {@code ParameterizedType List<T>} to {@code List<String>}, and resolve
099   * {@code Map<T, Something>} to {@code Map<String, Something>} etc. Similarly, {@code formal} and
100   * {@code actual} can be {@code Map<K, V>} and {@code Map<String, Integer>} respectively, or they
101   * can be {@code E[]} and {@code String[]} respectively, or even any arbitrary combination
102   * thereof.
103   *
104   * @param formal The type whose type variables or itself is mapped to other type(s). It's almost
105   *     always a bug if {@code formal} isn't a type variable and contains no type variable. Make
106   *     sure you are passing the two parameters in the right order.
107   * @param actual The type that the formal type variable(s) are mapped to. It can be or contain yet
108   *     other type variables, in which case these type variables will be further resolved if
109   *     corresponding mappings exist in the current {@code TypeResolver} instance.
110   */
111  public TypeResolver where(Type formal, Type actual) {
112    Map<TypeVariableKey, Type> mappings = Maps.newHashMap();
113    populateTypeMappings(mappings, checkNotNull(formal), checkNotNull(actual));
114    return where(mappings);
115  }
116
117  /** Returns a new {@code TypeResolver} with {@code variable} mapping to {@code type}. */
118  TypeResolver where(Map<TypeVariableKey, ? extends Type> mappings) {
119    return new TypeResolver(typeTable.where(mappings));
120  }

122  private static void populateTypeMappings(
123      Map<TypeVariableKey, Type> mappings, Type from, Type to) {
124    if (from.equals(to)) {
125      return;
126    }
127    new TypeVisitor() {
128      @Override
129      void visitTypeVariable(TypeVariable<?> typeVariable) {
130        mappings.put(new TypeVariableKey(typeVariable), to);
131      }
132
133      @Override
134      void visitWildcardType(WildcardType fromWildcardType) {
135        if (!(to instanceof WildcardType)) {
136          return; // okay to say <?> is anything
137        }
138        WildcardType toWildcardType = (WildcardType) to;
139        Type[] fromUpperBounds = fromWildcardType.getUpperBounds();
140        Type[] toUpperBounds = toWildcardType.getUpperBounds();
141        Type[] fromLowerBounds = fromWildcardType.getLowerBounds();
142        Type[] toLowerBounds = toWildcardType.getLowerBounds();
143        checkArgument(
144            fromUpperBounds.length == toUpperBounds.length
145                && fromLowerBounds.length == toLowerBounds.length,
146            "Incompatible type: %s vs. %s",
147            fromWildcardType,
148            to);
149        for (int i = 0; i < fromUpperBounds.length; i++) {
150          populateTypeMappings(mappings, fromUpperBounds[i], toUpperBounds[i]);
151        }
152        for (int i = 0; i < fromLowerBounds.length; i++) {
153          populateTypeMappings(mappings, fromLowerBounds[i], toLowerBounds[i]);
154        }
155      }
156
157      @Override
158      void visitParameterizedType(ParameterizedType fromParameterizedType) {
159        if (to instanceof WildcardType) {
160          return; // Okay to say Foo<A> is <?>
161        }
162        ParameterizedType toParameterizedType = expectArgument(ParameterizedType.class, to);
163        if (fromParameterizedType.getOwnerType() != null
164            && toParameterizedType.getOwnerType() != null) {
165          populateTypeMappings(
166              mappings, fromParameterizedType.getOwnerType(), toParameterizedType.getOwnerType());
167        }
168        checkArgument(
169            fromParameterizedType.getRawType().equals(toParameterizedType.getRawType()),
170            "Inconsistent raw type: %s vs. %s",
171            fromParameterizedType,
172            to);
173        Type[] fromArgs = fromParameterizedType.getActualTypeArguments();
174        Type[] toArgs = toParameterizedType.getActualTypeArguments();
175        checkArgument(
176            fromArgs.length == toArgs.length,
177            "%s not compatible with %s",
178            fromParameterizedType,
179            toParameterizedType);
180        for (int i = 0; i < fromArgs.length; i++) {
181          populateTypeMappings(mappings, fromArgs[i], toArgs[i]);
182        }
183      }
184
185      @Override
186      void visitGenericArrayType(GenericArrayType fromArrayType) {
187        if (to instanceof WildcardType) {
188          return; // Okay to say A[] is <?>
189        }
190        Type componentType = Types.getComponentType(to);
191        checkArgument(componentType != null, "%s is not an array type.", to);
192        populateTypeMappings(mappings, fromArrayType.getGenericComponentType(), componentType);
193      }
194
195      @Override
196      void visitClass(Class<?> fromClass) {
197        if (to instanceof WildcardType) {
198          return; // Okay to say Foo is <?>
199        }
200        // Can't map from a raw class to anything other than itself or a wildcard.
201        // You can't say "assuming String is Integer".
202        // And we don't support "assuming String is T"; user has to say "assuming T is String".
203        throw new IllegalArgumentException("No type mapping from " + fromClass + " to " + to);
204      }
205    }.visit(from);
206  }

208  /**
209   * Resolves all type variables in {@code type} and all downstream types and returns a
210   * corresponding type with type variables resolved.
211   */
212  public Type resolveType(Type type) {
213    checkNotNull(type);
214    if (type instanceof TypeVariable) {
215      return typeTable.resolve((TypeVariable<?>) type);
216    } else if (type instanceof ParameterizedType) {
217      return resolveParameterizedType((ParameterizedType) type);
218    } else if (type instanceof GenericArrayType) {
219      return resolveGenericArrayType((GenericArrayType) type);
220    } else if (type instanceof WildcardType) {
221      return resolveWildcardType((WildcardType) type);
222    } else {
223      // if Class<?>, no resolution needed, we are done.
224      return type;
225    }
226  }
227
228  Type[] resolveTypesInPlace(Type[] types) {
229    for (int i = 0; i < types.length; i++) {
230      types[i] = resolveType(types[i]);
231    }
232    return types;
233  }
234
235  private Type[] resolveTypes(Type[] types) {
236    Type[] result = new Type[types.length];
237    for (int i = 0; i < types.length; i++) {
238      result[i] = resolveType(types[i]);
239    }
240    return result;
241  }
242
243  private WildcardType resolveWildcardType(WildcardType type) {
244    Type[] lowerBounds = type.getLowerBounds();
245    Type[] upperBounds = type.getUpperBounds();
246    return new Types.WildcardTypeImpl(resolveTypes(lowerBounds), resolveTypes(upperBounds));
247  }
248
249  private Type resolveGenericArrayType(GenericArrayType type) {
250    Type componentType = type.getGenericComponentType();
251    Type resolvedComponentType = resolveType(componentType);
252    return Types.newArrayType(resolvedComponentType);
253  }
254
255  private ParameterizedType resolveParameterizedType(ParameterizedType type) {
256    Type owner = type.getOwnerType();
257    Type resolvedOwner = (owner == null) ? null : resolveType(owner);
258    Type resolvedRawType = resolveType(type.getRawType());
259
260    Type[] args = type.getActualTypeArguments();
261    Type[] resolvedArgs = resolveTypes(args);
262    return Types.newParameterizedTypeWithOwner(
263        resolvedOwner, (Class<?>) resolvedRawType, resolvedArgs);
264  }
265
266  private static <T> T expectArgument(Class<T> type, Object arg) {
267    try {
268      return type.cast(arg);
269    } catch (ClassCastException e) {
270      throw new IllegalArgumentException(arg + " is not a " + type.getSimpleName());
271    }
272  }

274  /** A TypeTable maintains mapping from {@link TypeVariable} to types. */
275  private static class TypeTable {
276    private final ImmutableMap<TypeVariableKey, Type> map;
277
278    TypeTable() {
279      this.map = ImmutableMap.of();
280    }
281
282    private TypeTable(ImmutableMap<TypeVariableKey, Type> map) {
283      this.map = map;
284    }
285
286    /** Returns a new {@code TypeResolver} with {@code variable} mapping to {@code type}. */
287    final TypeTable where(Map<TypeVariableKey, ? extends Type> mappings) {
288      ImmutableMap.Builder<TypeVariableKey, Type> builder = ImmutableMap.builder();
289      builder.putAll(map);
290      for (Entry<TypeVariableKey, ? extends Type> mapping : mappings.entrySet()) {
291        TypeVariableKey variable = mapping.getKey();
292        Type type = mapping.getValue();
293        checkArgument(!variable.equalsType(type), "Type variable %s bound to itself", variable);
294        builder.put(variable, type);
295      }
296      return new TypeTable(builder.buildOrThrow());
297    }
298
299    final Type resolve(TypeVariable<?> var) {
300      TypeTable unguarded = this;
301      TypeTable guarded =
302          new TypeTable() {
303            @Override
304            public Type resolveInternal(TypeVariable<?> intermediateVar, TypeTable forDependent) {
305              if (intermediateVar.getGenericDeclaration().equals(var.getGenericDeclaration())) {
306                return intermediateVar;
307              }
308              return unguarded.resolveInternal(intermediateVar, forDependent);
309            }
310          };
311      return resolveInternal(var, guarded);
312    }
313
314    /**
315     * Resolves {@code var} using the encapsulated type mapping. If it maps to yet another
316     * non-reified type or has bounds, {@code forDependants} is used to do further resolution, which
317     * doesn't try to resolve any type variable on generic declarations that are already being
318     * resolved.
319     *
320     * <p>Should only be called and overridden by {@link #resolve(TypeVariable)}.
321     */
322    Type resolveInternal(TypeVariable<?> var, TypeTable forDependants) {
323      Type type = map.get(new TypeVariableKey(var));
324      if (type == null) {
325        Type[] bounds = var.getBounds();
326        if (bounds.length == 0) {
327          return var;
328        }
329        Type[] resolvedBounds = new TypeResolver(forDependants).resolveTypes(bounds);
330        /*
331         * We'd like to simply create our own TypeVariable with the newly resolved bounds. There's
332         * just one problem: Starting with JDK 7u51, the JDK TypeVariable's equals() method doesn't
333         * recognize instances of our TypeVariable implementation. This is a problem because users
334         * compare TypeVariables from the JDK against TypeVariables returned by TypeResolver. To
335         * work with all JDK versions, TypeResolver must return the appropriate TypeVariable
336         * implementation in each of the three possible cases:
337         *
338         * 1. Prior to JDK 7u51, the JDK TypeVariable implementation interoperates with ours.
339         * Therefore, we can always create our own TypeVariable.
340         *
341         * 2. Starting with JDK 7u51, the JDK TypeVariable implementations does not interoperate
342         * with ours. Therefore, we have to be careful about whether we create our own TypeVariable:
343         *
344         * 2a. If the resolved types are identical to the original types, then we can return the
345         * original, identical JDK TypeVariable. By doing so, we sidestep the problem entirely.
346         *
347         * 2b. If the resolved types are different from the original types, things are trickier. The
348         * only way to get a TypeVariable instance for the resolved types is to create our own. The
349         * created TypeVariable will not interoperate with any JDK TypeVariable. But this is OK: We
350         * don't _want_ our new TypeVariable to be equal to the JDK TypeVariable because it has
351         * _different bounds_ than the JDK TypeVariable. And it wouldn't make sense for our new
352         * TypeVariable to be equal to any _other_ JDK TypeVariable, either, because any other JDK
353         * TypeVariable must have a different declaration or name. The only TypeVariable that our
354         * new TypeVariable _will_ be equal to is an equivalent TypeVariable that was also created
355         * by us. And that equality is guaranteed to hold because it doesn't involve the JDK
356         * TypeVariable implementation at all.
357         */
358        if (Types.NativeTypeVariableEquals.NATIVE_TYPE_VARIABLE_ONLY
359            && Arrays.equals(bounds, resolvedBounds)) {
360          return var;
361        }
362        return Types.newArtificialTypeVariable(
363            var.getGenericDeclaration(), var.getName(), resolvedBounds);
364      }
365      // in case the type is yet another type variable.
366      return new TypeResolver(forDependants).resolveType(type);
367    }
368  }

370  private static final class TypeMappingIntrospector extends TypeVisitor {
371
372    private final Map<TypeVariableKey, Type> mappings = Maps.newHashMap();
373
374    /**
375     * Returns type mappings using type parameters and type arguments found in the generic
376     * superclass and the super interfaces of {@code contextClass}.
377     */
378    static ImmutableMap<TypeVariableKey, Type> getTypeMappings(Type contextType) {
379      checkNotNull(contextType);
380      TypeMappingIntrospector introspector = new TypeMappingIntrospector();
381      introspector.visit(contextType);
382      return ImmutableMap.copyOf(introspector.mappings);
383    }
384
385    @Override
386    void visitClass(Class<?> clazz) {
387      visit(clazz.getGenericSuperclass());
388      visit(clazz.getGenericInterfaces());
389    }
390
391    @Override
392    void visitParameterizedType(ParameterizedType parameterizedType) {
393      Class<?> rawClass = (Class<?>) parameterizedType.getRawType();
394      TypeVariable<?>[] vars = rawClass.getTypeParameters();
395      Type[] typeArgs = parameterizedType.getActualTypeArguments();
396      checkState(vars.length == typeArgs.length);
397      for (int i = 0; i < vars.length; i++) {
398        map(new TypeVariableKey(vars[i]), typeArgs[i]);
399      }
400      visit(rawClass);
401      visit(parameterizedType.getOwnerType());
402    }
403
404    @Override
405    void visitTypeVariable(TypeVariable<?> t) {
406      visit(t.getBounds());
407    }
408
409    @Override
410    void visitWildcardType(WildcardType t) {
411      visit(t.getUpperBounds());
412    }
413
414    private void map(TypeVariableKey var, Type arg) {
415      if (mappings.containsKey(var)) {
416        // Mapping already established
417        // This is possible when following both superClass -> enclosingClass
418        // and enclosingclass -> superClass paths.
419        // Since we follow the path of superclass first, enclosing second,
420        // superclass mapping should take precedence.
421        return;
422      }
423      // First, check whether var -> arg forms a cycle
424      for (Type t = arg; t != null; t = mappings.get(TypeVariableKey.forLookup(t))) {
425        if (var.equalsType(t)) {
426          // cycle detected, remove the entire cycle from the mapping so that
427          // each type variable resolves deterministically to itself.
428          // Otherwise, an F -> T cycle will end up resolving both F and T
429          // nondeterministically to either F or T.
430          for (Type x = arg; x != null; x = mappings.remove(TypeVariableKey.forLookup(x))) {}
431          return;
432        }
433      }
434      mappings.put(var, arg);
435    }
436  }

438  // This is needed when resolving types against a context with wildcards
439  // For example:
440  // class Holder<T> {
441  //   void set(T data) {...}
442  // }
443  // Holder<List<?>> should *not* resolve the set() method to set(List<?> data).
444  // Instead, it should create a capture of the wildcard so that set() rejects any List<T>.
445  private static class WildcardCapturer {
446
447    static final WildcardCapturer INSTANCE = new WildcardCapturer();
448
449    private final AtomicInteger id;
450
451    private WildcardCapturer() {
452      this(new AtomicInteger());
453    }
454
455    private WildcardCapturer(AtomicInteger id) {
456      this.id = id;
457    }
458
459    final Type capture(Type type) {
460      checkNotNull(type);
461      if (type instanceof Class) {
462        return type;
463      }
464      if (type instanceof TypeVariable) {
465        return type;
466      }
467      if (type instanceof GenericArrayType) {
468        GenericArrayType arrayType = (GenericArrayType) type;
469        return Types.newArrayType(
470            notForTypeVariable().capture(arrayType.getGenericComponentType()));
471      }
472      if (type instanceof ParameterizedType) {
473        ParameterizedType parameterizedType = (ParameterizedType) type;
474        Class<?> rawType = (Class<?>) parameterizedType.getRawType();
475        TypeVariable<?>[] typeVars = rawType.getTypeParameters();
476        Type[] typeArgs = parameterizedType.getActualTypeArguments();
477        for (int i = 0; i < typeArgs.length; i++) {
478          typeArgs[i] = forTypeVariable(typeVars[i]).capture(typeArgs[i]);
479        }
480        return Types.newParameterizedTypeWithOwner(
481            notForTypeVariable().captureNullable(parameterizedType.getOwnerType()),
482            rawType,
483            typeArgs);
484      }
485      if (type instanceof WildcardType) {
486        WildcardType wildcardType = (WildcardType) type;
487        Type[] lowerBounds = wildcardType.getLowerBounds();
488        if (lowerBounds.length == 0) { // ? extends something changes to capture-of
489          return captureAsTypeVariable(wildcardType.getUpperBounds());
490        } else {
491          // TODO(benyu): handle ? super T somehow.
492          return type;
493        }
494      }
495      throw new AssertionError("must have been one of the known types");
496    }
497
498    TypeVariable<?> captureAsTypeVariable(Type[] upperBounds) {
499      String name =
500          "capture#" + id.incrementAndGet() + "-of ? extends " + Joiner.on('&').join(upperBounds);
501      return Types.newArtificialTypeVariable(WildcardCapturer.class, name, upperBounds);
502    }
503
504    private WildcardCapturer forTypeVariable(TypeVariable<?> typeParam) {
505      return new WildcardCapturer(id) {
506        @Override
507        TypeVariable<?> captureAsTypeVariable(Type[] upperBounds) {
508          Set<Type> combined = new LinkedHashSet<>(asList(upperBounds));
509          // Since this is an artificially generated type variable, we don't bother checking
510          // subtyping between declared type bound and actual type bound. So it's possible that we
511          // may generate something like <capture#1-of ? extends Foo&SubFoo>.
512          // Checking subtype between declared and actual type bounds
513          // adds recursive isSubtypeOf() call and feels complicated.
514          // There is no contract one way or another as long as isSubtypeOf() works as expected.
515          combined.addAll(asList(typeParam.getBounds()));
516          if (combined.size() > 1) { // Object is implicit and only useful if it's the only bound.
517            combined.remove(Object.class);
518          }
519          return super.captureAsTypeVariable(combined.toArray(new Type[0]));
520        }
521      };
522    }
523
524    private WildcardCapturer notForTypeVariable() {
525      return new WildcardCapturer(id);
526    }
527
528    private @Nullable Type captureNullable(@Nullable Type type) {
529      if (type == null) {
530        return null;
531      }
532      return capture(type);
533    }
534  }

536  /**
537   * Wraps around {@code TypeVariable<?>} to ensure that any two type variables are equal as long as
538   * they are declared by the same {@link java.lang.reflect.GenericDeclaration} and have the same
539   * name, even if their bounds differ.
540   *
541   * <p>While resolving a type variable from a {@code var -> type} map, we don't care whether the
542   * type variable's bound has been partially resolved. As long as the type variable "identity"
543   * matches.
544   *
545   * <p>On the other hand, if for example we are resolving {@code List<A extends B>} to {@code
546   * List<A extends String>}, we need to compare that {@code <A extends B>} is unequal to {@code <A
547   * extends String>} in order to decide to use the transformed type instead of the original type.
548   */
549  static final class TypeVariableKey {
550    private final TypeVariable<?> var;
551
552    TypeVariableKey(TypeVariable<?> var) {
553      this.var = checkNotNull(var);
554    }
555
556    @Override
557    public int hashCode() {
558      return Objects.hashCode(var.getGenericDeclaration(), var.getName());
559    }
560
561    @Override
562    public boolean equals(@Nullable Object obj) {
563      if (obj instanceof TypeVariableKey) {
564        TypeVariableKey that = (TypeVariableKey) obj;
565        return equalsTypeVariable(that.var);
566      } else {
567        return false;
568      }
569    }
570
571    @Override
572    public String toString() {
573      return var.toString();
574    }
575
576    /** Wraps {@code t} in a {@code TypeVariableKey} if it's a type variable. */
577    static @Nullable TypeVariableKey forLookup(Type t) {
578      if (t instanceof TypeVariable) {
579        return new TypeVariableKey((TypeVariable<?>) t);
580      } else {
581        return null;
582      }
583    }
584
585    /**
586     * Returns true if {@code type} is a {@code TypeVariable} with the same name and declared by the
587     * same {@code GenericDeclaration}.
588     */
589    boolean equalsType(Type type) {
590      if (type instanceof TypeVariable) {
591        return equalsTypeVariable((TypeVariable<?>) type);
592      } else {
593        return false;
594      }
595    }
596
597    private boolean equalsTypeVariable(TypeVariable<?> that) {
598      return var.getGenericDeclaration().equals(that.getGenericDeclaration())
599          && var.getName().equals(that.getName());
600    }
601  }
