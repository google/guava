017package com.google.common.collect;
018
019import com.google.common.annotations.GwtCompatible;
020import com.google.errorprone.annotations.DoNotCall;
021import java.util.ListIterator;
022import org.jspecify.annotations.Nullable;
023
024/**
025 * A list iterator that does not support {@link #remove}, {@link #add}, or {@link #set}.
026 *
027 * @since 7.0
028 * @author Louis Wasserman
029 */
030@GwtCompatible
031public abstract class UnmodifiableListIterator<E extends @Nullable Object>
032    extends UnmodifiableIterator<E> implements ListIterator<E> {
033  /** Constructor for use by subclasses. */
034  protected UnmodifiableListIterator() {}
035
036  /**
037   * Guaranteed to throw an exception and leave the underlying data unmodified.
038   *
039   * @throws UnsupportedOperationException always
040   * @deprecated Unsupported operation.
041   */
042  @Deprecated
043  @Override
044  @DoNotCall("Always throws UnsupportedOperationException")
045  public final void add(@ParametricNullness E e) {
046    throw new UnsupportedOperationException();
047  }
048
049  /**
050   * Guaranteed to throw an exception and leave the underlying data unmodified.
051   *
052   * @throws UnsupportedOperationException always
053   * @deprecated Unsupported operation.
054   */
055  @Deprecated
056  @Override
057  @DoNotCall("Always throws UnsupportedOperationException")
058  public final void set(@ParametricNullness E e) {
059    throw new UnsupportedOperationException();
060  }
061}
