017package com.google.common.collect;
018
019import com.google.common.annotations.Beta;
020import com.google.common.annotations.GwtIncompatible;
021import com.google.errorprone.annotations.CanIgnoreReturnValue;
022import com.google.errorprone.annotations.DoNotMock;
023
024/**
025 * Provides equivalent behavior to {@link String#intern} for other immutable types. Common
026 * implementations are available from the {@link Interners} class.
027 *
028 * @author Kevin Bourrillion
029 * @since 3.0
030 */
031@Beta
032@DoNotMock("Use Interners.new*Interner")
033@GwtIncompatible
034public interface Interner<E> {
035  /**
036   * Chooses and returns the representative instance for any of a collection of instances that are
037   * equal to each other. If two {@linkplain Object#equals equal} inputs are given to this method,
038   * both calls will return the same instance. That is, {@code intern(a).equals(a)} always holds,
039   * and {@code intern(a) == intern(b)} if and only if {@code a.equals(b)}. Note that {@code
040   * intern(a)} is permitted to return one instance now and a different instance later if the
041   * original interned instance was garbage-collected.
042   *
043   * <p><b>Warning:</b> do not use with mutable objects.
044   *
045   * @throws NullPointerException if {@code sample} is null
046   */
047  @CanIgnoreReturnValue // TODO(cpovirk): Consider removing this?
048  E intern(E sample);
049}
