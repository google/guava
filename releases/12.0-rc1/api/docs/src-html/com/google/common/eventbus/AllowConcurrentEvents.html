017    package com.google.common.eventbus;
018    
019    import com.google.common.annotations.Beta;
020    
021    import java.lang.annotation.ElementType;
022    import java.lang.annotation.Retention;
023    import java.lang.annotation.RetentionPolicy;
024    import java.lang.annotation.Target;
025    
026    /**
027     * Marks an event handling method as being thread-safe.  This annotation
028     * indicates that EventBus may invoke the event handler simultaneously from
029     * multiple threads.
030     *
031     * <p>This does not mark the method as an event handler, and so should be used
032     * in combination with {@link Subscribe}.
033     *
034     * @author Cliff Biffle
035     * @since 10.0
036     */
037    @Retention(RetentionPolicy.RUNTIME)
038    @Target(ElementType.METHOD)
039    @Beta
040    public @interface AllowConcurrentEvents {
041    }
