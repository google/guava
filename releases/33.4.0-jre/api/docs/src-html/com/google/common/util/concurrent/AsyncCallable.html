015package com.google.common.util.concurrent;
016
017import com.google.common.annotations.GwtCompatible;
018import java.util.concurrent.Future;
019import org.checkerframework.checker.nullness.qual.Nullable;
020
021/**
022 * Computes a value, possibly asynchronously. For an example usage and more information, see {@link
023 * Futures.FutureCombiner#callAsync(AsyncCallable, java.util.concurrent.Executor)}.
024 *
025 * <p>Much like {@link java.util.concurrent.Callable}, but returning a {@link ListenableFuture}
026 * result.
027 *
028 * @since 20.0
029 */
030@FunctionalInterface
031@GwtCompatible
032@ElementTypesAreNonnullByDefault
033public interface AsyncCallable<V extends @Nullable Object> {
034  /**
035   * Computes a result {@code Future}. The output {@code Future} need not be {@linkplain
036   * Future#isDone done}, making {@code AsyncCallable} suitable for asynchronous derivations.
037   *
038   * <p>Throwing an exception from this method is equivalent to returning a failing {@link
039   * ListenableFuture}.
040   */
041  ListenableFuture<V> call() throws Exception;
042}
