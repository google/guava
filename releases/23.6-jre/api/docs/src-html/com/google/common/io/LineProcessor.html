015package com.google.common.io;
016
017import com.google.common.annotations.Beta;
018import com.google.common.annotations.GwtIncompatible;
019import com.google.errorprone.annotations.CanIgnoreReturnValue;
020import java.io.IOException;
021
022/**
023 * A callback to be used with the streaming {@code readLines} methods.
024 *
025 * <p>{@link #processLine} will be called for each line that is read, and should return {@code
026 * false} when you want to stop processing.
027 *
028 * @author Miles Barr
029 * @since 1.0
030 */
031@Beta
032@GwtIncompatible
033public interface LineProcessor<T> {
034
035  /**
036   * This method will be called once for each line.
037   *
038   * @param line the line read from the input, without delimiter
039   * @return true to continue processing, false to stop
040   */
041  @CanIgnoreReturnValue // some uses know that their processor never returns false
042  boolean processLine(String line) throws IOException;
043
044  /** Return the result of processing all the lines. */
045  T getResult();
046}
