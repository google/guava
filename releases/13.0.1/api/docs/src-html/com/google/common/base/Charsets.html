017    package com.google.common.base;
018    
019    import com.google.common.annotations.GwtCompatible;
020    import com.google.common.annotations.GwtIncompatible;
021    
022    import java.nio.charset.Charset;
023    
024    /**
025     * Contains constant definitions for the six standard {@link Charset} instances, which are
026     * guaranteed to be supported by all Java platform implementations.
027     *
028     * <p>Assuming you're free to choose, note that <b>{@link #UTF_8} is widely preferred</b>.
029     *
030     * <p>See the Guava User Guide article on <a
031     * href="http://code.google.com/p/guava-libraries/wiki/StringsExplained#Charsets">
032     * {@code Charsets}</a>.
033     *
034     * @author Mike Bostock
035     * @since 1.0
036     */
037    @GwtCompatible(emulated = true)
038    public final class Charsets {
039      private Charsets() {}
040    
041      /**
042       * US-ASCII: seven-bit ASCII, the Basic Latin block of the Unicode character set (ISO646-US).
043       */
044      @GwtIncompatible("Non-UTF-8 Charset")
045      public static final Charset US_ASCII = Charset.forName("US-ASCII");
046    
047      /**
048       * ISO-8859-1: ISO Latin Alphabet Number 1 (ISO-LATIN-1).
049       */
050      @GwtIncompatible("Non-UTF-8 Charset")
051      public static final Charset ISO_8859_1 = Charset.forName("ISO-8859-1");
052    
053      /**
054       * UTF-8: eight-bit UCS Transformation Format.
055       */
056      public static final Charset UTF_8 = Charset.forName("UTF-8");
057    
058      /**
059       * UTF-16BE: sixteen-bit UCS Transformation Format, big-endian byte order.
060       */
061      @GwtIncompatible("Non-UTF-8 Charset")
062      public static final Charset UTF_16BE = Charset.forName("UTF-16BE");
063    
064      /**
065       * UTF-16LE: sixteen-bit UCS Transformation Format, little-endian byte order.
066       */
067      @GwtIncompatible("Non-UTF-8 Charset")
068      public static final Charset UTF_16LE = Charset.forName("UTF-16LE");
069    
070      /**
071       * UTF-16: sixteen-bit UCS Transformation Format, byte order identified by an optional byte-order
072       * mark.
073       */
074      @GwtIncompatible("Non-UTF-8 Charset")
075      public static final Charset UTF_16 = Charset.forName("UTF-16");
076    
077      /*
078       * Please do not add new Charset references to this class, unless those character encodings are
079       * part of the set required to be supported by all Java platform implementations! Any Charsets
080       * initialized here may cause unexpected delays when this class is loaded. See the Charset
081       * Javadocs for the list of built-in character encodings.
082       */
083    }
