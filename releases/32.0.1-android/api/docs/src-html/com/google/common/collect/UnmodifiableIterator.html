017package com.google.common.collect;
018
019import com.google.common.annotations.GwtCompatible;
020import com.google.errorprone.annotations.DoNotCall;
021import java.util.Iterator;
022import org.checkerframework.checker.nullness.qual.Nullable;
023
024/**
025 * An iterator that does not support {@link #remove}.
026 *
027 * <p>{@code UnmodifiableIterator} is used primarily in conjunction with implementations of {@link
028 * ImmutableCollection}, such as {@link ImmutableList}. You can, however, convert an existing
029 * iterator to an {@code UnmodifiableIterator} using {@link Iterators#unmodifiableIterator}.
030 *
031 * @author Jared Levy
032 * @since 2.0
033 */
034@GwtCompatible
035@ElementTypesAreNonnullByDefault
036public abstract class UnmodifiableIterator<E extends @Nullable Object> implements Iterator<E> {
037  /** Constructor for use by subclasses. */
038  protected UnmodifiableIterator() {}
039
040  /**
041   * Guaranteed to throw an exception and leave the underlying data unmodified.
042   *
043   * @throws UnsupportedOperationException always
044   * @deprecated Unsupported operation.
045   */
046  @Deprecated
047  @Override
048  @DoNotCall("Always throws UnsupportedOperationException")
049  public final void remove() {
050    throw new UnsupportedOperationException();
051  }
052}
