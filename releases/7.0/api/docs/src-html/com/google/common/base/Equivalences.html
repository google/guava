015    package com.google.common.base;
016    
017    import static com.google.common.base.Preconditions.checkNotNull;
018    
019    import com.google.common.annotations.Beta;
020    import com.google.common.annotations.GwtCompatible;
021    
022    import javax.annotation.Nullable;
023    
024    /**
025     * Contains static factory methods for creating {@code Equivalence} instances.
026     *
027     * <p>All methods returns serializable instances.
028     *
029     * @author Bob Lee
030     * @since 4
031     */
032    @Beta
033    @GwtCompatible
034    public final class Equivalences {
035      private Equivalences() {}
036    
037      /**
038       * Returns an equivalence that delegates to {@link Object#equals} and {@link Object#hashCode}.
039       * Does not support null values.
040       */
041      public static Equivalence<Object> equals() {
042        return Impl.EQUALS;
043      }
044    
045      /**
046       * Returns an equivalence that delegates to {@link Object#equals} and {@link Object#hashCode}.
047       * {@link Equivalence#equivalent} returns {@code true} if both values are null, or if neither
048       * value is null and {@link Object#equals} returns {@code true}. {@link Equivalence#hash} returns
049       * {@code 0} if passed a null value.
050       */
051      public static Equivalence<Object> nullAwareEquals() {
052        return Impl.NULL_AWARE_EQUALS;
053      }
054    
055      /**
056       * Returns an equivalence that uses {@code ==} to compare values and {@link
057       * System#identityHashCode(Object)} to compute the hash code.  {@link Equivalence#equivalent}
058       * returns {@code true} if both values are null, or if neither value is null and {@code ==}
059       * returns {@code true}. {@link Equivalence#hash} throws a {@link NullPointerException} if passed
060       * a null value.
061       */
062      public static Equivalence<Object> identity() {
063        return Impl.IDENTITY;
064      }
065    
066      private enum Impl implements Equivalence<Object> {
067        EQUALS {
068          public boolean equivalent(Object a, @Nullable Object b) {
069            checkNotNull(a);  // for GWT.
070            return a.equals(b);
071          }
072    
073          public int hash(Object o) {
074            checkNotNull(o);  // for GWT.
075            return o.hashCode();
076          }
077        },
078        IDENTITY {
079          public boolean equivalent(Object a, @Nullable Object b) {
080            return checkNotNull(a) == b;
081          }
082    
083          public int hash(@Nullable Object o) {
084            return System.identityHashCode(o);
085          }
086        },
087        NULL_AWARE_EQUALS {
088          public boolean equivalent(@Nullable Object a, @Nullable Object b) {
089            return Objects.equal(a, b);
090          }
091    
092          public int hash(@Nullable Object o) {
093            return (o == null) ? 0 : o.hashCode();
094          }
095        },
096      }
097    }
