052      /**
053       * Returns an uninterruptible view of a {@code Future}. If a thread is
054       * interrupted during an attempt to {@code get()} from the returned future, it
055       * continues to wait on the result until it is available or the timeout
056       * elapses, and only then re-interrupts the thread.
057       */
058      public static <V> UninterruptibleFuture<V> makeUninterruptible(
059          final Future<V> future) {
060        checkNotNull(future);
061        if (future instanceof UninterruptibleFuture<?>) {
062          return (UninterruptibleFuture<V>) future;
063        }
064        return new UninterruptibleFuture<V>() {
065          public boolean cancel(boolean mayInterruptIfRunning) {
066            return future.cancel(mayInterruptIfRunning);
067          }
068          public boolean isCancelled() {
069            return future.isCancelled();
070          }
071          public boolean isDone() {
072            return future.isDone();
073          }
074    
075          public V get(long timeoutDuration, TimeUnit timeoutUnit)
076              throws TimeoutException, ExecutionException {
077            boolean interrupted = false;
078            try {
079              long timeoutNanos = timeoutUnit.toNanos(timeoutDuration);
080              long end = System.nanoTime() + timeoutNanos;
081              while (true) {
082                try {
083                  return future.get(timeoutNanos, NANOSECONDS);
084                } catch (InterruptedException e) {
085                  // Future treats negative timeouts just like zero.
086                  timeoutNanos = end - System.nanoTime();
087                  interrupted = true;
088                }
089              }
090            } finally {
091              if (interrupted) {
092                Thread.currentThread().interrupt();
093              }
094            }
095          }
096    
097          public V get() throws ExecutionException {
098            boolean interrupted = false;
099            try {
100              while (true) {
101                try {
102                  return future.get();
103                } catch (InterruptedException ignored) {
104                  interrupted = true;
105                }
106              }
107            } finally {
108              if (interrupted) {
109                Thread.currentThread().interrupt();
110              }
111            }
112          }
113        };
114      }

118       * returned future will create a thread to wait for the source future to

120       *

134       * Creates a {@link CheckedFuture} out of a normal {@link Future} and a
135       * {@link Function} that maps from {@link Exception} instances into the
136       * appropriate checked type.
137       *
138       * <p>The given mapping function will be applied to an
139       * {@link InterruptedException}, a {@link CancellationException}, or an

143      public static <V, X extends Exception> CheckedFuture<V, X> makeChecked(
144          Future<V> future, Function<Exception, X> mapper) {
145        return new MappingCheckedFuture<V, X>(makeListenable(future), mapper);

149       * Creates a {@code ListenableFuture} which has its value set immediately upon
150       * construction. The getters just return the value. This {@code Future} can't
151       * be canceled or timed out and its {@code isDone()} method always returns
152       * {@code true}. It's useful for returning something that implements the
153       * {@code ListenableFuture} interface but already has the result.
154       */
155      public static <V> ListenableFuture<V> immediateFuture(@Nullable V value) {
156        ValueFuture<V> future = ValueFuture.create();
157        future.set(value);
158        return future;

162       * Returns a {@code CheckedFuture} which has its value set immediately upon
163       * construction.
164       *
165       * <p>The returned {@code Future} can't be cancelled, and its {@code isDone()}
166       * method always returns {@code true}. Calling {@code get()} or {@code
167       * checkedGet()} will immediately return the provided value.
168       */
169      public static <V, X extends Exception> CheckedFuture<V, X>
170          immediateCheckedFuture(@Nullable V value) {
171        ValueFuture<V> future = ValueFuture.create();
172        future.set(value);
173        return Futures.makeChecked(future, new Function<Exception, X>() {
174          public X apply(Exception e) {
175            throw new AssertionError("impossible");

200       * Returns a {@code CheckedFuture} which has an exception set immediately upon

202       *
203       * <p>The returned {@code Future} can't be cancelled, and its {@code isDone()}
204       * method always returns {@code true}. Calling {@code get()} will immediately

206       * ExecutionException}, and calling {@code checkedGet()} will throw the

222      /**
223       * Returns a new {@code ListenableFuture} whose result is asynchronously
224       * derived from the result of the given {@code Future}. More precisely, the
225       * returned {@code Future} takes its result from a {@code Future} produced by
226       * applying the given {@code Function} to the result of the original {@code
227       * Future}.
228       *
229       * <p>Successful cancellation of either the input future or the result of
230       * function application will cause the returned future to be cancelled.
231       * Cancelling the returned future will succeed if it is currently running.
232       * In this case, attempts will be made to cancel the input future and the
233       * result of the function, however there is no guarantee of success.
234       *
235       * <p>TODO: Add a version that accepts a normal {@code Future}
236       *
237       * <p>The typical use for this method would be when a RPC call is dependent on
238       * the results of another RPC.  One would call the first RPC (input), create a
239       * function that calls another RPC based on input's result, and then call
240       * chain on input and that function to get a {@code ListenableFuture} of
241       * the result.
242       *
243       * @param input The future to chain
244       * @param function A function to chain the results of the provided future
245       *     to the results of the returned future.  This will be run in the thread
246       *     that notifies input it is complete.
247       * @return A future that holds result of the chain.
248       */
249      public static <I, O> ListenableFuture<O> chain(ListenableFuture<I> input,
250          Function<? super I, ? extends ListenableFuture<? extends O>> function) {
251        return chain(input, function, MoreExecutors.sameThreadExecutor());
252      }

255       * Returns a new {@code ListenableFuture} whose result is asynchronously
256       * derived from the result of the given {@code Future}. More precisely, the
257       * returned {@code Future} takes its result from a {@code Future} produced by
258       * applying the given {@code Function} to the result of the original {@code
259       * Future}.
260       *
261       * <p>Successful cancellation of either the input future or the result of
262       * function application will cause the returned future to be cancelled.
263       * Cancelling the returned future will succeed if it is currently running.
264       * In this case, attempts will be made to cancel the input future and the
265       * result of the function, however there is no guarantee of success.
266       *
267       * <p>This version allows an arbitrary executor to be passed in for running
268       * the chained Function. When using {@link MoreExecutors#sameThreadExecutor},
269       * the thread chained Function executes in will be whichever thread set the
270       * result of the input Future, which may be the network thread in the case of
271       * RPC-based Futures.
272       *
273       * @param input The future to chain
274       * @param function A function to chain the results of the provided future
275       *     to the results of the returned future.
276       * @param exec Executor to run the function in.
277       * @return A future that holds result of the chain.
278       */
279      public static <I, O> ListenableFuture<O> chain(ListenableFuture<I> input,
280          Function<? super I, ? extends ListenableFuture<? extends O>> function,
281          Executor exec) {
282        ChainingListenableFuture<I, O> chain =
283            new ChainingListenableFuture<I, O>(function, input);
284        input.addListener(chain, exec);
285        return chain;
286      }

291       * Future}.

293       * <p>Successful cancellation of the input future will cause the returned
294       * future to be cancelled.  Cancelling the returned future will succeed if it

296       * input future, however there is no guarantee of success.
297       *
298       * <p>An example use of this method is to convert a serializable object
299       * returned from an RPC into a POJO.
300       *
301       * @param future The future to compose

304       *     that notifies input it is complete.

312      /**
313       * Returns a new {@code ListenableFuture} whose result is the product of
314       * applying the given {@code Function} to the result of the given {@code
315       * Future}.
316       *
317       * <p>Successful cancellation of the input future will cause the returned
318       * future to be cancelled.  Cancelling the returned future will succeed if it
319       * is currently running.  In this case, an attempt will be made to cancel the
320       * input future, however there is no guarantee of success.
321       *
322       * <p>An example use of this method is to convert a serializable object
323       * returned from an RPC into a POJO.
324       *
325       * <p>This version allows an arbitrary executor to be passed in for running
326       * the chained Function. When using {@link MoreExecutors#sameThreadExecutor},
327       * the thread chained Function executes in will be whichever thread set the
328       * result of the input Future, which may be the network thread in the case of
329       * RPC-based Futures.
330       *
331       * @param future The future to compose
332       * @param function A Function to compose the results of the provided future
333       *     to the results of the returned future.
334       * @param exec Executor to run the function in.
335       * @return A future that holds result of the composition.
336       * @since 2
337       */
338      public static <I, O> ListenableFuture<O> compose(ListenableFuture<I> future,
339          final Function<? super I, ? extends O> function, Executor exec) {
340        checkNotNull(function);
341        Function<I, ListenableFuture<O>> wrapperFunction
342            = new Function<I, ListenableFuture<O>>() {
343                @Override public ListenableFuture<O> apply(I input) {
344                  O output = function.apply(input);
345                  return immediateFuture(output);
346                }
347            };
348        return chain(future, wrapperFunction, exec);
349      }

351      /**
352       * Returns a new {@code Future} whose result is the product of applying the
353       * given {@code Function} to the result of the given {@code Future}.
354       *
355       * <p>An example use of this method is to convert a Future that produces a
356       * handle to an object to a future that produces the object itself.
357       *
358       * <p>Each call to {@code Future<O>.get(*)} results in a call to
359       * {@code Future<I>.get(*)}, but {@code function} is only applied once, so it
360       * is assumed that {@code Future<I>.get(*)} is idempotent.
361       *
362       * <p>When calling {@link Future#get(long, TimeUnit)} on the returned
363       * future, the timeout only applies to the future passed in to this method.
364       * Any additional time taken by applying {@code function} is not considered.
365       * (Exception: If the input future is a {@link ListenableFuture}, timeouts
366       * will be strictly enforced.)
367       *
368       * @param future The future to compose
369       * @param function A Function to compose the results of the provided future
370       *     to the results of the returned future.  This will be run in the thread
371       *     that calls one of the varieties of {@code get()}.
372       * @return A future that computes result of the composition.
373       */
374      public static <I, O> Future<O> compose(final Future<I> future,
375          final Function<? super I, ? extends O> function) {
376        if (future instanceof ListenableFuture) {
377          return compose((ListenableFuture<I>) future, function);
378        }
379        checkNotNull(future);
380        checkNotNull(function);
381        return new Future<O>() {
382    
383          /*
384           * Concurrency detail:
385           *
386           * <p>To preserve the idempotency of calls to this.get(*) calls to the
387           * function are only applied once. A lock is required to prevent multiple
388           * applications of the function. The calls to future.get(*) are performed
389           * outside the lock, as is required to prevent calls to
390           * get(long, TimeUnit) to persist beyond their timeout.
391           *
392           * <p>Calls to future.get(*) on every call to this.get(*) also provide
393           * the cancellation behavior for this.
394           *
395           * <p>(Consider: in thread A, call get(), in thread B call get(long,
396           * TimeUnit). Thread B may have to wait for Thread A to finish, which
397           * would be unacceptable.)
398           *
399           * <p>Note that each call to Future<O>.get(*) results in a call to
400           * Future<I>.get(*), but the function is only applied once, so
401           * Future<I>.get(*) is assumed to be idempotent.
402           */
403    
404          private final Object lock = new Object();
405          private boolean set = false;
406          private O value = null;
407          private ExecutionException exception = null;
408    
409          @Override
410          public O get() throws InterruptedException, ExecutionException {
411            return apply(future.get());
412          }
413    
414          @Override
415          public O get(long timeout, TimeUnit unit) throws InterruptedException,
416              ExecutionException, TimeoutException {
417            return apply(future.get(timeout, unit));
418          }
419    
420          private O apply(I raw) throws ExecutionException {
421            synchronized (lock) {
422              if (!set) {
423                try {
424                  value = function.apply(raw);
425                } catch (RuntimeException e) {
426                  exception = new ExecutionException(e);
427                } catch (Error e) {
428                  exception = new ExecutionException(e);
429                }
430                set = true;
431              }
432    
433              if (exception != null) {
434                throw exception;
435              }
436              return value;
437            }
438          }
439    
440          @Override
441          public boolean cancel(boolean mayInterruptIfRunning) {
442            return future.cancel(mayInterruptIfRunning);
443          }
444    
445          @Override
446          public boolean isCancelled() {
447            return future.isCancelled();
448          }
449    
450          @Override
451          public boolean isDone() {
452            return future.isDone();
453          }
454        };
455      }

457      /**
458       * An implementation of {@code ListenableFuture} that also implements
459       * {@code Runnable} so that it can be used to nest ListenableFutures.
460       * Once the passed-in {@code ListenableFuture} is complete, it calls the
461       * passed-in {@code Function} to generate the result.
462       *
463       * <p>If the function throws any checked exceptions, they should be wrapped
464       * in a {@code UndeclaredThrowableException} so that this class can get
465       * access to the cause.
466       */
467      private static class ChainingListenableFuture<I, O>
468          extends AbstractListenableFuture<O> implements Runnable {
469    
470        private Function<? super I, ? extends ListenableFuture<? extends O>>
471            function;
472        private ListenableFuture<? extends I> inputFuture;
473        private volatile ListenableFuture<? extends O> outputFuture;
474        private final BlockingQueue<Boolean> mayInterruptIfRunningChannel =
475            new LinkedBlockingQueue<Boolean>(1);
476        private final CountDownLatch outputCreated = new CountDownLatch(1);
477    
478        private ChainingListenableFuture(
479            Function<? super I, ? extends ListenableFuture<? extends O>> function,
480            ListenableFuture<? extends I> inputFuture) {
481          this.function = checkNotNull(function);
482          this.inputFuture = checkNotNull(inputFuture);
483        }
484    
485        /**
486         * Delegate the get() to the input and output futures, in case
487         * their implementations defer starting computation until their
488         * own get() is invoked.
489         */
490        public O get() throws InterruptedException, ExecutionException {
491          if (!isDone()) {
492            // Invoking get on the inputFuture will ensure our own run()
493            // method below is invoked as a listener when inputFuture sets
494            // its value.  Therefore when get() returns we should then see
495            // the outputFuture be created.
496            ListenableFuture<? extends I> inputFuture = this.inputFuture;
497            if (inputFuture != null) {
498              inputFuture.get();
499            }
500    
501            // If our listener was scheduled to run on an executor we may
502            // need to wait for our listener to finish running before the
503            // outputFuture has been constructed by the function.
504            outputCreated.await();
505    
506            // Like above with the inputFuture, we have a listener on
507            // the outputFuture that will set our own value when its
508            // value is set.  Invoking get will ensure the output can
509            // complete and invoke our listener, so that we can later
510            // get the result.
511            ListenableFuture<? extends O> outputFuture = this.outputFuture;
512            if (outputFuture != null) {
513              outputFuture.get();
514            }
515          }
516          return super.get();
517        }
518    
519        /**
520         * Delegate the get() to the input and output futures, in case
521         * their implementations defer starting computation until their
522         * own get() is invoked.
523         */
524        public O get(long timeout, TimeUnit unit) throws TimeoutException,
525            ExecutionException, InterruptedException {
526          if (!isDone()) {
527            // Use a single time unit so we can decrease remaining timeout
528            // as we wait for various phases to complete.
529            if (unit != NANOSECONDS) {
530              timeout = NANOSECONDS.convert(timeout, unit);
531              unit = NANOSECONDS;
532            }
533    
534            // Invoking get on the inputFuture will ensure our own run()
535            // method below is invoked as a listener when inputFuture sets
536            // its value.  Therefore when get() returns we should then see
537            // the outputFuture be created.
538            ListenableFuture<? extends I> inputFuture = this.inputFuture;
539            if (inputFuture != null) {
540              long start = System.nanoTime();
541              inputFuture.get(timeout, unit);
542              timeout -= Math.max(0, System.nanoTime() - start);
543            }
544    
545            // If our listener was scheduled to run on an executor we may
546            // need to wait for our listener to finish running before the
547            // outputFuture has been constructed by the function.
548            long start = System.nanoTime();
549            if (!outputCreated.await(timeout, unit)) {
550              throw new TimeoutException();
551            }
552            timeout -= Math.max(0, System.nanoTime() - start);
553    
554            // Like above with the inputFuture, we have a listener on
555            // the outputFuture that will set our own value when its
556            // value is set.  Invoking get will ensure the output can
557            // complete and invoke our listener, so that we can later
558            // get the result.
559            ListenableFuture<? extends O> outputFuture = this.outputFuture;
560            if (outputFuture != null) {
561              outputFuture.get(timeout, unit);
562            }
563          }
564          return super.get(timeout, unit);
565        }
566    
567        @Override
568        public boolean cancel(boolean mayInterruptIfRunning) {
569          if (cancel()) {
570            try {
571              // This should never block since only one thread is allowed to cancel
572              // this Future.
573              mayInterruptIfRunningChannel.put(mayInterruptIfRunning);
574            } catch (InterruptedException ignored) {
575              Thread.currentThread().interrupt();
576            }
577            cancel(inputFuture, mayInterruptIfRunning);
578            cancel(outputFuture, mayInterruptIfRunning);
579            return true;
580          }
581          return false;
582        }
583    
584        private void cancel(@Nullable Future<?> future,
585            boolean mayInterruptIfRunning) {
586          if (future != null) {
587            future.cancel(mayInterruptIfRunning);
588          }
589        }
590    
591        public void run() {
592          try {
593            I sourceResult;
594            try {
595              sourceResult = makeUninterruptible(inputFuture).get();
596            } catch (CancellationException e) {
597              // Cancel this future and return.
598              cancel();
599              return;
600            } catch (ExecutionException e) {
601              // Set the cause of the exception as this future's exception
602              setException(e.getCause());
603              return;
604            }
605    
606            final ListenableFuture<? extends O> outputFuture = this.outputFuture =
607                function.apply(sourceResult);
608            if (isCancelled()) {
609              // Handles the case where cancel was called while the function was
610              // being applied.
611              try {
612                // There is a gap in cancel(boolean) between calling cancel() and
613                // storing the value of mayInterruptIfRunning, so this thread needs
614                // to block, waiting for that value.
615                outputFuture.cancel(mayInterruptIfRunningChannel.take());
616              } catch (InterruptedException ignored) {
617                Thread.currentThread().interrupt();
618              }
619              this.outputFuture = null;
620              return;
621            }
622            outputFuture.addListener(new Runnable() {
623                public void run() {
624                  try {
625                    // Here it would have been nice to have had an
626                    // UninterruptibleListenableFuture, but we don't want to start a
627                    // combinatorial explosion of interfaces, so we have to make do.
628                    set(makeUninterruptible(outputFuture).get());
629                  } catch (CancellationException e) {
630                    // Cancel this future and return.
631                    cancel();
632                    return;
633                  } catch (ExecutionException e) {
634                    // Set the cause of the exception as this future's exception
635                    setException(e.getCause());
636                  } finally {
637                    // Don't pin inputs beyond completion
638                    ChainingListenableFuture.this.outputFuture = null;
639                  }
640                }
641              }, MoreExecutors.sameThreadExecutor());
642          } catch (UndeclaredThrowableException e) {
643            // Set the cause of the exception as this future's exception
644            setException(e.getCause());
645          } catch (RuntimeException e) {
646            // This exception is irrelevant in this thread, but useful for the
647            // client
648            setException(e);
649          } catch (Error e) {
650            // Propagate errors up ASAP - our superclass will rethrow the error
651            setException(e);
652          } finally {
653            // Don't pin inputs beyond completion
654            function = null;
655            inputFuture = null;
656            // Allow our get routines to examine outputFuture now.
657            outputCreated.countDown();
658          }
659        }
660      }

662      /**
663       * A checked future that uses a function to map from exceptions to the
664       * appropriate checked type.
665       */
666      private static class MappingCheckedFuture<V, X extends Exception> extends
667          AbstractCheckedFuture<V, X> {
668    
669        final Function<Exception, X> mapper;
670    
671        MappingCheckedFuture(ListenableFuture<V> delegate,
672            Function<Exception, X> mapper) {
673          super(delegate);
674    
675          this.mapper = checkNotNull(mapper);
676        }
677    
678        @Override
679        protected X mapException(Exception e) {
680          return mapper.apply(e);
681        }
682      }
683    
684      /**
685       * An adapter to turn a {@link Future} into a {@link ListenableFuture}.  This
686       * will wait on the future to finish, and when it completes, run the
687       * listeners.  This implementation will wait on the source future
688       * indefinitely, so if the source future never completes, the adapter will
689       * never complete either.
690       *
691       * <p>If the delegate future is interrupted or throws an unexpected unchecked
692       * exception, the listeners will not be invoked.
693       */
694      private static class ListenableFutureAdapter<V> extends ForwardingFuture<V>
695          implements ListenableFuture<V> {
696    
697        private static final Executor adapterExecutor =
698            Executors.newCachedThreadPool();
699    
700        // The execution list to hold our listeners.
701        private final ExecutionList executionList = new ExecutionList();
702    
703        // This allows us to only start up a thread waiting on the delegate future
704        // when the first listener is added.
705        private final AtomicBoolean hasListeners = new AtomicBoolean(false);
706    
707        // The delegate future.
708        private final Future<V> delegate;
709    
710        ListenableFutureAdapter(final Future<V> delegate) {
711          this.delegate = checkNotNull(delegate);
712        }
713    
714        @Override
715        protected Future<V> delegate() {
716          return delegate;
717        }
718    
719        @Override
720        public void addListener(Runnable listener, Executor exec) {
721    
722          // When a listener is first added, we run a task that will wait for
723          // the delegate to finish, and when it is done will run the listeners.
724          if (!hasListeners.get() && hasListeners.compareAndSet(false, true)) {
725            adapterExecutor.execute(new Runnable() {
726              @Override
727              public void run() {
728                try {
729                  delegate.get();
730                } catch (CancellationException e) {
731                  // The task was cancelled, so it is done, run the listeners.
732                } catch (InterruptedException e) {
733                  // This thread was interrupted.  This should never happen, so we
734                  // throw an IllegalStateException.
735                  throw new IllegalStateException("Adapter thread interrupted!", e);
736                } catch (ExecutionException e) {
737                  // The task caused an exception, so it is done, run the listeners.
738                }
739                executionList.run();
740              }
741            });
742          }
743          executionList.add(listener, exec);
744        }
745      }
