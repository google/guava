146   * Returns an immutable map instance containing the given entries. Internally, the returned map
147   * will be backed by an {@link EnumMap}.
148   *
149   * <p>The iteration order of the returned map follows the enum's iteration order, not the order in
150   * which the elements appear in the given map.
151   *
152   * @param map the map to make an immutable copy of
153   * @return an immutable map containing those entries
154   * @since 14.0

156  @GwtCompatible(serializable = true)
157  @J2ktIncompatible
158  public static <K extends Enum<K>, V> ImmutableMap<K, V> immutableEnumMap(
159      Map<K, ? extends V> map) {
160    if (map instanceof ImmutableEnumMap) {
161      @SuppressWarnings("unchecked") // safe covariant cast
162      ImmutableEnumMap<K, V> result = (ImmutableEnumMap<K, V>) map;
163      return result;
164    }
165    Iterator<? extends Entry<K, ? extends V>> entryItr = map.entrySet().iterator();
166    if (!entryItr.hasNext()) {
167      return ImmutableMap.of();
168    }
169    Entry<K, ? extends V> entry1 = entryItr.next();
170    K key1 = entry1.getKey();
171    V value1 = entry1.getValue();
172    checkEntryNotNull(key1, value1);
173    Class<K> clazz = key1.getDeclaringClass();
174    EnumMap<K, V> enumMap = new EnumMap<>(clazz);
175    enumMap.put(key1, value1);
176    while (entryItr.hasNext()) {
177      Entry<K, ? extends V> entry = entryItr.next();
178      K key = entry.getKey();
179      V value = entry.getValue();
180      checkEntryNotNull(key, value);
181      enumMap.put(key, value);
182    }
183    return ImmutableEnumMap.asImmutable(enumMap);
184  }

281  /**
282   * Returns a capacity that is sufficient to keep the map from being resized as long as it grows no
283   * larger than expectedSize and the load factor is â‰¥ its default (0.75).
284   */
285  static int capacity(int expectedSize) {
286    if (expectedSize < 3) {
287      checkNonnegative(expectedSize, "expectedSize");
288      return expectedSize + 1;
289    }
290    if (expectedSize < Ints.MAX_POWER_OF_TWO) {
291      // This seems to be consistent across JDKs. The capacity argument to HashMap and LinkedHashMap
292      // ends up being used to compute a "threshold" size, beyond which the internal table
293      // will be resized. That threshold is ceilingPowerOfTwo(capacity*loadFactor), where
294      // loadFactor is 0.75 by default. So with the calculation here we ensure that the
295      // threshold is equal to ceilingPowerOfTwo(expectedSize). There is a separate code
296      // path when the first operation on the new map is putAll(otherMap). There, prior to
297      // https://github.com/openjdk/jdk/commit/3e393047e12147a81e2899784b943923fc34da8e, a bug
298      // meant that sometimes a too-large threshold is calculated. However, this new threshold is
299      // independent of the initial capacity, except that it won't be lower than the threshold
300      // computed from that capacity. Because the internal table is only allocated on the first
301      // write, we won't see copying because of the new threshold. So it is always OK to use the
302      // calculation here.
303      return (int) Math.ceil(expectedSize / 0.75);
304    }
305    return Integer.MAX_VALUE; // any large value
306  }

470  /**
471   * Computes the difference between two maps. This difference is an immutable snapshot of the state
472   * of the maps at the time this method is called. It will never change, even if the maps change at
473   * a later time.
474   *
475   * <p>Since this method uses {@code HashMap} instances internally, the keys of the supplied maps
476   * must be well-behaved with respect to {@link Object#equals} and {@link Object#hashCode}.
477   *
478   * <p><b>Note:</b>If you only need to know whether two maps have the same mappings, call {@code
479   * left.equals(right)} instead of this method.
480   *
481   * @param left the map to treat as the "left" map for purposes of comparison
482   * @param right the map to treat as the "right" map for purposes of comparison
483   * @return the difference between the two maps
484   */
485  @SuppressWarnings("unchecked")
486  public static <K extends @Nullable Object, V extends @Nullable Object>
487      MapDifference<K, V> difference(
488          Map<? extends K, ? extends V> left, Map<? extends K, ? extends V> right) {
489    if (left instanceof SortedMap) {
490      SortedMap<K, ? extends V> sortedLeft = (SortedMap<K, ? extends V>) left;
491      return difference(sortedLeft, right);
492    }
493    /*
494     * This cast is safe: The Equivalence-accepting overload of difference() (which we call below)
495     * has a weird signature because Equivalence is itself a little weird. Still, we know that
496     * Equivalence.equals() can handle all inputs, and we know that the resulting MapDifference will
497     * contain only Ks and Vs (as opposed to possibly containing @Nullable objects even when K and V
498     * are *not* @Nullable).
499     *
500     * An alternative to suppressing the warning would be to inline the body of the other
501     * difference() method into this one.
502     */
503    @SuppressWarnings("nullness")
504    MapDifference<K, V> result =
505        (MapDifference<K, V>) difference(left, right, Equivalence.equals());
506    return result;
507  }

510   * Computes the difference between two maps. This difference is an immutable snapshot of the state
511   * of the maps at the time this method is called. It will never change, even if the maps change at
512   * a later time.
513   *
514   * <p>Since this method uses {@code HashMap} instances internally, the keys of the supplied maps
515   * must be well-behaved with respect to {@link Object#equals} and {@link Object#hashCode}.
516   *
517   * @param left the map to treat as the "left" map for purposes of comparison
518   * @param right the map to treat as the "right" map for purposes of comparison
519   * @param valueEquivalence the equivalence relationship to use to compare values
520   * @return the difference between the two maps
521   * @since 10.0
522   */
523  /*
524   * This method should really be annotated to accept maps with @Nullable value types. Fortunately,
525   * no existing Google callers appear to pass null values (much less pass null values *and* run a
526   * nullness checker).
527   *
528   * Still, if we decide that we want to make that work, we'd need to introduce a new type parameter
529   * for the Equivalence input type:
530   *
531   * <E, K extends @Nullable Object, V extends @Nullable E> ... difference(..., Equivalence<E> ...)
532   *
533   * Maybe we should, even though it will break source compatibility.
534   *
535   * Alternatively, this is a case in which it would be useful to be able to express Equivalence<?
536   * super @Nonnull T>).
537   *
538   * As things stand now, though, we have to either:
539   *
540   * - require non-null inputs so that we can guarantee non-null outputs
541   *
542   * - accept nullable inputs but force users to cope with nullable outputs
543   *
544   * And the non-null option is far more useful to existing users.
545   *
546   * (Vaguely related: Another thing we could consider is an overload that accepts a BiPredicate:
547   * https://github.com/google/guava/issues/3913)
548   */
549  public static <K extends @Nullable Object, V> MapDifference<K, V> difference(
550      Map<? extends K, ? extends V> left,
551      Map<? extends K, ? extends V> right,
552      Equivalence<? super V> valueEquivalence) {
553    Preconditions.checkNotNull(valueEquivalence);
554
555    Map<K, V> onlyOnLeft = newLinkedHashMap();
556    Map<K, V> onlyOnRight = new LinkedHashMap<>(right); // will whittle it down
557    Map<K, V> onBoth = newLinkedHashMap();
558    Map<K, MapDifference.ValueDifference<V>> differences = newLinkedHashMap();
559    doDifference(left, right, valueEquivalence, onlyOnLeft, onlyOnRight, onBoth, differences);
560    return new MapDifferenceImpl<>(onlyOnLeft, onlyOnRight, onBoth, differences);
561  }

571   *
572   * <p><b>Note:</b>If you only need to know whether two sorted maps have the same mappings, call
573   * {@code left.equals(right)} instead of this method.
574   *
575   * @param left the map to treat as the "left" map for purposes of comparison
576   * @param right the map to treat as the "right" map for purposes of comparison
577   * @return the difference between the two maps
578   * @since 11.0
579   */
580  public static <K extends @Nullable Object, V extends @Nullable Object>
581      SortedMapDifference<K, V> difference(
582          SortedMap<K, ? extends V> left, Map<? extends K, ? extends V> right) {
583    checkNotNull(left);
584    checkNotNull(right);
585    Comparator<? super K> comparator = orNaturalOrder(left.comparator());
586    SortedMap<K, V> onlyOnLeft = Maps.newTreeMap(comparator);
587    SortedMap<K, V> onlyOnRight = Maps.newTreeMap(comparator);
588    onlyOnRight.putAll(right); // will whittle it down
589    SortedMap<K, V> onBoth = Maps.newTreeMap(comparator);
590    SortedMap<K, MapDifference.ValueDifference<V>> differences = Maps.newTreeMap(comparator);
591
592    /*
593     * V is a possibly nullable type, but we decided to declare Equivalence with a type parameter
594     * that is restricted to non-nullable types. Still, this code is safe: We made that decision
595     * about Equivalence not because Equivalence is null-hostile but because *every* Equivalence can
596     * handle null inputs -- and thus it would be meaningless for the type system to distinguish
597     * between "an Equivalence for nullable Foo" and "an Equivalence for non-nullable Foo."
598     *
599     * (And the unchecked cast is safe because Equivalence is contravariant.)
600     */
601    @SuppressWarnings({"nullness", "unchecked"})
602    Equivalence<V> equalsEquivalence = (Equivalence<V>) Equivalence.equals();
603
604    doDifference(left, right, equalsEquivalence, onlyOnLeft, onlyOnRight, onBoth, differences);
605    return new SortedMapDifferenceImpl<>(onlyOnLeft, onlyOnRight, onBoth, differences);
606  }

608  private static <K extends @Nullable Object, V extends @Nullable Object> void doDifference(
609      Map<? extends K, ? extends V> left,
610      Map<? extends K, ? extends V> right,
611      Equivalence<? super V> valueEquivalence,
612      Map<K, V> onlyOnLeft,
613      Map<K, V> onlyOnRight,
614      Map<K, V> onBoth,
615      Map<K, MapDifference.ValueDifference<V>> differences) {
616    for (Entry<? extends K, ? extends V> entry : left.entrySet()) {
617      K leftKey = entry.getKey();
618      V leftValue = entry.getValue();
619      if (right.containsKey(leftKey)) {
620        /*
621         * The cast is safe because onlyOnRight contains all the keys of right.
622         *
623         * TODO(cpovirk): Consider checking onlyOnRight.containsKey instead of right.containsKey.
624         * That could change behavior if the input maps use different equivalence relations (and so
625         * a key that appears once in `right` might appear multiple times in `left`). We don't
626         * guarantee behavior in that case, anyway, and the current behavior is likely undesirable.
627         * So that's either a reason to feel free to change it or a reason to not bother thinking
628         * further about this.
629         */
630        V rightValue = uncheckedCastNullableTToT(onlyOnRight.remove(leftKey));
631        if (valueEquivalence.equivalent(leftValue, rightValue)) {
632          onBoth.put(leftKey, leftValue);
633        } else {
634          differences.put(leftKey, ValueDifferenceImpl.create(leftValue, rightValue));
635        }
636      } else {
637        onlyOnLeft.put(leftKey, leftValue);
638      }
639    }
640  }

665      this.onBoth = unmodifiableMap(onBoth);
666      this.differences = unmodifiableMap(differences);
667    }

669    @Override

671      return onlyOnLeft.isEmpty() && onlyOnRight.isEmpty() && differences.isEmpty();
672    }

674    @Override

677    }
678

682    }
683
684    @Override

686      return onBoth;
687    }
688
689    @Override

692    }
693
694    @Override

696      if (object == this) {
697        return true;
698      }
699      if (object instanceof MapDifference) {
700        MapDifference<?, ?> other = (MapDifference<?, ?>) object;
701        return entriesOnlyOnLeft().equals(other.entriesOnlyOnLeft())
702            && entriesOnlyOnRight().equals(other.entriesOnlyOnRight())
703            && entriesInCommon().equals(other.entriesInCommon())
704            && entriesDiffering().equals(other.entriesDiffering());
705      }
706      return false;
707    }

709    @Override
710    public int hashCode() {
711      return Objects.hashCode(
712          entriesOnlyOnLeft(), entriesOnlyOnRight(), entriesInCommon(), entriesDiffering());
713    }

715    @Override

718        return "equal";
719      }

721      StringBuilder result = new StringBuilder("not equal");
722      if (!onlyOnLeft.isEmpty()) {
723        result.append(": only on left=").append(onlyOnLeft);
724      }
725      if (!onlyOnRight.isEmpty()) {
726        result.append(": only on right=").append(onlyOnRight);
727      }
728      if (!differences.isEmpty()) {
729        result.append(": value differences=").append(differences);
730      }
731      return result.toString();
732    }
