015package com.google.common.base;
016
017import com.google.common.annotations.GwtCompatible;
018import org.checkerframework.checker.nullness.qual.Nullable;
019
020/**
021 * Exception thrown upon the failure of a <a
022 * href="https://github.com/google/guava/wiki/ConditionalFailuresExplained">verification check</a>,
023 * including those performed by the convenience methods of the {@link Verify} class.
024 *
025 * @since 17.0
026 */
027@GwtCompatible
028public class VerifyException extends RuntimeException {
029  /** Constructs a {@code VerifyException} with no message. */
030  public VerifyException() {}
031
032  /** Constructs a {@code VerifyException} with the message {@code message}. */
033  public VerifyException(@Nullable String message) {
034    super(message);
035  }
036
037  /**
038   * Constructs a {@code VerifyException} with the cause {@code cause} and a message that is {@code
039   * null} if {@code cause} is null, and {@code cause.toString()} otherwise.
040   *
041   * @since 19.0
042   */
043  public VerifyException(@Nullable Throwable cause) {
044    super(cause);
045  }
046
047  /**
048   * Constructs a {@code VerifyException} with the message {@code message} and the cause {@code
049   * cause}.
050   *
051   * @since 19.0
052   */
053  public VerifyException(@Nullable String message, @Nullable Throwable cause) {
054    super(message, cause);
055  }
056}
