015package com.google.common.reflect;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018import static java.util.Objects.requireNonNull;
019
020import com.google.common.annotations.Beta;
021import com.google.common.collect.FluentIterable;
022import com.google.common.collect.ImmutableList;
023import java.lang.annotation.Annotation;
024import java.lang.reflect.AnnotatedElement;
025import java.lang.reflect.AnnotatedType;
026import javax.annotation.CheckForNull;
027import org.checkerframework.checker.nullness.qual.Nullable;
028
029/**
030 * Represents a method or constructor parameter.
031 *
032 * <p><b>Note:</b> Since Java 8 introduced {@link java.lang.reflect.Parameter} to represent method
033 * and constructor parameters, this class is no longer necessary. We intend to deprecate it in a
034 * future version.
035 *
036 * @author Ben Yu
037 * @since 14.0
038 */
039@Beta
040@ElementTypesAreNonnullByDefault
041public final class Parameter implements AnnotatedElement {
042
043  private final Invokable<?, ?> declaration;
044  private final int position;
045  private final TypeToken<?> type;
046  private final ImmutableList<Annotation> annotations;
047
048  /**
049   * An {@link AnnotatedType} instance, or {@code null} under Android VMs (possible only when using
050   * the Android flavor of Guava). The field is declared with a type of {@code Object} to avoid
051   * compatibility problems on Android VMs. The corresponding accessor method, however, can have the
052   * more specific return type as long as users are careful to guard calls to it with version checks
053   * or reflection: Android VMs ignore the types of elements that aren't used.
054   */
055  private final @Nullable Object annotatedType;
056
057  Parameter(
058      Invokable<?, ?> declaration,
059      int position,
060      TypeToken<?> type,
061      Annotation[] annotations,
062      @Nullable Object annotatedType) {
063    this.declaration = declaration;
064    this.position = position;
065    this.type = type;
066    this.annotations = ImmutableList.copyOf(annotations);
067    this.annotatedType = annotatedType;
068  }
069
070  /** Returns the type of the parameter. */
071  public TypeToken<?> getType() {
072    return type;
073  }
074
075  /** Returns the {@link Invokable} that declares this parameter. */
076  public Invokable<?, ?> getDeclaringInvokable() {
077    return declaration;
078  }
079
080  @Override
081  public boolean isAnnotationPresent(Class<? extends Annotation> annotationType) {
082    return getAnnotation(annotationType) != null;
083  }
084
085  @Override
086  @CheckForNull
087  public <A extends Annotation> A getAnnotation(Class<A> annotationType) {
088    checkNotNull(annotationType);
089    for (Annotation annotation : annotations) {
090      if (annotationType.isInstance(annotation)) {
091        return annotationType.cast(annotation);
092      }
093    }
094    return null;
095  }
096
097  @Override
098  public Annotation[] getAnnotations() {
099    return getDeclaredAnnotations();
100  }
101
102  /** @since 18.0 */
103  @Override
104  public <A extends Annotation> A[] getAnnotationsByType(Class<A> annotationType) {
105    return getDeclaredAnnotationsByType(annotationType);
106  }
107
108  /** @since 18.0 */
109  @Override
110  public Annotation[] getDeclaredAnnotations() {
111    return annotations.toArray(new Annotation[0]);
112  }
113
114  /** @since 18.0 */
115  @Override
116  @CheckForNull
117  public <A extends Annotation> A getDeclaredAnnotation(Class<A> annotationType) {
118    checkNotNull(annotationType);
119    return FluentIterable.from(annotations).filter(annotationType).first().orNull();
120  }
121
122  /** @since 18.0 */
123  @Override
124  public <A extends Annotation> A[] getDeclaredAnnotationsByType(Class<A> annotationType) {
125    @Nullable
126    A[] result = FluentIterable.from(annotations).filter(annotationType).toArray(annotationType);
127    @SuppressWarnings("nullness") // safe because the input list contains no nulls
128    A[] cast = (A[]) result;
129    return cast;
130  }
131
132  /**
133   * Returns the {@link AnnotatedType} of the parameter.
134   *
135   * <p>This method will fail if run under an Android VM.
136   *
137   * @since 25.1 for guava-jre (available since 32.0.0 in guava-android)
138   */
139  @SuppressWarnings({"Java7ApiChecker", "AndroidJdkLibsChecker"})
140  @IgnoreJRERequirement
141  public AnnotatedType getAnnotatedType() {
142    return requireNonNull((AnnotatedType) annotatedType);
143  }
144
145  @Override
146  public boolean equals(@CheckForNull Object obj) {
147    if (obj instanceof Parameter) {
148      Parameter that = (Parameter) obj;
149      return position == that.position && declaration.equals(that.declaration);
150    }
151    return false;
152  }
153
154  @Override
155  public int hashCode() {
156    return position;
157  }
158
159  @Override
160  public String toString() {
161    return type + " arg" + position;
162  }
163}
