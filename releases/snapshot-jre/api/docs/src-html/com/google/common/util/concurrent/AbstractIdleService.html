015package com.google.common.util.concurrent;
016
017import static com.google.common.util.concurrent.Platform.restoreInterruptIfIsInterruptedException;
018
019import com.google.common.annotations.GwtIncompatible;
020import com.google.common.annotations.J2ktIncompatible;
021import com.google.common.base.Supplier;
022import com.google.errorprone.annotations.CanIgnoreReturnValue;
023import com.google.j2objc.annotations.WeakOuter;
024import java.time.Duration;
025import java.util.concurrent.Executor;
026import java.util.concurrent.TimeUnit;
027import java.util.concurrent.TimeoutException;
028
029/**
030 * Base class for services that do not need a thread while "running" but may need one during startup
031 * and shutdown. Subclasses can implement {@link #startUp} and {@link #shutDown} methods, each which
032 * run in an executor which by default uses a separate thread for each method.
033 *
034 * @author Chris Nokleberg
035 * @since 1.0
036 */
037@GwtIncompatible
038@J2ktIncompatible
039@ElementTypesAreNonnullByDefault
040public abstract class AbstractIdleService implements Service {
041
042  /* Thread names will look like {@code "MyService STARTING"}. */
043  private final Supplier<String> threadNameSupplier = new ThreadNameSupplier();
044
045  @WeakOuter
046  private final class ThreadNameSupplier implements Supplier<String> {
047    @Override
048    public String get() {
049      return serviceName() + " " + state();
050    }
051  }
052
053  /* use AbstractService for state management */
054  private final Service delegate = new DelegateService();
055
056  @WeakOuter
057  private final class DelegateService extends AbstractService {
058    @Override
059    protected final void doStart() {
060      MoreExecutors.renamingDecorator(executor(), threadNameSupplier)
061          .execute(
062              new Runnable() {
063                @Override
064                public void run() {
065                  try {
066                    startUp();
067                    notifyStarted();
068                  } catch (Throwable t) {
069                    restoreInterruptIfIsInterruptedException(t);
070                    notifyFailed(t);
071                  }
072                }
073              });
074    }
075
076    @Override
077    protected final void doStop() {
078      MoreExecutors.renamingDecorator(executor(), threadNameSupplier)
079          .execute(
080              new Runnable() {
081                @Override
082                public void run() {
083                  try {
084                    shutDown();
085                    notifyStopped();
086                  } catch (Throwable t) {
087                    restoreInterruptIfIsInterruptedException(t);
088                    notifyFailed(t);
089                  }
090                }
091              });
092    }
093
094    @Override
095    public String toString() {
096      return AbstractIdleService.this.toString();
097    }
098  }
099
100  /** Constructor for use by subclasses. */
101  protected AbstractIdleService() {}
102
103  /** Start the service. */
104  protected abstract void startUp() throws Exception;
105
106  /** Stop the service. */
107  protected abstract void shutDown() throws Exception;
108
109  /**
110   * Returns the {@link Executor} that will be used to run this service. Subclasses may override
111   * this method to use a custom {@link Executor}, which may configure its worker thread with a
112   * specific name, thread group or priority. The returned executor's {@link
113   * Executor#execute(Runnable) execute()} method is called when this service is started and
114   * stopped, and should return promptly.
115   */
116  protected Executor executor() {
117    return new Executor() {
118      @Override
119      public void execute(Runnable command) {
120        MoreExecutors.newThread(threadNameSupplier.get(), command).start();
121      }
122    };
123  }
124
125  @Override
126  public String toString() {
127    return serviceName() + " [" + state() + "]";
128  }
129
130  @Override
131  public final boolean isRunning() {
132    return delegate.isRunning();
133  }
134
135  @Override
136  public final State state() {
137    return delegate.state();
138  }
139
140  /** @since 13.0 */
141  @Override
142  public final void addListener(Listener listener, Executor executor) {
143    delegate.addListener(listener, executor);
144  }
145
146  /** @since 14.0 */
147  @Override
148  public final Throwable failureCause() {
149    return delegate.failureCause();
150  }
151
152  /** @since 15.0 */
153  @CanIgnoreReturnValue
154  @Override
155  public final Service startAsync() {
156    delegate.startAsync();
157    return this;
158  }
159
160  /** @since 15.0 */
161  @CanIgnoreReturnValue
162  @Override
163  public final Service stopAsync() {
164    delegate.stopAsync();
165    return this;
166  }
167
168  /** @since 15.0 */
169  @Override
170  public final void awaitRunning() {
171    delegate.awaitRunning();
172  }
173
174  /** @since 28.0 */
175  @Override
176  public final void awaitRunning(Duration timeout) throws TimeoutException {
177    Service.super.awaitRunning(timeout);
178  }
179
180  /** @since 15.0 */
181  @Override
182  public final void awaitRunning(long timeout, TimeUnit unit) throws TimeoutException {
183    delegate.awaitRunning(timeout, unit);
184  }
185
186  /** @since 15.0 */
187  @Override
188  public final void awaitTerminated() {
189    delegate.awaitTerminated();
190  }
191
192  /** @since 28.0 */
193  @Override
194  public final void awaitTerminated(Duration timeout) throws TimeoutException {
195    Service.super.awaitTerminated(timeout);
196  }
197
198  /** @since 15.0 */
199  @Override
200  public final void awaitTerminated(long timeout, TimeUnit unit) throws TimeoutException {
201    delegate.awaitTerminated(timeout, unit);
202  }
203
204  /**
205   * Returns the name of this service. {@link AbstractIdleService} may include the name in debugging
206   * output.
207   *
208   * @since 14.0
209   */
210  protected String serviceName() {
211    return getClass().getSimpleName();
212  }
213}
