015package com.google.common.reflect;
016
017import com.google.errorprone.annotations.CanIgnoreReturnValue;
018import com.google.errorprone.annotations.DoNotMock;
019import java.util.Map;
020import javax.annotation.CheckForNull;
021import org.checkerframework.checker.nullness.qual.NonNull;
022import org.checkerframework.checker.nullness.qual.Nullable;
023
024/**
025 * A map, each entry of which maps a {@link TypeToken} to an instance of that type. In addition to
026 * implementing {@code Map}, the additional type-safe operations {@link #putInstance} and {@link
027 * #getInstance} are available.
028 *
029 * <p>Generally, implementations don't support {@link #put} and {@link #putAll} because there is no
030 * way to check an object at runtime to be an instance of a {@link TypeToken}. Instead, caller
031 * should use the type safe {@link #putInstance}.
032 *
033 * <p>Also, if caller suppresses unchecked warnings and passes in an {@code Iterable<String>} for
034 * type {@code Iterable<Integer>}, the map won't be able to detect and throw type error.
035 *
036 * <p>Like any other {@code Map<Class, Object>}, this map may contain entries for primitive types,
037 * and a primitive type and its corresponding wrapper type may map to different values.
038 *
039 * @param <B> the common supertype that all entries must share; often this is simply {@link Object}
040 * @author Ben Yu
041 * @since 13.0
042 */
043@DoNotMock("Use ImmutableTypeToInstanceMap or MutableTypeToInstanceMap")
044@ElementTypesAreNonnullByDefault
045public interface TypeToInstanceMap<B extends @Nullable Object>
046    extends Map<TypeToken<? extends @NonNull B>, B> {
047
048  /**
049   * Returns the value the specified class is mapped to, or {@code null} if no entry for this class
050   * is present. This will only return a value that was bound to this specific class, not a value
051   * that may have been bound to a subtype.
052   *
053   * <p>{@code getInstance(Foo.class)} is equivalent to {@code
054   * getInstance(TypeToken.of(Foo.class))}.
055   */
056  @CheckForNull
057  <T extends @NonNull B> T getInstance(Class<T> type);
058
059  /**
060   * Returns the value the specified type is mapped to, or {@code null} if no entry for this type is
061   * present. This will only return a value that was bound to this specific type, not a value that
062   * may have been bound to a subtype.
063   */
064  @CheckForNull
065  <T extends @NonNull B> T getInstance(TypeToken<T> type);
066
067  /**
068   * Maps the specified class to the specified value. Does <i>not</i> associate this value with any
069   * of the class's supertypes.
070   *
071   * <p>{@code putInstance(Foo.class, foo)} is equivalent to {@code
072   * putInstance(TypeToken.of(Foo.class), foo)}.
073   *
074   * @return the value previously associated with this class (possibly {@code null}), or {@code
075   *     null} if there was no previous entry.
076   */
077  @CanIgnoreReturnValue
078  @CheckForNull
079  <T extends B> T putInstance(Class<@NonNull T> type, @ParametricNullness T value);
080
081  /**
082   * Maps the specified type to the specified value. Does <i>not</i> associate this value with any
083   * of the type's supertypes.
084   *
085   * @return the value previously associated with this type (possibly {@code null}), or {@code null}
086   *     if there was no previous entry.
087   */
088  @CanIgnoreReturnValue
089  @CheckForNull
090  <T extends B> T putInstance(TypeToken<@NonNull T> type, @ParametricNullness T value);
091}
