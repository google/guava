040/**
041 * A {@link BiMap} whose contents will never change, with many other important properties detailed
042 * at {@link ImmutableCollection}.
043 *
044 * @author Jared Levy
045 * @since 2.0
046 */
047@GwtCompatible(serializable = true, emulated = true)
048@ElementTypesAreNonnullByDefault
049public abstract class ImmutableBiMap<K, V> extends ImmutableBiMapFauxverideShim<K, V>
050    implements BiMap<K, V> {

052  /**
053   * Returns a {@link Collector} that accumulates elements into an {@code ImmutableBiMap} whose keys
054   * and values are the result of applying the provided mapping functions to the input elements.
055   * Entries appear in the result {@code ImmutableBiMap} in encounter order.
056   *
057   * <p>If the mapped keys or values contain duplicates (according to {@link
058   * Object#equals(Object)}), an {@code IllegalArgumentException} is thrown when the collection
059   * operation is performed. (This differs from the {@code Collector} returned by {@link
060   * Collectors#toMap(Function, Function)}, which throws an {@code IllegalStateException}.)
061   *
062   * @since 21.0
063   */
064  public static <T extends @Nullable Object, K, V>
065      Collector<T, ?, ImmutableBiMap<K, V>> toImmutableBiMap(
066          Function<? super T, ? extends K> keyFunction,
067          Function<? super T, ? extends V> valueFunction) {
068    return CollectCollectors.toImmutableBiMap(keyFunction, valueFunction);
069  }

312  /**
313   * A builder for creating immutable bimap instances, especially {@code public static final} bimaps
314   * ("constant bimaps"). Example:
315   *
316   * <pre>{@code
317   * static final ImmutableBiMap<String, Integer> WORD_TO_INT =
318   *     new ImmutableBiMap.Builder<String, Integer>()
319   *         .put("one", 1)
320   *         .put("two", 2)
321   *         .put("three", 3)
322   *         .buildOrThrow();
323   * }</pre>
324   *
325   * <p>For <i>small</i> immutable bimaps, the {@code ImmutableBiMap.of()} methods are even more
326   * convenient.
327   *
328   * <p>By default, a {@code Builder} will generate bimaps that iterate over entries in the order
329   * they were inserted into the builder. For example, in the above example, {@code
330   * WORD_TO_INT.entrySet()} is guaranteed to iterate over the entries in the order {@code "one"=1,
331   * "two"=2, "three"=3}, and {@code keySet()} and {@code values()} respect the same order. If you
332   * want a different order, consider using {@link #orderEntriesByValue(Comparator)}, which changes
333   * this builder to sort entries by value.
334   *
335   * <p>Builder instances can be reused - it is safe to call {@link #buildOrThrow} multiple times to
336   * build multiple bimaps in series. Each bimap is a superset of the bimaps created before it.
337   *
338   * @since 2.0
339   */
340  public static final class Builder<K, V> extends ImmutableMap.Builder<K, V> {
341
342    /**
343     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
344     * ImmutableBiMap#builder}.
345     */
346    public Builder() {}
347
348    Builder(int size) {
349      super(size);
350    }

382    @CanIgnoreReturnValue

396    @CanIgnoreReturnValue

400      super.putAll(entries);
401      return this;
402    }
403
404    /**
405     * Configures this {@code Builder} to order entries by value according to the specified
406     * comparator.
407     *
408     * <p>The sort order is stable, that is, if two entries have values that compare as equivalent,
409     * the entry that was inserted first will be first in the built map's iteration order.
410     *
411     * @throws IllegalStateException if this method was already called
412     * @since 19.0
413     */
414    @CanIgnoreReturnValue
415    @Beta
416    @Override
417    public Builder<K, V> orderEntriesByValue(Comparator<? super V> valueComparator) {
418      super.orderEntriesByValue(valueComparator);
419      return this;
420    }
421
422    @Override
423    @CanIgnoreReturnValue
424    Builder<K, V> combine(ImmutableMap.Builder<K, V> builder) {
425      super.combine(builder);
426      return this;
427    }
428
429    /**
430     * Returns a newly-created immutable bimap. The iteration order of the returned bimap is the
431     * order in which entries were inserted into the builder, unless {@link #orderEntriesByValue}
432     * was called, in which case entries are sorted by value.
433     *
434     * <p>Prefer the equivalent method {@link #buildOrThrow()} to make it explicit that the method
435     * will throw an exception if there are duplicate keys or values. The {@code build()} method
436     * will soon be deprecated.
437     *
438     * @throws IllegalArgumentException if duplicate keys or values were added
439     */
440    @Override
441    public ImmutableBiMap<K, V> build() {
442      return buildOrThrow();
443    }
444
445    /**
446     * Returns a newly-created immutable bimap, or throws an exception if any key or value was added
447     * more than once. The iteration order of the returned bimap is the order in which entries were
448     * inserted into the builder, unless {@link #orderEntriesByValue} was called, in which case
449     * entries are sorted by value.
450     *
451     * @throws IllegalArgumentException if duplicate keys or values were added
452     * @since 31.0
453     */
454    @Override
455    public ImmutableBiMap<K, V> buildOrThrow() {
456      switch (size) {
457        case 0:
458          return of();
459        case 1:
460          // requireNonNull is safe because the first `size` elements have been filled in.
461          Entry<K, V> onlyEntry = requireNonNull(entries[0]);
462          return of(onlyEntry.getKey(), onlyEntry.getValue());
463        default:
464          /*
465           * If entries is full, or if hash flooding is detected, then this implementation may end
466           * up using the entries array directly and writing over the entry objects with
467           * non-terminal entries, but this is safe; if this Builder is used further, it will grow
468           * the entries array (so it can't affect the original array), and future build() calls
469           * will always copy any entry objects that cannot be safely reused.
470           */
471          if (valueComparator != null) {
472            if (entriesUsed) {
473              entries = Arrays.copyOf(entries, size);
474            }
475            Arrays.sort(
476                entries,
477                0,
478                size,
479                Ordering.from(valueComparator).onResultOf(Maps.<V>valueFunction()));
480          }
481          entriesUsed = true;
482          return RegularImmutableBiMap.fromEntryArray(size, entries);
483      }
484    }
485
486    /**
487     * Throws {@link UnsupportedOperationException}. This method is inherited from {@link
488     * ImmutableMap.Builder}, but it does not make sense for bimaps.
489     *
490     * @throws UnsupportedOperationException always
491     * @deprecated This method does not make sense for bimaps and should not be called.
492     * @since 31.1
493     */
494    @DoNotCall
495    @Deprecated
496    @Override
497    public ImmutableBiMap<K, V> buildKeepingLast() {
498      throw new UnsupportedOperationException("Not supported for bimaps");
499    }
500
501    @Override
502    @VisibleForTesting
503    ImmutableBiMap<K, V> buildJdkBacked() {
504      checkState(
505          valueComparator == null,
506          "buildJdkBacked is for tests only, doesn't support orderEntriesByValue");
507      switch (size) {
508        case 0:
509          return of();
510        case 1:
511          // requireNonNull is safe because the first `size` elements have been filled in.
512          Entry<K, V> onlyEntry = requireNonNull(entries[0]);
513          return of(onlyEntry.getKey(), onlyEntry.getValue());
514        default:
515          entriesUsed = true;
516          return RegularImmutableBiMap.fromEntryArray(size, entries);
517      }
518    }
519  }

521  /**
522   * Returns an immutable bimap containing the same entries as {@code map}. If {@code map} somehow
523   * contains entries with duplicate keys (for example, if it is a {@code SortedMap} whose
524   * comparator is not <i>consistent with equals</i>), the results of this method are undefined.
525   *
526   * <p>The returned {@code BiMap} iterates over entries in the same order as the {@code entrySet}
527   * of the original map.
528   *
529   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
530   * safe to do so. The exact circumstances under which a copy will or will not be performed are
531   * undocumented and subject to change.
532   *
533   * @throws IllegalArgumentException if two keys have the same value or two values have the same
534   *     key
535   * @throws NullPointerException if any key or value in {@code map} is null
536   */
537  public static <K, V> ImmutableBiMap<K, V> copyOf(Map<? extends K, ? extends V> map) {
538    if (map instanceof ImmutableBiMap) {
539      @SuppressWarnings("unchecked") // safe since map is not writable
540      ImmutableBiMap<K, V> bimap = (ImmutableBiMap<K, V>) map;
541      // TODO(lowasser): if we need to make a copy of a BiMap because the
542      // forward map is a view, don't make a copy of the non-view delegate map
543      if (!bimap.isPartialView()) {
544        return bimap;
545      }
546    }
547    return copyOf(map.entrySet());
548  }
549
550  /**
551   * Returns an immutable bimap containing the given entries. The returned bimap iterates over
552   * entries in the same order as the original iterable.
553   *
554   * @throws IllegalArgumentException if two keys have the same value or two values have the same
555   *     key
556   * @throws NullPointerException if any key, value, or entry is null
557   * @since 19.0
558   */
559  @Beta
560  public static <K, V> ImmutableBiMap<K, V> copyOf(
561      Iterable<? extends Entry<? extends K, ? extends V>> entries) {
562    @SuppressWarnings("unchecked") // we'll only be using getKey and getValue, which are covariant
563    Entry<K, V>[] entryArray = (Entry<K, V>[]) Iterables.toArray(entries, EMPTY_ENTRY_ARRAY);
564    switch (entryArray.length) {
565      case 0:
566        return of();
567      case 1:
568        Entry<K, V> entry = entryArray[0];
569        return of(entry.getKey(), entry.getValue());
570      default:
571        /*
572         * The current implementation will end up using entryArray directly, though it will write
573         * over the (arbitrary, potentially mutable) Entry objects actually stored in entryArray.
574         */
575        return RegularImmutableBiMap.fromEntries(entryArray);
576    }
577  }

618  /**
619   * Serialized type for all ImmutableBiMap instances. It captures the logical contents and they are
620   * reconstructed using public factory methods. This ensures that the implementation types remain
621   * as implementation details.
622   *
623   * <p>Since the bimap is immutable, ImmutableBiMap doesn't require special logic for keeping the
624   * bimap and its inverse in sync during serialization, the way AbstractBiMap does.
625   */
626  private static class SerializedForm<K, V> extends ImmutableMap.SerializedForm<K, V> {
627    SerializedForm(ImmutableBiMap<K, V> bimap) {
628      super(bimap);
629    }
630
631    @Override
632    Builder<K, V> makeBuilder(int size) {
633      return new Builder<>(size);
634    }
635
636    private static final long serialVersionUID = 0;
637  }
638
639  @Override
640  Object writeReplace() {
641    return new SerializedForm<>(this);
642  }
