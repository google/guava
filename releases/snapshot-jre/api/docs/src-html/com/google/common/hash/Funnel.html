015package com.google.common.hash;
016
017import com.google.common.annotations.Beta;
018import com.google.errorprone.annotations.DoNotMock;
019import java.io.Serializable;
020
021/**
022 * An object which can send data from an object of type {@code T} into a {@code PrimitiveSink}.
023 * Implementations for common types can be found in {@link Funnels}.
024 *
025 * <p>Note that serialization of {@linkplain BloomFilter bloom filters} requires the proper
026 * serialization of funnels. When possible, it is recommended that funnels be implemented as a
027 * single-element enum to maintain serialization guarantees. See Effective Java (2nd Edition), Item
028 * 3: "Enforce the singleton property with a private constructor or an enum type". For example:
029 *
030 * <pre>{@code
031 * public enum PersonFunnel implements Funnel<Person> {
032 *   INSTANCE;
033 *   public void funnel(Person person, PrimitiveSink into) {
034 *     into.putUnencodedChars(person.getFirstName())
035 *         .putUnencodedChars(person.getLastName())
036 *         .putInt(person.getAge());
037 *   }
038 * }
039 * }</pre>
040 *
041 * @author Dimitris Andreou
042 * @since 11.0
043 */
044@Beta
045@DoNotMock("Implement with a lambda")
046public interface Funnel<T> extends Serializable {
047
048  /**
049   * Sends a stream of data from the {@code from} object into the sink {@code into}. There is no
050   * requirement that this data be complete enough to fully reconstitute the object later.
051   *
052   * @since 12.0 (in Guava 11.0, {@code PrimitiveSink} was named {@code Sink})
053   */
054  void funnel(T from, PrimitiveSink into);
055}
