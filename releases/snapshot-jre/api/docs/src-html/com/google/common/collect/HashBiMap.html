047/**
048 * A {@link BiMap} backed by two hash tables. This implementation allows null keys and values. A
049 * {@code HashBiMap} and its inverse are both serializable.
050 *
051 * <p>This implementation guarantees insertion-based iteration order of its keys.
052 *
053 * <p>See the Guava User Guide article on <a href=
054 * "https://github.com/google/guava/wiki/NewCollectionTypesExplained#bimap">{@code BiMap} </a>.
055 *
056 * @author Louis Wasserman
057 * @author Mike Bostock
058 * @since 2.0
059 */
060@GwtCompatible(emulated = true)
061@ElementTypesAreNonnullByDefault
062public final class HashBiMap<K extends @Nullable Object, V extends @Nullable Object>
063    extends IteratorBasedAbstractMap<K, V> implements BiMap<K, V>, Serializable {
064
065  /** Returns a new, empty {@code HashBiMap} with the default initial capacity (16). */
066  public static <K extends @Nullable Object, V extends @Nullable Object> HashBiMap<K, V> create() {
067    return create(16);
068  }
069
070  /**
071   * Constructs a new, empty bimap with the specified expected size.
072   *
073   * @param expectedSize the expected number of entries
074   * @throws IllegalArgumentException if the specified expected size is negative
075   */
076  public static <K extends @Nullable Object, V extends @Nullable Object> HashBiMap<K, V> create(
077      int expectedSize) {
078    return new HashBiMap<>(expectedSize);
079  }
080
081  /**
082   * Constructs a new bimap containing initial values from {@code map}. The bimap is created with an
083   * initial capacity sufficient to hold the mappings in the specified map.
084   */
085  public static <K extends @Nullable Object, V extends @Nullable Object> HashBiMap<K, V> create(
086      Map<? extends K, ? extends V> map) {
087    HashBiMap<K, V> bimap = create(map.size());
088    bimap.putAll(map);
089    return bimap;
090  }

092  private static final class BiEntry<K extends @Nullable Object, V extends @Nullable Object>
093      extends ImmutableEntry<K, V> {
094    final int keyHash;
095    final int valueHash;
096
097    // All BiEntry instances are strongly reachable from owning HashBiMap through
098    // "HashBiMap.hashTableKToV" and "BiEntry.nextInKToVBucket" references.
099    // Under that assumption, the remaining references can be safely marked as @Weak.
100    // Using @Weak is necessary to avoid retain-cycles between BiEntry instances on iOS,
101    // which would cause memory leaks when non-empty HashBiMap with cyclic BiEntry
102    // instances is deallocated.
103    @CheckForNull BiEntry<K, V> nextInKToVBucket;
104    @Weak @CheckForNull BiEntry<K, V> nextInVToKBucket;
105
106    @Weak @CheckForNull BiEntry<K, V> nextInKeyInsertionOrder;
107    @Weak @CheckForNull BiEntry<K, V> prevInKeyInsertionOrder;
108
109    BiEntry(@ParametricNullness K key, int keyHash, @ParametricNullness V value, int valueHash) {
110      super(key, value);
111      this.keyHash = keyHash;
112      this.valueHash = valueHash;
113    }
114  }
115
116  private static final double LOAD_FACTOR = 1.0;
117
118  /*
119   * The following two arrays may *contain* nulls, but they are never *themselves* null: Even though
120   * they are not initialized inline in the constructor, they are initialized from init(), which the
121   * constructor calls (as does readObject()).
122   */
123  private transient @Nullable BiEntry<K, V>[] hashTableKToV;
124  private transient @Nullable BiEntry<K, V>[] hashTableVToK;
125  @Weak @CheckForNull private transient BiEntry<K, V> firstInKeyInsertionOrder;
126  @Weak @CheckForNull private transient BiEntry<K, V> lastInKeyInsertionOrder;
127  private transient int size;
128  private transient int mask;
129  private transient int modCount;
130
131  private HashBiMap(int expectedSize) {
132    init(expectedSize);
133  }
134
135  private void init(int expectedSize) {
136    checkNonnegative(expectedSize, "expectedSize");
137    int tableSize = Hashing.closedTableSize(expectedSize, LOAD_FACTOR);
138    this.hashTableKToV = createTable(tableSize);
139    this.hashTableVToK = createTable(tableSize);
140    this.firstInKeyInsertionOrder = null;
141    this.lastInKeyInsertionOrder = null;
142    this.size = 0;
143    this.mask = tableSize - 1;
144    this.modCount = 0;
145  }

147  /**
148   * Finds and removes {@code entry} from the bucket linked lists in both the key-to-value direction
149   * and the value-to-key direction.
150   */
151  private void delete(BiEntry<K, V> entry) {
152    int keyBucket = entry.keyHash & mask;
153    BiEntry<K, V> prevBucketEntry = null;
154    for (BiEntry<K, V> bucketEntry = hashTableKToV[keyBucket];
155        true;
156        bucketEntry = bucketEntry.nextInKToVBucket) {
157      if (bucketEntry == entry) {
158        if (prevBucketEntry == null) {
159          hashTableKToV[keyBucket] = entry.nextInKToVBucket;
160        } else {
161          prevBucketEntry.nextInKToVBucket = entry.nextInKToVBucket;
162        }
163        break;
164      }
165      prevBucketEntry = bucketEntry;
166    }
167
168    int valueBucket = entry.valueHash & mask;
169    prevBucketEntry = null;
170    for (BiEntry<K, V> bucketEntry = hashTableVToK[valueBucket];
171        true;
172        bucketEntry = bucketEntry.nextInVToKBucket) {
173      if (bucketEntry == entry) {
174        if (prevBucketEntry == null) {
175          hashTableVToK[valueBucket] = entry.nextInVToKBucket;
176        } else {
177          prevBucketEntry.nextInVToKBucket = entry.nextInVToKBucket;
178        }
179        break;
180      }
181      prevBucketEntry = bucketEntry;
182    }
183
184    if (entry.prevInKeyInsertionOrder == null) {
185      firstInKeyInsertionOrder = entry.nextInKeyInsertionOrder;
186    } else {
187      entry.prevInKeyInsertionOrder.nextInKeyInsertionOrder = entry.nextInKeyInsertionOrder;
188    }
189
190    if (entry.nextInKeyInsertionOrder == null) {
191      lastInKeyInsertionOrder = entry.prevInKeyInsertionOrder;
192    } else {
193      entry.nextInKeyInsertionOrder.prevInKeyInsertionOrder = entry.prevInKeyInsertionOrder;
194    }
195
196    size--;
197    modCount++;
198  }
199
200  private void insert(BiEntry<K, V> entry, @CheckForNull BiEntry<K, V> oldEntryForKey) {
201    int keyBucket = entry.keyHash & mask;
202    entry.nextInKToVBucket = hashTableKToV[keyBucket];
203    hashTableKToV[keyBucket] = entry;
204
205    int valueBucket = entry.valueHash & mask;
206    entry.nextInVToKBucket = hashTableVToK[valueBucket];
207    hashTableVToK[valueBucket] = entry;
208
209    if (oldEntryForKey == null) {
210      entry.prevInKeyInsertionOrder = lastInKeyInsertionOrder;
211      entry.nextInKeyInsertionOrder = null;
212      if (lastInKeyInsertionOrder == null) {
213        firstInKeyInsertionOrder = entry;
214      } else {
215        lastInKeyInsertionOrder.nextInKeyInsertionOrder = entry;
216      }
217      lastInKeyInsertionOrder = entry;
218    } else {
219      entry.prevInKeyInsertionOrder = oldEntryForKey.prevInKeyInsertionOrder;
220      if (entry.prevInKeyInsertionOrder == null) {
221        firstInKeyInsertionOrder = entry;
222      } else {
223        entry.prevInKeyInsertionOrder.nextInKeyInsertionOrder = entry;
224      }
225      entry.nextInKeyInsertionOrder = oldEntryForKey.nextInKeyInsertionOrder;
226      if (entry.nextInKeyInsertionOrder == null) {
227        lastInKeyInsertionOrder = entry;
228      } else {
229        entry.nextInKeyInsertionOrder.prevInKeyInsertionOrder = entry;
230      }
231    }
232
233    size++;
234    modCount++;
235  }

237  @CheckForNull
238  private BiEntry<K, V> seekByKey(@CheckForNull Object key, int keyHash) {
239    for (BiEntry<K, V> entry = hashTableKToV[keyHash & mask];
240        entry != null;
241        entry = entry.nextInKToVBucket) {
242      if (keyHash == entry.keyHash && Objects.equal(key, entry.key)) {
243        return entry;
244      }
245    }
246    return null;
247  }
248
249  @CheckForNull
250  private BiEntry<K, V> seekByValue(@CheckForNull Object value, int valueHash) {
251    for (BiEntry<K, V> entry = hashTableVToK[valueHash & mask];
252        entry != null;
253        entry = entry.nextInVToKBucket) {
254      if (valueHash == entry.valueHash && Objects.equal(value, entry.value)) {
255        return entry;
256      }
257    }
258    return null;
259  }

294  @CheckForNull
295  private V put(@ParametricNullness K key, @ParametricNullness V value, boolean force) {
296    int keyHash = smearedHash(key);
297    int valueHash = smearedHash(value);
298
299    BiEntry<K, V> oldEntryForKey = seekByKey(key, keyHash);
300    if (oldEntryForKey != null
301        && valueHash == oldEntryForKey.valueHash
302        && Objects.equal(value, oldEntryForKey.value)) {
303      return value;
304    }
305
306    BiEntry<K, V> oldEntryForValue = seekByValue(value, valueHash);
307    if (oldEntryForValue != null) {
308      if (force) {
309        delete(oldEntryForValue);
310      } else {
311        throw new IllegalArgumentException("value already present: " + value);
312      }
313    }
314
315    BiEntry<K, V> newEntry = new BiEntry<>(key, keyHash, value, valueHash);
316    if (oldEntryForKey != null) {
317      delete(oldEntryForKey);
318      insert(newEntry, oldEntryForKey);
319      oldEntryForKey.prevInKeyInsertionOrder = null;
320      oldEntryForKey.nextInKeyInsertionOrder = null;
321      return oldEntryForKey.value;
322    } else {
323      insert(newEntry, null);
324      rehashIfNecessary();
325      return null;
326    }
327  }

336  @CanIgnoreReturnValue
337  @CheckForNull
338  private K putInverse(@ParametricNullness V value, @ParametricNullness K key, boolean force) {
339    int valueHash = smearedHash(value);
340    int keyHash = smearedHash(key);
341
342    BiEntry<K, V> oldEntryForValue = seekByValue(value, valueHash);
343    BiEntry<K, V> oldEntryForKey = seekByKey(key, keyHash);
344    if (oldEntryForValue != null
345        && keyHash == oldEntryForValue.keyHash
346        && Objects.equal(key, oldEntryForValue.key)) {
347      return key;
348    } else if (oldEntryForKey != null && !force) {
349      throw new IllegalArgumentException("key already present: " + key);
350    }
351
352    /*
353     * The ordering here is important: if we deleted the key entry and then the value entry,
354     * the key entry's prev or next pointer might point to the dead value entry, and when we
355     * put the new entry in the key entry's position in iteration order, it might invalidate
356     * the linked list.
357     */
358
359    if (oldEntryForValue != null) {
360      delete(oldEntryForValue);
361    }
362
363    if (oldEntryForKey != null) {
364      delete(oldEntryForKey);
365    }
366
367    BiEntry<K, V> newEntry = new BiEntry<>(key, keyHash, value, valueHash);
368    insert(newEntry, oldEntryForKey);
369
370    if (oldEntryForKey != null) {
371      oldEntryForKey.prevInKeyInsertionOrder = null;
372      oldEntryForKey.nextInKeyInsertionOrder = null;
373    }
374    if (oldEntryForValue != null) {
375      oldEntryForValue.prevInKeyInsertionOrder = null;
376      oldEntryForValue.nextInKeyInsertionOrder = null;
377    }
378    rehashIfNecessary();
379    return Maps.keyOrNull(oldEntryForValue);
380  }
381
382  private void rehashIfNecessary() {
383    @Nullable BiEntry<K, V>[] oldKToV = hashTableKToV;
384    if (Hashing.needsResizing(size, oldKToV.length, LOAD_FACTOR)) {
385      int newTableSize = oldKToV.length * 2;
386
387      this.hashTableKToV = createTable(newTableSize);
388      this.hashTableVToK = createTable(newTableSize);
389      this.mask = newTableSize - 1;
390      this.size = 0;
391
392      for (BiEntry<K, V> entry = firstInKeyInsertionOrder;
393          entry != null;
394          entry = entry.nextInKeyInsertionOrder) {
395        insert(entry, entry);
396      }
397      this.modCount++;
398    }
399  }

436  abstract class Itr<T extends @Nullable Object> implements Iterator<T> {
437    @CheckForNull BiEntry<K, V> next = firstInKeyInsertionOrder;
438    @CheckForNull BiEntry<K, V> toRemove = null;
439    int expectedModCount = modCount;
440    int remaining = size();
441
442    @Override
443    public boolean hasNext() {
444      if (modCount != expectedModCount) {
445        throw new ConcurrentModificationException();
446      }
447      return next != null && remaining > 0;
448    }
449
450    @Override
451    public T next() {
452      if (!hasNext()) {
453        throw new NoSuchElementException();
454      }
455
456      // requireNonNull is safe because of the hasNext check.
457      BiEntry<K, V> entry = requireNonNull(next);
458      next = entry.nextInKeyInsertionOrder;
459      toRemove = entry;
460      remaining--;
461      return output(entry);
462    }
463
464    @Override
465    public void remove() {
466      if (modCount != expectedModCount) {
467        throw new ConcurrentModificationException();
468      }
469      if (toRemove == null) {
470        throw new IllegalStateException("no calls to next() since the last call to remove()");
471      }
472      delete(toRemove);
473      expectedModCount = modCount;
474      toRemove = null;
475    }
476
477    abstract T output(BiEntry<K, V> entry);
478  }

520  @Override
521  Iterator<Entry<K, V>> entryIterator() {
522    return new Itr<Entry<K, V>>() {
523      @Override
524      Entry<K, V> output(BiEntry<K, V> entry) {
525        return new MapEntry(entry);
526      }
527
528      class MapEntry extends AbstractMapEntry<K, V> {
529        BiEntry<K, V> delegate;
530
531        MapEntry(BiEntry<K, V> entry) {
532          this.delegate = entry;
533        }
534
535        @Override
536        @ParametricNullness
537        public K getKey() {
538          return delegate.key;
539        }
540
541        @Override
542        @ParametricNullness
543        public V getValue() {
544          return delegate.value;
545        }
546
547        @Override
548        @ParametricNullness
549        public V setValue(@ParametricNullness V value) {
550          V oldValue = delegate.value;
551          int valueHash = smearedHash(value);
552          if (valueHash == delegate.valueHash && Objects.equal(value, oldValue)) {
553            return value;
554          }
555          checkArgument(seekByValue(value, valueHash) == null, "value already present: %s", value);
556          delete(delegate);
557          BiEntry<K, V> newEntry = new BiEntry<>(delegate.key, delegate.keyHash, value, valueHash);
558          insert(newEntry, delegate);
559          delegate.prevInKeyInsertionOrder = null;
560          delegate.nextInKeyInsertionOrder = null;
561          expectedModCount = modCount;
562          if (toRemove == delegate) {
563            toRemove = newEntry;
564          }
565          delegate = newEntry;
566          return oldValue;
567        }
568      }
569    };
570  }

572  @Override
573  public void forEach(BiConsumer<? super K, ? super V> action) {
574    checkNotNull(action);
575    for (BiEntry<K, V> entry = firstInKeyInsertionOrder;
576        entry != null;
577        entry = entry.nextInKeyInsertionOrder) {
578      action.accept(entry.key, entry.value);
579    }
580  }

582  @Override
583  public void replaceAll(BiFunction<? super K, ? super V, ? extends V> function) {
584    checkNotNull(function);
585    BiEntry<K, V> oldFirst = firstInKeyInsertionOrder;
586    clear();
587    for (BiEntry<K, V> entry = oldFirst; entry != null; entry = entry.nextInKeyInsertionOrder) {
588      put(entry.key, function.apply(entry.key, entry.value));
589    }
590  }

733            delete(delegate);
