017package com.google.common.io;
018
019import com.google.common.annotations.Beta;
020import com.google.common.annotations.GwtIncompatible;
021import com.google.common.annotations.J2ktIncompatible;
022import com.google.j2objc.annotations.J2ObjCIncompatible;
023import java.nio.file.SecureDirectoryStream;
024
025/**
026 * Options for use with recursive delete methods ({@link MoreFiles#deleteRecursively} and {@link
027 * MoreFiles#deleteDirectoryContents}).
028 *
029 * @since 21.0
030 * @author Colin Decker
031 */
032@Beta
033@J2ktIncompatible
034@GwtIncompatible
035@J2ObjCIncompatible // java.nio.file
036@ElementTypesAreNonnullByDefault
037public enum RecursiveDeleteOption {
038  /**
039   * Specifies that the recursive delete should not throw an exception when it can't be guaranteed
040   * that it can be done securely, without vulnerability to race conditions (i.e. when the file
041   * system does not support {@link SecureDirectoryStream}).
042   *
043   * <p><b>Warning:</b> On a file system that supports symbolic links, it is possible for an
044   * insecure recursive delete to delete files and directories that are <i>outside</i> the directory
045   * being deleted. This can happen if, after checking that a file is a directory (and not a
046   * symbolic link), that directory is deleted and replaced by a symbolic link to an outside
047   * directory before the call that opens the directory to read its entries. File systems that
048   * support {@code SecureDirectoryStream} do not have this vulnerability.
049   */
050  ALLOW_INSECURE
051}
