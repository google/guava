038/**
039 * A class for arithmetic on values of type {@code long}. Where possible, methods are defined and
040 * named analogously to their {@code BigInteger} counterparts.
041 *
042 * <p>The implementations of many methods in this class are based on material from Henry S. Warren,
043 * Jr.'s <i>Hacker's Delight</i>, (Addison Wesley, 2002).
044 *
045 * <p>Similar functionality for {@code int} and for {@link BigInteger} can be found in {@link
046 * IntMath} and {@link BigIntegerMath} respectively. For other common operations on {@code long}
047 * values, see {@link com.google.common.primitives.Longs}.
048 *
049 * @author Louis Wasserman
050 * @since 11.0
051 */
052@GwtCompatible(emulated = true)

057  @VisibleForTesting static final long MAX_SIGNED_POWER_OF_TWO = 1L << (Long.SIZE - 2);

059  /**
060   * Returns the smallest power of two greater than or equal to {@code x}. This is equivalent to
061   * {@code checkedPow(2, log2(x, CEILING))}.
062   *
063   * @throws IllegalArgumentException if {@code x <= 0}
064   * @throws ArithmeticException of the next-higher power of two is not representable as a {@code
065   *     long}, i.e. when {@code x > 2^62}
066   * @since 20.0
067   */
068  @Beta
069  public static long ceilingPowerOfTwo(long x) {
070    checkPositive("x", x);
071    if (x > MAX_SIGNED_POWER_OF_TWO) {
072      throw new ArithmeticException("ceilingPowerOfTwo(" + x + ") is not representable as a long");
073    }
074    return 1L << -Long.numberOfLeadingZeros(x - 1);
075  }

077  /**
078   * Returns the largest power of two less than or equal to {@code x}. This is equivalent to {@code
079   * checkedPow(2, log2(x, FLOOR))}.
080   *
081   * @throws IllegalArgumentException if {@code x <= 0}
082   * @since 20.0
083   */
084  @Beta
085  public static long floorPowerOfTwo(long x) {
086    checkPositive("x", x);
087
088    // Long.highestOneBit was buggy on GWT.  We've fixed it, but I'm not certain when the fix will
089    // be released.
090    return 1L << ((Long.SIZE - 1) - Long.numberOfLeadingZeros(x));
091  }

093  /**
094   * Returns {@code true} if {@code x} represents a power of two.
095   *
096   * <p>This differs from {@code Long.bitCount(x) == 1}, because {@code
097   * Long.bitCount(Long.MIN_VALUE) == 1}, but {@link Long#MIN_VALUE} is not a power of two.
098   */
099  @SuppressWarnings("ShortCircuitBoolean")
100  public static boolean isPowerOfTwo(long x) {
101    return x > 0 & (x & (x - 1)) == 0;
102  }

104  /**
105   * Returns 1 if {@code x < y} as unsigned longs, and 0 otherwise. Assumes that x - y fits into a
106   * signed long. The implementation is branch-free, and benchmarks suggest it is measurably faster
107   * than the straightforward ternary expression.
108   */
109  @VisibleForTesting
110  static int lessThanBranchFree(long x, long y) {
111    // Returns the sign bit of x - y.
112    return (int) (~~(x - y) >>> (Long.SIZE - 1));
113  }

115  /**
116   * Returns the base-2 logarithm of {@code x}, rounded according to the specified rounding mode.
117   *
118   * @throws IllegalArgumentException if {@code x <= 0}
119   * @throws ArithmeticException if {@code mode} is {@link RoundingMode#UNNECESSARY} and {@code x}
120   *     is not a power of two
121   */
122  @SuppressWarnings("fallthrough")
123  // TODO(kevinb): remove after this warning is disabled globally
124  public static int log2(long x, RoundingMode mode) {
125    checkPositive("x", x);
126    switch (mode) {
127      case UNNECESSARY:
128        checkRoundingUnnecessary(isPowerOfTwo(x));
129        // fall through
130      case DOWN:
131      case FLOOR:
132        return (Long.SIZE - 1) - Long.numberOfLeadingZeros(x);
133
134      case UP:
135      case CEILING:
136        return Long.SIZE - Long.numberOfLeadingZeros(x - 1);
137
138      case HALF_DOWN:
139      case HALF_UP:
140      case HALF_EVEN:
141        // Since sqrt(2) is irrational, log2(x) - logFloor cannot be exactly 0.5
142        int leadingZeros = Long.numberOfLeadingZeros(x);
143        long cmp = MAX_POWER_OF_SQRT2_UNSIGNED >>> leadingZeros;
144        // floor(2^(logFloor + 0.5))
145        int logFloor = (Long.SIZE - 1) - leadingZeros;
146        return logFloor + lessThanBranchFree(cmp, x);
147    }
148    throw new AssertionError("impossible");
149  }

154  /**
155   * Returns the base-10 logarithm of {@code x}, rounded according to the specified rounding mode.
156   *
157   * @throws IllegalArgumentException if {@code x <= 0}
158   * @throws ArithmeticException if {@code mode} is {@link RoundingMode#UNNECESSARY} and {@code x}
159   *     is not a power of ten
160   */
161  @J2ktIncompatible
162  @GwtIncompatible // TODO
163  @SuppressWarnings("fallthrough")
164  // TODO(kevinb): remove after this warning is disabled globally
165  public static int log10(long x, RoundingMode mode) {
166    checkPositive("x", x);
167    int logFloor = log10Floor(x);
168    long floorPow = powersOf10[logFloor];
169    switch (mode) {
170      case UNNECESSARY:
171        checkRoundingUnnecessary(x == floorPow);
172        // fall through
173      case FLOOR:
174      case DOWN:
175        return logFloor;
176      case CEILING:
177      case UP:
178        return logFloor + lessThanBranchFree(floorPow, x);
179      case HALF_DOWN:
180      case HALF_UP:
181      case HALF_EVEN:
182        // sqrt(10) is irrational, so log10(x)-logFloor is never exactly 0.5
183        return logFloor + lessThanBranchFree(halfPowersOf10[logFloor], x);
184    }
185    throw new AssertionError();
186  }

188  @J2ktIncompatible
189  @GwtIncompatible // TODO
190  static int log10Floor(long x) {
191    /*
192     * Based on Hacker's Delight Fig. 11-5, the two-table-lookup, branch-free implementation.
193     *
194     * The key idea is that based on the number of leading zeros (equivalently, floor(log2(x))), we
195     * can narrow the possible floor(log10(x)) values to two. For example, if floor(log2(x)) is 6,
196     * then 64 <= x < 128, so floor(log10(x)) is either 1 or 2.
197     */
198    int y = maxLog10ForLeadingZeros[Long.numberOfLeadingZeros(x)];
199    /*
200     * y is the higher of the two possible values of floor(log10(x)). If x < 10^y, then we want the
201     * lower of the two possible values, or y - 1, otherwise, we want y.
202     */
203    return y - lessThanBranchFree(x, powersOf10[y]);
204  }

265  /**
266   * Returns {@code b} to the {@code k}th power. Even if the result overflows, it will be equal to
267   * {@code BigInteger.valueOf(b).pow(k).longValue()}. This implementation runs in {@code O(log k)}
268   * time.
269   *
270   * @throws IllegalArgumentException if {@code k < 0}
271   */
272  @J2ktIncompatible
273  @GwtIncompatible // TODO
274  public static long pow(long b, int k) {
275    checkNonNegative("exponent", k);
276    if (-2 <= b && b <= 2) {
277      switch ((int) b) {
278        case 0:
279          return (k == 0) ? 1 : 0;
280        case 1:
281          return 1;
282        case (-1):
283          return ((k & 1) == 0) ? 1 : -1;
284        case 2:
285          return (k < Long.SIZE) ? 1L << k : 0;
286        case (-2):
287          if (k < Long.SIZE) {
288            return ((k & 1) == 0) ? 1L << k : -(1L << k);
289          } else {
290            return 0;
291          }
292        default:
293          throw new AssertionError();
294      }
295    }
296    for (long accum = 1; ; k >>= 1) {
297      switch (k) {
298        case 0:
299          return accum;
300        case 1:
301          return accum * b;
302        default:
303          accum *= ((k & 1) == 0) ? 1 : b;
304          b *= b;
305      }
306    }
307  }

309  /**
310   * Returns the square root of {@code x}, rounded with the specified rounding mode.
311   *
312   * @throws IllegalArgumentException if {@code x < 0}
313   * @throws ArithmeticException if {@code mode} is {@link RoundingMode#UNNECESSARY} and {@code
314   *     sqrt(x)} is not an integer
315   */
316  @J2ktIncompatible
317  @GwtIncompatible // TODO
318  public static long sqrt(long x, RoundingMode mode) {
319    checkNonNegative("x", x);
320    if (fitsInInt(x)) {
321      return IntMath.sqrt((int) x, mode);
322    }
323    /*
324     * Let k be the true value of floor(sqrt(x)), so that
325     *
326     *            k * k <= x          <  (k + 1) * (k + 1)
327     * (double) (k * k) <= (double) x <= (double) ((k + 1) * (k + 1))
328     *          since casting to double is nondecreasing.
329     *          Note that the right-hand inequality is no longer strict.
330     * Math.sqrt(k * k) <= Math.sqrt(x) <= Math.sqrt((k + 1) * (k + 1))
331     *          since Math.sqrt is monotonic.
332     * (long) Math.sqrt(k * k) <= (long) Math.sqrt(x) <= (long) Math.sqrt((k + 1) * (k + 1))
333     *          since casting to long is monotonic
334     * k <= (long) Math.sqrt(x) <= k + 1
335     *          since (long) Math.sqrt(k * k) == k, as checked exhaustively in
336     *          {@link LongMathTest#testSqrtOfPerfectSquareAsDoubleIsPerfect}
337     */
338    long guess = (long) Math.sqrt((double) x);
339    // Note: guess is always <= FLOOR_SQRT_MAX_LONG.
340    long guessSquared = guess * guess;
341    // Note (2013-2-26): benchmarks indicate that, inscrutably enough, using if statements is
342    // faster here than using lessThanBranchFree.
343    switch (mode) {
344      case UNNECESSARY:
345        checkRoundingUnnecessary(guessSquared == x);
346        return guess;
347      case FLOOR:
348      case DOWN:
349        if (x < guessSquared) {
350          return guess - 1;
351        }
352        return guess;
353      case CEILING:
354      case UP:
355        if (x > guessSquared) {
356          return guess + 1;
357        }
358        return guess;
359      case HALF_DOWN:
360      case HALF_UP:
361      case HALF_EVEN:
362        long sqrtFloor = guess - ((x < guessSquared) ? 1 : 0);
363        long halfSquare = sqrtFloor * sqrtFloor + sqrtFloor;
364        /*
365         * We wish to test whether or not x <= (sqrtFloor + 0.5)^2 = halfSquare + 0.25. Since both x
366         * and halfSquare are integers, this is equivalent to testing whether or not x <=
367         * halfSquare. (We have to deal with overflow, though.)
368         *
369         * If we treat halfSquare as an unsigned long, we know that
370         *            sqrtFloor^2 <= x < (sqrtFloor + 1)^2
371         * halfSquare - sqrtFloor <= x < halfSquare + sqrtFloor + 1
372         * so |x - halfSquare| <= sqrtFloor.  Therefore, it's safe to treat x - halfSquare as a
373         * signed long, so lessThanBranchFree is safe for use.
374         */
375        return sqrtFloor + lessThanBranchFree(halfSquare, x);
376    }
377    throw new AssertionError();
378  }

380  /**
381   * Returns the result of dividing {@code p} by {@code q}, rounding using the specified {@code
382   * RoundingMode}.
383   *
384   * @throws ArithmeticException if {@code q == 0}, or if {@code mode == UNNECESSARY} and {@code a}
385   *     is not an integer multiple of {@code b}
386   */
387  @J2ktIncompatible
388  @GwtIncompatible // TODO
389  @SuppressWarnings("fallthrough")
390  public static long divide(long p, long q, RoundingMode mode) {
391    checkNotNull(mode);
392    long div = p / q; // throws if q == 0
393    long rem = p - q * div; // equals p % q
394
395    if (rem == 0) {
396      return div;
397    }
398
399    /*
400     * Normal Java division rounds towards 0, consistently with RoundingMode.DOWN. We just have to
401     * deal with the cases where rounding towards 0 is wrong, which typically depends on the sign of
402     * p / q.
403     *
404     * signum is 1 if p and q are both nonnegative or both negative, and -1 otherwise.
405     */
406    int signum = 1 | (int) ((p ^ q) >> (Long.SIZE - 1));
407    boolean increment;
408    switch (mode) {
409      case UNNECESSARY:
410        checkRoundingUnnecessary(rem == 0);
411        // fall through
412      case DOWN:
413        increment = false;
414        break;
415      case UP:
416        increment = true;
417        break;
418      case CEILING:
419        increment = signum > 0;
420        break;
421      case FLOOR:
422        increment = signum < 0;
423        break;
424      case HALF_EVEN:
425      case HALF_DOWN:
426      case HALF_UP:
427        long absRem = abs(rem);
428        long cmpRemToHalfDivisor = absRem - (abs(q) - absRem);
429        // subtracting two nonnegative longs can't overflow
430        // cmpRemToHalfDivisor has the same sign as compare(abs(rem), abs(q) / 2).
431        if (cmpRemToHalfDivisor == 0) { // exactly on the half mark
432          increment = (mode == HALF_UP || (mode == HALF_EVEN && (div & 1) != 0));
433        } else {
434          increment = cmpRemToHalfDivisor > 0; // closer to the UP value
435        }
436        break;
437      default:
438        throw new AssertionError();
439    }
440    return increment ? div + signum : div;
441  }

496  /**
497   * Returns the greatest common divisor of {@code a, b}. Returns {@code 0} if {@code a == 0 && b ==
498   * 0}.
499   *
500   * @throws IllegalArgumentException if {@code a < 0} or {@code b < 0}
501   */
502  public static long gcd(long a, long b) {
503    /*
504     * The reason we require both arguments to be >= 0 is because otherwise, what do you return on
505     * gcd(0, Long.MIN_VALUE)? BigInteger.gcd would return positive 2^63, but positive 2^63 isn't an
506     * int.
507     */
508    checkNonNegative("a", a);
509    checkNonNegative("b", b);
510    if (a == 0) {
511      // 0 % b == 0, so b divides a, but the converse doesn't hold.
512      // BigInteger.gcd is consistent with this decision.
513      return b;
514    } else if (b == 0) {
515      return a; // similar logic
516    }
517    /*
518     * Uses the binary GCD algorithm; see http://en.wikipedia.org/wiki/Binary_GCD_algorithm. This is
519     * >60% faster than the Euclidean algorithm in benchmarks.
520     */
521    int aTwos = Long.numberOfTrailingZeros(a);
522    a >>= aTwos; // divide out all 2s
523    int bTwos = Long.numberOfTrailingZeros(b);
524    b >>= bTwos; // divide out all 2s
525    while (a != b) { // both a, b are odd
526      // The key to the binary GCD algorithm is as follows:
527      // Both a and b are odd. Assume a > b; then gcd(a - b, b) = gcd(a, b).
528      // But in gcd(a - b, b), a - b is even and b is odd, so we can divide out powers of two.
529
530      // We bend over backwards to avoid branching, adapting a technique from
531      // http://graphics.stanford.edu/~seander/bithacks.html#IntegerMinOrMax
532
533      long delta = a - b; // can't overflow, since a and b are nonnegative
534
535      long minDeltaOrZero = delta & (delta >> (Long.SIZE - 1));
536      // equivalent to Math.min(delta, 0)
537
538      a = delta - minDeltaOrZero - minDeltaOrZero; // sets a to Math.abs(a - b)
539      // a is now nonnegative and even
540
541      b += minDeltaOrZero; // sets b to min(old a, b)
542      a >>= Long.numberOfTrailingZeros(a); // divide out all 2s, since 2 doesn't divide b
543    }
544    return a << min(aTwos, bTwos);
545  }

573  /**
574   * Returns the product of {@code a} and {@code b}, provided it does not overflow.
575   *
576   * @throws ArithmeticException if {@code a * b} overflows in signed {@code long} arithmetic
577   */
578  @SuppressWarnings("ShortCircuitBoolean")
579  public static long checkedMultiply(long a, long b) {
580    // Hacker's Delight, Section 2-12
581    int leadingZeros =
582        Long.numberOfLeadingZeros(a)
583            + Long.numberOfLeadingZeros(~a)
584            + Long.numberOfLeadingZeros(b)
585            + Long.numberOfLeadingZeros(~b);
586    /*
587     * If leadingZeros > Long.SIZE + 1 it's definitely fine, if it's < Long.SIZE it's definitely
588     * bad. We do the leadingZeros check to avoid the division below if at all possible.
589     *
590     * Otherwise, if b == Long.MIN_VALUE, then the only allowed values of a are 0 and 1. We take
591     * care of all a < 0 with their own check, because in particular, the case a == -1 will
592     * incorrectly pass the division check below.
593     *
594     * In all other cases, we check that either a is 0 or the result is consistent with division.
595     */
596    if (leadingZeros > Long.SIZE + 1) {
597      return a * b;
598    }
599    checkNoOverflow(leadingZeros >= Long.SIZE, "checkedMultiply", a, b);
600    checkNoOverflow(a >= 0 | b != Long.MIN_VALUE, "checkedMultiply", a, b);
601    long result = a * b;
602    checkNoOverflow(a == 0 || result / a == b, "checkedMultiply", a, b);
603    return result;
604  }

612  @J2ktIncompatible
613  @GwtIncompatible // TODO

615  public static long checkedPow(long b, int k) {
616    checkNonNegative("exponent", k);
617    if (b >= -2 & b <= 2) {
618      switch ((int) b) {
619        case 0:
620          return (k == 0) ? 1 : 0;
621        case 1:
622          return 1;
623        case (-1):
624          return ((k & 1) == 0) ? 1 : -1;
625        case 2:
626          checkNoOverflow(k < Long.SIZE - 1, "checkedPow", b, k);
627          return 1L << k;
628        case (-2):
629          checkNoOverflow(k < Long.SIZE, "checkedPow", b, k);
630          return ((k & 1) == 0) ? (1L << k) : (-1L << k);
631        default:
632          throw new AssertionError();
633      }
634    }
635    long accum = 1;
636    while (true) {
637      switch (k) {
638        case 0:
639          return accum;
640        case 1:
641          return checkedMultiply(accum, b);
642        default:
643          if ((k & 1) != 0) {
644            accum = checkedMultiply(accum, b);
645          }
646          k >>= 1;
647          if (k > 0) {
648            checkNoOverflow(
649                -FLOOR_SQRT_MAX_LONG <= b && b <= FLOOR_SQRT_MAX_LONG, "checkedPow", b, k);
650            b *= b;
651          }
652      }
653    }
654  }

656  /**
657   * Returns the sum of {@code a} and {@code b} unless it would overflow or underflow in which case
658   * {@code Long.MAX_VALUE} or {@code Long.MIN_VALUE} is returned, respectively.
659   *
660   * @since 20.0
661   */
662  @Beta
663  @SuppressWarnings("ShortCircuitBoolean")
664  public static long saturatedAdd(long a, long b) {
665    long naiveSum = a + b;
666    if ((a ^ b) < 0 | (a ^ naiveSum) >= 0) {
667      // If a and b have different signs or a has the same sign as the result then there was no
668      // overflow, return.
669      return naiveSum;
670    }
671    // we did over/under flow, if the sign is negative we should return MAX otherwise MIN
672    return Long.MAX_VALUE + ((naiveSum >>> (Long.SIZE - 1)) ^ 1);
673  }

675  /**
676   * Returns the difference of {@code a} and {@code b} unless it would overflow or underflow in
677   * which case {@code Long.MAX_VALUE} or {@code Long.MIN_VALUE} is returned, respectively.
678   *
679   * @since 20.0
680   */
681  @Beta
682  @SuppressWarnings("ShortCircuitBoolean")
683  public static long saturatedSubtract(long a, long b) {
684    long naiveDifference = a - b;
685    if ((a ^ b) >= 0 | (a ^ naiveDifference) >= 0) {
686      // If a and b have the same signs or a has the same sign as the result then there was no
687      // overflow, return.
688      return naiveDifference;
689    }
690    // we did over/under flow
691    return Long.MAX_VALUE + ((naiveDifference >>> (Long.SIZE - 1)) ^ 1);
692  }

694  /**
695   * Returns the product of {@code a} and {@code b} unless it would overflow or underflow in which
696   * case {@code Long.MAX_VALUE} or {@code Long.MIN_VALUE} is returned, respectively.
697   *
698   * @since 20.0
699   */
700  @Beta
701  @SuppressWarnings("ShortCircuitBoolean")
702  public static long saturatedMultiply(long a, long b) {
703    // see checkedMultiply for explanation
704    int leadingZeros =
705        Long.numberOfLeadingZeros(a)
706            + Long.numberOfLeadingZeros(~a)
707            + Long.numberOfLeadingZeros(b)
708            + Long.numberOfLeadingZeros(~b);
709    if (leadingZeros > Long.SIZE + 1) {
710      return a * b;
711    }
712    // the return value if we will overflow (which we calculate by overflowing a long :) )
713    long limit = Long.MAX_VALUE + ((a ^ b) >>> (Long.SIZE - 1));
714    if (leadingZeros < Long.SIZE | (a < 0 & b == Long.MIN_VALUE)) {
715      // overflow
716      return limit;
717    }
718    long result = a * b;
719    if (a == 0 || result / a == b) {
720      return result;
721    }
722    return limit;
723  }

725  /**
726   * Returns the {@code b} to the {@code k}th power, unless it would overflow or underflow in which
727   * case {@code Long.MAX_VALUE} or {@code Long.MIN_VALUE} is returned, respectively.
728   *
729   * @since 20.0
730   */
731  @Beta
732  @SuppressWarnings("ShortCircuitBoolean")
733  public static long saturatedPow(long b, int k) {
734    checkNonNegative("exponent", k);
735    if (b >= -2 & b <= 2) {
736      switch ((int) b) {
737        case 0:
738          return (k == 0) ? 1 : 0;
739        case 1:
740          return 1;
741        case (-1):
742          return ((k & 1) == 0) ? 1 : -1;
743        case 2:
744          if (k >= Long.SIZE - 1) {
745            return Long.MAX_VALUE;
746          }
747          return 1L << k;
748        case (-2):
749          if (k >= Long.SIZE) {
750            return Long.MAX_VALUE + (k & 1);
751          }
752          return ((k & 1) == 0) ? (1L << k) : (-1L << k);
753        default:
754          throw new AssertionError();
755      }
756    }
757    long accum = 1;
758    // if b is negative and k is odd then the limit is MIN otherwise the limit is MAX
759    long limit = Long.MAX_VALUE + ((b >>> (Long.SIZE - 1)) & (k & 1));
760    while (true) {
761      switch (k) {
762        case 0:
763          return accum;
764        case 1:
765          return saturatedMultiply(accum, b);
766        default:
767          if ((k & 1) != 0) {
768            accum = saturatedMultiply(accum, b);
769          }
770          k >>= 1;
771          if (k > 0) {
772            if (-FLOOR_SQRT_MAX_LONG > b | b > FLOOR_SQRT_MAX_LONG) {
773              return limit;
774            }
775            b *= b;
776          }
777      }
778    }

820  /**
821   * Returns {@code n} choose {@code k}, also known as the binomial coefficient of {@code n} and
822   * {@code k}, or {@link Long#MAX_VALUE} if the result does not fit in a {@code long}.
823   *
824   * @throws IllegalArgumentException if {@code n < 0}, {@code k < 0}, or {@code k > n}
825   */
826  public static long binomial(int n, int k) {
827    checkNonNegative("n", n);
828    checkNonNegative("k", k);
829    checkArgument(k <= n, "k (%s) > n (%s)", k, n);
830    if (k > (n >> 1)) {
831      k = n - k;
832    }
833    switch (k) {
834      case 0:
835        return 1;
836      case 1:
837        return n;
838      default:
839        if (n < factorials.length) {
840          return factorials[n] / (factorials[k] * factorials[n - k]);
841        } else if (k >= biggestBinomials.length || n > biggestBinomials[k]) {
842          return Long.MAX_VALUE;
843        } else if (k < biggestSimpleBinomials.length && n <= biggestSimpleBinomials[k]) {
844          // guaranteed not to overflow
845          long result = n--;
846          for (int i = 2; i <= k; n--, i++) {
847            result *= n;
848            result /= i;
849          }
850          return result;
851        } else {
852          int nBits = LongMath.log2(n, RoundingMode.CEILING);
853
854          long result = 1;
855          long numerator = n--;
856          long denominator = 1;
857
858          int numeratorBits = nBits;
859          // This is an upper bound on log2(numerator, ceiling).
860
861          /*
862           * We want to do this in long math for speed, but want to avoid overflow. We adapt the
863           * technique previously used by BigIntegerMath: maintain separate numerator and
864           * denominator accumulators, multiplying the fraction into result when near overflow.
865           */
866          for (int i = 2; i <= k; i++, n--) {
867            if (numeratorBits + nBits < Long.SIZE - 1) {
868              // It's definitely safe to multiply into numerator and denominator.
869              numerator *= n;
870              denominator *= i;
871              numeratorBits += nBits;
872            } else {
873              // It might not be safe to multiply into numerator and denominator,
874              // so multiply (numerator / denominator) into result.
875              result = multiplyFraction(result, numerator, denominator);
876              numerator = n;
877              denominator = i;
878              numeratorBits = nBits;
879            }
880          }
881          return multiplyFraction(result, numerator, denominator);
882        }
883    }
884  }
885
886  /** Returns (x * numerator / denominator), which is assumed to come out to an integral value. */
887  static long multiplyFraction(long x, long numerator, long denominator) {
888    if (x == 1) {
889      return numerator / denominator;
890    }
891    long commonDivisor = gcd(x, denominator);
892    x /= commonDivisor;
893    denominator /= commonDivisor;
894    // We know gcd(x, denominator) = 1, and x * numerator / denominator is exact,
895    // so denominator must be a divisor of numerator.
896    return x * (numerator / denominator);
897  }

1006  /**
1007   * Returns {@code true} if {@code n} is a <a
1008   * href="http://mathworld.wolfram.com/PrimeNumber.html">prime number</a>: an integer <i>greater
1009   * than one</i> that cannot be factored into a product of <i>smaller</i> positive integers.
1010   * Returns {@code false} if {@code n} is zero, one, or a composite number (one which <i>can</i> be
1011   * factored into smaller positive integers).
1012   *
1013   * <p>To test larger numbers, use {@link BigInteger#isProbablePrime}.
1014   *
1015   * @throws IllegalArgumentException if {@code n} is negative
1016   * @since 20.0
1017   */
1018  @J2ktIncompatible
1019  @GwtIncompatible // TODO
1020  @Beta
1021  public static boolean isPrime(long n) {
1022    if (n < 2) {
1023      checkNonNegative("n", n);
1024      return false;
1025    }
1026    if (n < 66) {
1027      // Encode all primes less than 66 into mask without 0 and 1.
1028      long mask =
1029          (1L << (2 - 2))
1030              | (1L << (3 - 2))
1031              | (1L << (5 - 2))
1032              | (1L << (7 - 2))
1033              | (1L << (11 - 2))
1034              | (1L << (13 - 2))
1035              | (1L << (17 - 2))
1036              | (1L << (19 - 2))
1037              | (1L << (23 - 2))
1038              | (1L << (29 - 2))
1039              | (1L << (31 - 2))
1040              | (1L << (37 - 2))
1041              | (1L << (41 - 2))
1042              | (1L << (43 - 2))
1043              | (1L << (47 - 2))
1044              | (1L << (53 - 2))
1045              | (1L << (59 - 2))
1046              | (1L << (61 - 2));
1047      // Look up n within the mask.
1048      return ((mask >> ((int) n - 2)) & 1) != 0;
1049    }
1050
1051    if ((SIEVE_30 & (1 << (n % 30))) != 0) {
1052      return false;
1053    }
1054    if (n % 7 == 0 || n % 11 == 0 || n % 13 == 0) {
1055      return false;
1056    }
1057    if (n < 17 * 17) {
1058      return true;
1059    }
1060
1061    for (long[] baseSet : millerRabinBaseSets) {
1062      if (n <= baseSet[0]) {
1063        for (int i = 1; i < baseSet.length; i++) {
1064          if (!MillerRabinTester.test(baseSet[i], n)) {
1065            return false;
1066          }
1067        }
1068        return true;
1069      }
1070    }
1071    throw new AssertionError();
1072  }

1106  private enum MillerRabinTester {
1107    /** Works for inputs  FLOOR_SQRT_MAX_LONG. */
1108    SMALL {
1109      @Override
1110      long mulMod(long a, long b, long m) {
1111        /*
1112         * lowasser, 2015-Feb-12: Benchmarks suggest that changing this to UnsignedLongs.remainder
1113         * and increasing the threshold to 2^32 doesn't pay for itself, and adding another enum
1114         * constant hurts performance further -- I suspect because bimorphic implementation is a
1115         * sweet spot for the JVM.
1116         */
1117        return (a * b) % m;
1118      }
1119
1120      @Override
1121      long squareMod(long a, long m) {
1122        return (a * a) % m;
1123      }
1124    },
1125    /** Works for all nonnegative signed longs. */
1126    LARGE {
1127      /** Returns (a + b) mod m. Precondition: {@code 0 <= a}, {@code b < m < 2^63}. */
1128      private long plusMod(long a, long b, long m) {
1129        return (a >= m - b) ? (a + b - m) : (a + b);
1130      }
1131
1132      /** Returns (a * 2^32) mod m. a may be any unsigned long. */
1133      private long times2ToThe32Mod(long a, long m) {
1134        int remainingPowersOf2 = 32;
1135        do {
1136          int shift = min(remainingPowersOf2, Long.numberOfLeadingZeros(a));
1137          // shift is either the number of powers of 2 left to multiply a by, or the biggest shift
1138          // possible while keeping a in an unsigned long.
1139          a = UnsignedLongs.remainder(a << shift, m);
1140          remainingPowersOf2 -= shift;
1141        } while (remainingPowersOf2 > 0);
1142        return a;
1143      }
1144
1145      @Override
1146      long mulMod(long a, long b, long m) {
1147        long aHi = a >>> 32; // < 2^31
1148        long bHi = b >>> 32; // < 2^31
1149        long aLo = a & 0xFFFFFFFFL; // < 2^32
1150        long bLo = b & 0xFFFFFFFFL; // < 2^32
1151
1152        /*
1153         * a * b == aHi * bHi * 2^64 + (aHi * bLo + aLo * bHi) * 2^32 + aLo * bLo.
1154         *       == (aHi * bHi * 2^32 + aHi * bLo + aLo * bHi) * 2^32 + aLo * bLo
1155         *
1156         * We carry out this computation in modular arithmetic. Since times2ToThe32Mod accepts any
1157         * unsigned long, we don't have to do a mod on every operation, only when intermediate
1158         * results can exceed 2^63.
1159         */
1160        long result = times2ToThe32Mod(aHi * bHi /* < 2^62 */, m); // < m < 2^63
1161        result += aHi * bLo; // aHi * bLo < 2^63, result < 2^64
1162        if (result < 0) {
1163          result = UnsignedLongs.remainder(result, m);
1164        }
1165        // result < 2^63 again
1166        result += aLo * bHi; // aLo * bHi < 2^63, result < 2^64
1167        result = times2ToThe32Mod(result, m); // result < m < 2^63
1168        return plusMod(result, UnsignedLongs.remainder(aLo * bLo /* < 2^64 */, m), m);
1169      }
1170
1171      @Override
1172      long squareMod(long a, long m) {
1173        long aHi = a >>> 32; // < 2^31
1174        long aLo = a & 0xFFFFFFFFL; // < 2^32
1175
1176        /*
1177         * a^2 == aHi^2 * 2^64 + aHi * aLo * 2^33 + aLo^2
1178         *     == (aHi^2 * 2^32 + aHi * aLo * 2) * 2^32 + aLo^2
1179         * We carry out this computation in modular arithmetic.  Since times2ToThe32Mod accepts any
1180         * unsigned long, we don't have to do a mod on every operation, only when intermediate
1181         * results can exceed 2^63.
1182         */
1183        long result = times2ToThe32Mod(aHi * aHi /* < 2^62 */, m); // < m < 2^63
1184        long hiLo = aHi * aLo * 2;
1185        if (hiLo < 0) {
1186          hiLo = UnsignedLongs.remainder(hiLo, m);
1187        }
1188        // hiLo < 2^63
1189        result += hiLo; // result < 2^64
1190        result = times2ToThe32Mod(result, m); // result < m < 2^63
1191        return plusMod(result, UnsignedLongs.remainder(aLo * aLo /* < 2^64 */, m), m);
1192      }
1193    };

1195    static boolean test(long base, long n) {
1196      // Since base will be considered % n, it's okay if base > FLOOR_SQRT_MAX_LONG,
1197      // so long as n <= FLOOR_SQRT_MAX_LONG.
1198      return ((n <= FLOOR_SQRT_MAX_LONG) ? SMALL : LARGE).testWitness(base, n);
1199    }

1207    /** Returns a^p mod m. */
1208    private long powMod(long a, long p, long m) {
1209      long res = 1;
1210      for (; p != 0; p >>= 1) {
1211        if ((p & 1) != 0) {
1212          res = mulMod(res, a, m);
1213        }
1214        a = squareMod(a, m);
1215      }
1216      return res;
1217    }

1219    /** Returns true if n is a strong probable prime relative to the specified base. */
1220    private boolean testWitness(long base, long n) {
1221      int r = Long.numberOfTrailingZeros(n - 1);
1222      long d = (n - 1) >> r;
1223      base %= n;
1224      if (base == 0) {
1225        return true;
1226      }
1227      // Calculate a := base^d mod n.
1228      long a = powMod(base, d, n);
1229      // n passes this test if
1230      //    base^d = 1 (mod n)
1231      // or base^(2^j * d) = -1 (mod n) for some 0 <= j < r.
1232      if (a == 1) {
1233        return true;
1234      }
1235      int j = 0;
1236      while (a != n - 1) {
1237        if (++j == r) {
1238          return false;
1239        }
1240        a = squareMod(a, n);
1241      }
1242      return true;
1243    }
1244  }

1246  /**
1247   * Returns {@code x}, rounded to a {@code double} with the specified rounding mode. If {@code x}
1248   * is precisely representable as a {@code double}, its {@code double} value will be returned;
1249   * otherwise, the rounding will choose between the two nearest representable values with {@code
1250   * mode}.
1251   *
1252   * <p>For the case of {@link RoundingMode#HALF_EVEN}, this implementation uses the IEEE 754
1253   * default rounding mode: if the two nearest representable values are equally near, the one with
1254   * the least significant bit zero is chosen. (In such cases, both of the nearest representable
1255   * values are even integers; this method returns the one that is a multiple of a greater power of
1256   * two.)
1257   *
1258   * @throws ArithmeticException if {@code mode} is {@link RoundingMode#UNNECESSARY} and {@code x}
1259   *     is not precisely representable as a {@code double}
1260   * @since 30.0
1261   */
1262  @SuppressWarnings("deprecation")
1263  @J2ktIncompatible
1264  @GwtIncompatible
1265  public static double roundToDouble(long x, RoundingMode mode) {
1266    // Logic adapted from ToDoubleRounder.
1267    double roundArbitrarily = (double) x;
1268    long roundArbitrarilyAsLong = (long) roundArbitrarily;
1269    int cmpXToRoundArbitrarily;
1270
1271    if (roundArbitrarilyAsLong == Long.MAX_VALUE) {
1272      /*
1273       * For most values, the conversion from roundArbitrarily to roundArbitrarilyAsLong is
1274       * lossless. In that case we can compare x to roundArbitrarily using Longs.compare(x,
1275       * roundArbitrarilyAsLong). The exception is for values where the conversion to double rounds
1276       * up to give roundArbitrarily equal to 2^63, so the conversion back to long overflows and
1277       * roundArbitrarilyAsLong is Long.MAX_VALUE. (This is the only way this condition can occur as
1278       * otherwise the conversion back to long pads with zero bits.) In this case we know that
1279       * roundArbitrarily > x. (This is important when x == Long.MAX_VALUE ==
1280       * roundArbitrarilyAsLong.)
1281       */
1282      cmpXToRoundArbitrarily = -1;
1283    } else {
1284      cmpXToRoundArbitrarily = Longs.compare(x, roundArbitrarilyAsLong);
1285    }
1286
1287    switch (mode) {
1288      case UNNECESSARY:
1289        checkRoundingUnnecessary(cmpXToRoundArbitrarily == 0);
1290        return roundArbitrarily;
1291      case FLOOR:
1292        return (cmpXToRoundArbitrarily >= 0)
1293            ? roundArbitrarily
1294            : DoubleUtils.nextDown(roundArbitrarily);
1295      case CEILING:
1296        return (cmpXToRoundArbitrarily <= 0) ? roundArbitrarily : Math.nextUp(roundArbitrarily);
1297      case DOWN:
1298        if (x >= 0) {
1299          return (cmpXToRoundArbitrarily >= 0)
1300              ? roundArbitrarily
1301              : DoubleUtils.nextDown(roundArbitrarily);
1302        } else {
1303          return (cmpXToRoundArbitrarily <= 0) ? roundArbitrarily : Math.nextUp(roundArbitrarily);
1304        }
1305      case UP:
1306        if (x >= 0) {
1307          return (cmpXToRoundArbitrarily <= 0) ? roundArbitrarily : Math.nextUp(roundArbitrarily);
1308        } else {
1309          return (cmpXToRoundArbitrarily >= 0)
1310              ? roundArbitrarily
1311              : DoubleUtils.nextDown(roundArbitrarily);
1312        }
1313      case HALF_DOWN:
1314      case HALF_UP:
1315      case HALF_EVEN:
1316        {
1317          long roundFloor;
1318          double roundFloorAsDouble;
1319          long roundCeiling;
1320          double roundCeilingAsDouble;
1321
1322          if (cmpXToRoundArbitrarily >= 0) {
1323            roundFloorAsDouble = roundArbitrarily;
1324            roundFloor = roundArbitrarilyAsLong;
1325            roundCeilingAsDouble = Math.nextUp(roundArbitrarily);
1326            roundCeiling = (long) Math.ceil(roundCeilingAsDouble);
1327          } else {
1328            roundCeilingAsDouble = roundArbitrarily;
1329            roundCeiling = roundArbitrarilyAsLong;
1330            roundFloorAsDouble = DoubleUtils.nextDown(roundArbitrarily);
1331            roundFloor = (long) Math.floor(roundFloorAsDouble);
1332          }
1333
1334          long deltaToFloor = x - roundFloor;
1335          long deltaToCeiling = roundCeiling - x;
1336
1337          if (roundCeiling == Long.MAX_VALUE) {
1338            // correct for Long.MAX_VALUE as discussed above: roundCeilingAsDouble must be 2^63, but
1339            // roundCeiling is 2^63-1.
1340            deltaToCeiling++;
1341          }
1342
1343          int diff = Longs.compare(deltaToFloor, deltaToCeiling);
1344          if (diff < 0) { // closer to floor
1345            return roundFloorAsDouble;
1346          } else if (diff > 0) { // closer to ceiling
1347            return roundCeilingAsDouble;
1348          }
1349          // halfway between the representable values; do the half-whatever logic
1350          switch (mode) {
1351            case HALF_EVEN:
1352              return ((DoubleUtils.getSignificand(roundFloorAsDouble) & 1L) == 0)
1353                  ? roundFloorAsDouble
1354                  : roundCeilingAsDouble;
1355            case HALF_DOWN:
1356              return (x >= 0) ? roundFloorAsDouble : roundCeilingAsDouble;
1357            case HALF_UP:
1358              return (x >= 0) ? roundCeilingAsDouble : roundFloorAsDouble;
1359            default:
1360              throw new AssertionError("impossible");
1361          }
1362        }
1363    }
1364    throw new AssertionError("impossible");
1365  }
