015package com.google.common.util.concurrent;
016
017import com.google.common.annotations.GwtIncompatible;
018import javax.annotation.CheckForNull;
019
020/**
021 * Unchecked version of {@link java.util.concurrent.TimeoutException}.
022 *
023 * @author Kevin Bourrillion
024 * @since 1.0
025 */
026@GwtIncompatible
027@ElementTypesAreNonnullByDefault
028public class UncheckedTimeoutException extends RuntimeException {
029  public UncheckedTimeoutException() {}
030
031  public UncheckedTimeoutException(@CheckForNull String message) {
032    super(message);
033  }
034
035  public UncheckedTimeoutException(@CheckForNull Throwable cause) {
036    super(cause);
037  }
038
039  public UncheckedTimeoutException(@CheckForNull String message, @CheckForNull Throwable cause) {
040    super(message, cause);
041  }
042
043  private static final long serialVersionUID = 0;
044}
