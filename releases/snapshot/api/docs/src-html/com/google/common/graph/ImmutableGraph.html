017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkArgument;
020import static com.google.common.base.Preconditions.checkNotNull;
021import static com.google.common.graph.GraphConstants.NETWORK_WITH_PARALLEL_EDGE;
022
023import com.google.common.annotations.Beta;
024import com.google.common.collect.ImmutableMap;
025import com.google.common.graph.DirectedNodeAdjacencies.Adjacency;
026
027import java.util.Set;
028
029/**
030 * A {@link Graph} whose relationships are constant. Instances of this class may be obtained
031 * with {@link #copyOf(Graph)}.
032 *
033 * @author James Sexton
034 * @author Joshua O'Madadhain
035 * @author Omar Darwish
036 * @param <N> Node parameter type
037 * @since 20.0
038 */
039@Beta
040public final class ImmutableGraph<N> extends AbstractConfigurableGraph<N> {
041
042  private ImmutableGraph(Graph<N> graph) {
043    super(GraphBuilder.from(graph), getNodeConnections(graph));
044  }
045
046  /**
047   * Returns an immutable copy of {@code graph}.
048   */
049  @SuppressWarnings("unchecked")
050  public static <N> ImmutableGraph<N> copyOf(Graph<N> graph) {
051    // TODO(b/28087289): we can remove this restriction when Graph supports parallel edges
052    checkArgument(!((graph instanceof Network) && ((Network<N, ?>) graph).allowsParallelEdges()),
053        NETWORK_WITH_PARALLEL_EDGE);
054    return (graph instanceof ImmutableGraph)
055        ? (ImmutableGraph<N>) graph
056        : new ImmutableGraph<N>(graph);
057  }
058
059  /**
060   * Simply returns its argument.
061   *
062   * @deprecated no need to use this
063   */
064  @Deprecated
065  public static <N> ImmutableGraph<N> copyOf(ImmutableGraph<N> graph) {
066    return checkNotNull(graph);
067  }
068
069  private static <N> ImmutableMap<N, NodeAdjacencies<N>> getNodeConnections(Graph<N> graph) {
070    // ImmutableMap.Builder maintains the order of the elements as inserted, so the map will
071    // have whatever ordering the graph's nodes do, so ImmutableSortedMap is unnecessary even if the
072    // input nodes are sorted.
073    ImmutableMap.Builder<N, NodeAdjacencies<N>> nodeConnections = ImmutableMap.builder();
074    for (N node : graph.nodes()) {
075      nodeConnections.put(node, nodeConnectionsOf(graph, node));
076    }
077    return nodeConnections.build();
078  }
079
080  private static <N> NodeAdjacencies<N> nodeConnectionsOf(Graph<N> graph, N node) {
081    return graph.isDirected()
082        ? DirectedNodeAdjacencies.ofImmutable(createAdjacencyMap(
083            graph, node), graph.predecessors(node).size(), graph.successors(node).size())
084        : UndirectedNodeAdjacencies.ofImmutable(graph.adjacentNodes(node));
085  }
086
087  private static <N> ImmutableMap<N, Adjacency> createAdjacencyMap(Graph<N> graph, N node) {
088    Set<N> predecessors = graph.predecessors(node);
089    Set<N> successors = graph.successors(node);
090    ImmutableMap.Builder<N, Adjacency> nodeAdjacencies = ImmutableMap.builder();
091    for (N adjacentNode : graph.adjacentNodes(node)) {
092      nodeAdjacencies.put(adjacentNode,
093          getAdjacency(predecessors.contains(adjacentNode), successors.contains(adjacentNode)));
094    }
095    return nodeAdjacencies.build();
096  }
097
098  private static Adjacency getAdjacency(boolean isPredecessor, boolean isSuccesor) {
099    if (isPredecessor && isSuccesor) {
100      return Adjacency.BOTH;
101    } else if (isPredecessor) {
102      return Adjacency.PRED;
103    } else if (isSuccesor) {
104      return Adjacency.SUCC;
105    } else {
106      throw new IllegalStateException();
107    }
108  }
109}
