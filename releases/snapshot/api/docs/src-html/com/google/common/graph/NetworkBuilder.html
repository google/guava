017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkArgument;
020
021import com.google.common.annotations.Beta;
022import com.google.common.base.Optional;
023
024import java.util.Comparator;
025
026/**
027 * A builder for constructing instances of {@link Network} with user-defined properties.
028 *
029 * <p>A graph built by this class will have the following properties by default:
030 * <ul>
031 * <li>does not allow parallel edges
032 * <li>allows self-loops
033 * </ul>
034 *
035 * @author James Sexton
036 * @author Joshua O'Madadhain
037 * @since 20.0
038 */
039// TODO(b/24620028): Add support for sorted nodes/edges. Use the same pattern as CacheBuilder
040// to narrow the generic <N, E> type when Comparators are provided.
041// TODO(user): try creating an abstract superclass that this and GraphBuilder could derive from.
042@Beta
043public final class NetworkBuilder<N, E> {
044  final boolean directed;
045  boolean allowsParallelEdges = false;
046  boolean allowsSelfLoops = true;
047  Comparator<N> nodeComparator = null;
048  Comparator<E> edgeComparator = null;
049  Optional<Integer> expectedNodeCount = Optional.absent();
050  Optional<Integer> expectedEdgeCount = Optional.absent();
051
052  /**
053   * Creates a new instance with the specified edge directionality.
054   *
055   * @param directed if true, creates an instance for graphs whose edges are each directed;
056   *      if false, creates an instance for graphs whose edges are each undirected.
057   */
058  private NetworkBuilder(boolean directed) {
059    this.directed = directed;
060  }
061
062  /**
063   * Returns a {@link NetworkBuilder} for building directed graphs.
064   */
065  public static NetworkBuilder<Object, Object> directed() {
066    return new NetworkBuilder<Object, Object>(true);
067  }
068
069  /**
070   * Returns a {@link NetworkBuilder} for building undirected graphs.
071   */
072  public static NetworkBuilder<Object, Object> undirected() {
073    return new NetworkBuilder<Object, Object>(false);
074  }
075
076  /**
077   * Returns a {@link NetworkBuilder} initialized with all properties queryable from {@code graph}.
078   *
079   * <p>The "queryable" properties are those that are exposed through the {@link Network} interface,
080   * such as {@link Network#isDirected()}. Other properties, such as
081   * {@link #expectedNodeCount(int)}, are not set in the new builder.
082   */
083  public static <N, E> NetworkBuilder<N, E> from(Network<N, E> graph) {
084    return new NetworkBuilder<N, E>(graph.isDirected())
085        .allowsParallelEdges(graph.allowsParallelEdges())
086        .allowsSelfLoops(graph.allowsSelfLoops());
087  }
088
089  /**
090   * Specifies whether the graph will allow parallel edges. Attempting to add a parallel edge to
091   * a graph that does not allow them will throw an {@link UnsupportedOperationException}.
092   */
093  public NetworkBuilder<N, E> allowsParallelEdges(boolean allowsParallelEdges) {
094    this.allowsParallelEdges = allowsParallelEdges;
095    return this;
096  }
097
098  /**
099   * Specifies whether the graph will allow self-loops (edges that connect a node to itself).
100   * Attempting to add a self-loop to a graph that does not allow them will throw an
101   * {@link UnsupportedOperationException}.
102   */
103  public NetworkBuilder<N, E> allowsSelfLoops(boolean allowsSelfLoops) {
104    this.allowsSelfLoops = allowsSelfLoops;
105    return this;
106  }
107
108  /**
109   * Specifies the expected number of nodes in the graph.
110   *
111   * @throws IllegalArgumentException if {@code expectedNodeCount} is negative
112   */
113  public NetworkBuilder<N, E> expectedNodeCount(int expectedNodeCount) {
114    checkArgument(expectedNodeCount >= 0, "The expected number of nodes can't be negative: %s",
115        expectedNodeCount);
116    this.expectedNodeCount = Optional.of(expectedNodeCount);
117    return this;
118  }
119
120  /**
121   * Specifies the expected number of edges in the graph.
122   *
123   * @throws IllegalArgumentException if {@code expectedEdgeCount} is negative
124   */
125  public NetworkBuilder<N, E> expectedEdgeCount(int expectedEdgeCount) {
126    checkArgument(expectedEdgeCount >= 0, "The expected number of edges can't be negative: %s",
127        expectedEdgeCount);
128    this.expectedEdgeCount = Optional.of(expectedEdgeCount);
129    return this;
130  }
131
132  /**
133   * Returns an empty {@link MutableNetwork} with the properties of this {@link NetworkBuilder}.
134   */
135  public <N1 extends N, E1 extends E> MutableNetwork<N1, E1> build() {
136    return new ConfigurableMutableNetwork<N1, E1>(this);
137  }
138}
