058  private static <N> Set<Endpoints<N>> endpointsInternal(final Graph<N> graph) {
059    if (graph instanceof Network && !allowsParallelEdges(graph)) {
060      // Use an optimized implementation for networks without parallel edges.
061      return endpointsSimpleNetwork(castToNetwork(graph));
062    }
063
064    return new AbstractSet<Endpoints<N>>() {
065      @Override
066      public Iterator<Endpoints<N>> iterator() {
067        return graph.isDirected()
068            ? new DirectedEndpointsIterator<N>(graph)
069            : new UndirectedEndpointsIterator<N>(graph);
070      }
071
072      @Override
073      public int size() {
074        boolean directed = graph.isDirected();
075        long endpointsCount = 0L;
076        for (N node : graph.nodes()) {
077          Set<N> successors = graph.successors(node);
078          endpointsCount += successors.size();
079          if (!directed && successors.contains(node)) {
080            endpointsCount++; // count self-loops twice in the undirected case
081          }
082        }
083        if (!directed) {
084          // In undirected graphs, every pair of adjacent nodes has been counted twice.
085          checkState((endpointsCount & 1) == 0);
086          endpointsCount >>>= 1;
087        }
088        return Ints.saturatedCast(endpointsCount);
089      }
090
091      @Override
092      public boolean contains(Object obj) {
093        if (!(obj instanceof Endpoints)) {
094          return false;
095        }
096        return containsEndpoints(graph, (Endpoints<?>) obj);
097      }
098    };
099  }
100
101  private static <N> Set<Endpoints<N>> endpointsSimpleNetwork(final Network<N, ?> graph) {
102    checkState(!graph.allowsParallelEdges());
103    return new AbstractSet<Endpoints<N>>() {
104      @Override
105      public Iterator<Endpoints<N>> iterator() {
106        return Iterators.transform(
107            graph.edges().iterator(),
108            new Function<Object, Endpoints<N>>() {
109              @Override
110              public Endpoints<N> apply(Object edge) {
111                return graph.incidentNodes(edge);
112              }
113            });
114      }
115
116      @Override
117      public int size() {
118        return graph.edges().size();
119      }
120
121      @Override
122      public boolean contains(Object obj) {
123        if (!(obj instanceof Endpoints)) {
124          return false;
125        }
126        return containsEndpoints(graph, (Endpoints<?>) obj);
127      }
128    };
129  }
130
131  private static boolean containsEndpoints(Graph<?> graph, Endpoints<?> endpoints) {
132    return graph.isDirected() == (endpoints instanceof Endpoints.Directed)
133        && graph.nodes().contains(endpoints.nodeA())
134        && graph.successors(endpoints.nodeA()).contains(endpoints.nodeB());
135  }

137  /**
138   * Returns an unmodifiable view of edges that are parallel to {@code edge}, i.e. the set of edges
139   * that connect the same nodes in the same direction (if any). An edge is not parallel to itself.
140   *
141   * @throws IllegalArgumentException if {@code edge} is not present in {@code graph}
142   */
143  public static <N, E> Set<E> parallelEdges(Network<N, E> graph, Object edge) {
144    Endpoints<N> endpoints = graph.incidentNodes(edge); // Verifies that edge is in graph
145    if (!graph.allowsParallelEdges()) {
146      return ImmutableSet.of();
147    }
148    return Sets.difference(graph.edgesConnecting(endpoints.nodeA(), endpoints.nodeB()),
149        ImmutableSet.of(edge)); // An edge is not parallel to itself.
150  }

156  /**
157   * Returns an induced subgraph of {@code graph}. This subgraph is a new graph that contains
158   * all of the nodes in {@code nodes}, and all of the edges from {@code graph} for which the
159   * edge's incident nodes are both contained by {@code nodes}.
160   *
161   * @throws IllegalArgumentException if any element in {@code nodes} is not a node in the graph
162   */
163  public static <N, E> MutableNetwork<N, E> inducedSubgraph(Network<N, E> graph,
164      Iterable<? extends N> nodes) {
165    NetworkBuilder<N, E> builder = NetworkBuilder.from(graph);
166    if (nodes instanceof Collection) {
167      builder = builder.expectedNodeCount(((Collection<?>) nodes).size());
168    }
169    MutableNetwork<N, E> subgraph = builder.build();
170    for (N node : nodes) {
171      subgraph.addNode(node);
172    }
173    for (N node : subgraph.nodes()) {
174      for (E edge : graph.outEdges(node)) {
175        N otherNode = graph.incidentNodes(edge).otherNode(node);
176        if (subgraph.nodes().contains(otherNode)) {
177          subgraph.addEdge(edge, node, otherNode);
178        }
179      }
180    }
181    return subgraph;
182  }

184  /**
185   * Creates a mutable copy of {@code graph}, using the same nodes and edges.
186   */
187  public static <N> MutableGraph<N> copyOf(Graph<N> graph) {
188    checkNotNull(graph, "graph");
189    // TODO(b/28087289): we can remove this restriction when Graph supports parallel edges
190    checkArgument(!allowsParallelEdges(graph), NETWORK_WITH_PARALLEL_EDGE);
191    MutableGraph<N> copy = GraphBuilder.from(graph)
192        .expectedNodeCount(graph.nodes().size())
193        .build();
194
195    for (N node : graph.nodes()) {
196      checkState(copy.addNode(node));
197    }
198    for (Endpoints<N> endpoints : endpointsInternal(graph)) {
199      checkState(copy.addEdge(endpoints.nodeA(), endpoints.nodeB()));
200    }
201
202    return copy;
203  }
204
205  /**
206   * Creates a mutable copy of {@code graph}, using the same node and edge elements.
207   */
208  public static <N, E> MutableNetwork<N, E> copyOf(Network<N, E> graph) {
209    checkNotNull(graph, "graph");
210    MutableNetwork<N, E> copy = NetworkBuilder.from(graph)
211        .expectedNodeCount(graph.nodes().size())
212        .expectedEdgeCount(graph.edges().size())
213        .build();
214
215    for (N node : graph.nodes()) {
216      checkState(copy.addNode(node));
217    }
218    for (E edge : graph.edges()) {
219      Endpoints<N> endpoints = graph.incidentNodes(edge);
220      checkState(copy.addEdge(edge, endpoints.nodeA(), endpoints.nodeB()));
221    }
222
223    return copy;
224  }

226  /**
227   * Returns true iff {@code graph1} and {@code graph2} are equal as defined by
228   * {@link Graph#equals(Object)}.
229   */
230  public static boolean equal(@Nullable Graph<?> graph1, @Nullable Graph<?> graph2) {
231    // If both graphs are Network instances, use equal(Network, Network) instead
232    if (graph1 instanceof Network && graph2 instanceof Network) {
233      return equal(castToNetwork(graph1), castToNetwork(graph2));
234    }
235
236    // Otherwise, if either graph is a Network (but not both), they can't be equal.
237    if (graph1 instanceof Network || graph2 instanceof Network) {
238      return false;
239    }
240
241    if (graph1 == graph2) {
242      return true;
243    }
244
245    if (graph1 == null || graph2 == null) {
246      return false;
247    }
248
249    if (graph1.isDirected() != graph2.isDirected()) {
250      return false;
251    }
252
253    if (!graph1.nodes().equals(graph2.nodes())) {
254      return false;
255    }
256
257    for (Object node : graph1.nodes()) {
258      if (!graph1.successors(node).equals(graph2.successors(node))) {
259        return false;
260      }
261    }
262
263    return true;
264  }

266  /**
267   * Returns true iff {@code graph1} and {@code graph2} are equal as defined by
268   * {@link Network#equals(Object)}.
269   */
270  public static boolean equal(@Nullable Network<?, ?> graph1, @Nullable Network<?, ?> graph2) {
271    if (graph1 == graph2) {
272      return true;
273    }
274
275    if (graph1 == null || graph2 == null) {
276      return false;
277    }
278
279    if (graph1.isDirected() != graph2.isDirected()) {
280      return false;
281    }
282
283    if (!graph1.nodes().equals(graph2.nodes()) || !graph1.edges().equals(graph2.edges())) {
284      return false;
285    }
286
287    for (Object edge : graph1.edges()) {
288      if (!graph1.incidentNodes(edge).equals(graph2.incidentNodes(edge))) {
289        return false;
290      }
291    }
292
293    return true;
294  }

318    if (graph instanceof Network) {

347   * Returns a map that is a live view of {@code graph}, with nodes as keys
348   * and the set of outgoing edges as values.
349   */
350  private static <N, E> Map<N, Set<E>> nodeToOutEdges(final Network<N, E> graph) {
351    checkNotNull(graph, "graph");
352    return Maps.asMap(graph.nodes(), new Function<N, Set<E>>() {
353      @Override
354      public Set<E> apply(N node) {
355        return graph.outEdges(node);
356      }
357    });
358  }

361   * Returns a map that is a live view of {@code graph}, with nodes as keys
362   * and the set of successor nodes as values.
363   */
364  private static <N> Map<N, Set<N>> nodeToSuccessorNodes(final Graph<N> graph) {
365    checkNotNull(graph, "graph");
366    return Maps.asMap(graph.nodes(), new Function<N, Set<N>>() {
367      @Override
368      public Set<N> apply(N node) {
369        return graph.successors(node);
370      }

418  private abstract static class AbstractEndpointsIterator<N>
419      extends AbstractIterator<Endpoints<N>> {
420    private final Graph<N> graph;
421    private final Iterator<N> nodeIterator;
422
423    N node = null; // null is safe as an initial value because graphs do not allow null nodes
424    Iterator<N> successorIterator = ImmutableSet.<N>of().iterator();
425
426    AbstractEndpointsIterator(Graph<N> graph) {
427      this.graph = graph;
428      this.nodeIterator = graph.nodes().iterator();
429    }
430
431    /**
432     * Called after {@link #successorIterator} is exhausted. Advances {@link #node} to the next node
433     * and updates {@link #successorIterator} to iterate through the successors of {@link #node}.
434     */
435    final boolean advance() {
436      checkState(!successorIterator.hasNext());
437      if (!nodeIterator.hasNext()) {
438        return false;
439      }
440      node = nodeIterator.next();
441      successorIterator = graph.successors(node).iterator();
442      return true;
443    }
444  }

446  /**
447   * If the graph is directed, each ordered [source, target] pair will be visited once if there is
448   * one or more edge connecting them.
449   */
450  private static final class DirectedEndpointsIterator<N> extends AbstractEndpointsIterator<N> {
451    DirectedEndpointsIterator(Graph<N> graph){
452      super(graph);
453    }
454
455    @Override
456    protected Endpoints<N> computeNext() {
457      while (true) {
458        if (successorIterator.hasNext()) {
459          return Endpoints.ofDirected(node, successorIterator.next());
460        }
461        if (!advance()) {
462          return endOfData();
463        }
464      }
465    }
466  }

468  /**
469   * If the graph is undirected, each unordered [node, otherNode] pair (except self-loops) will be
470   * visited twice if there is one or more edge connecting them. To avoid returning duplicate
471   * {@link Endpoints}, we keep track of the nodes that we have visited. When processing node pairs,
472   * we skip if the "other node" is in the visited set, as shown below:
473   *
474   * Nodes = {N1, N2, N3, N4}
475   *    N2           __
476   *   /  \         |  |
477   * N1----N3      N4__|
478   *
479   * Visited Nodes = {}
480   * Endpoints [N1, N2] - return
481   * Endpoints [N1, N3] - return
482   * Visited Nodess = {N1}
483   * Endpoints [N2, N1] - skip
484   * Endpoints [N2, N3] - return
485   * Visited Nodes = {N1, N2}
486   * Endpoints [N3, N1] - skip
487   * Endpoints [N3, N2] - skip
488   * Visited Nodes = {N1, N2, N3}
489   * Endpoints [N4, N4] - return
490   * Visited Nodes = {N1, N2, N3, N4}
491   */
492  private static final class UndirectedEndpointsIterator<N> extends AbstractEndpointsIterator<N> {
493    private Set<N> visitedNodes;
494
495    UndirectedEndpointsIterator(Graph<N> graph) {
496      super(graph);
497      this.visitedNodes = Sets.newHashSetWithExpectedSize(graph.nodes().size());
498    }
499
500    @Override
501    protected Endpoints<N> computeNext() {
502      while (true) {
503        while (successorIterator.hasNext()) {
504          N otherNode = successorIterator.next();
505          if (!visitedNodes.contains(otherNode)) {
506            return Endpoints.ofUndirected(node, otherNode);
507          }
508        }
509        // Add to visited set *after* processing neighbors so we still include self-loops.
510        visitedNodes.add(node);
511        if (!advance()) {
512          visitedNodes = null;
513          return endOfData();
514        }
515      }
516    }
517  }
