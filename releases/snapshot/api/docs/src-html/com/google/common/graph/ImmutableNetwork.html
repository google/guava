017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkNotNull;
020
021import com.google.common.base.Function;
022import com.google.common.collect.ImmutableMap;
023import com.google.common.collect.ImmutableSet;
024import com.google.common.collect.Maps;
025
026import java.util.Map;
027import java.util.Set;
028
029/**
030 * A {@link Network} whose relationships are constant. Instances of this class may be obtained
031 * with {@link #copyOf(Network)}.
032 *
033 * <p>The time complexity of {@code edgesConnecting(node1, node2)} is O(min(outD_node1, inD_node2)).
034 *
035 * @author James Sexton
036 * @author Joshua O'Madadhain
037 * @author Omar Darwish
038 * @param <N> Node parameter type
039 * @param <E> Edge parameter type
040 */
041public final class ImmutableNetwork<N, E> extends AbstractConfigurableNetwork<N, E> {
042
043  private ImmutableNetwork(Network<N, E> graph) {
044    super(NetworkBuilder.from(graph), getNodeConnections(graph), getEdgeToReferenceNode(graph));
045  }
046
047  /**
048   * Returns an immutable copy of {@code graph}.
049   */
050  public static <N, E> ImmutableNetwork<N, E> copyOf(Network<N, E> graph) {
051    return (graph instanceof ImmutableNetwork)
052        ? (ImmutableNetwork<N, E>) graph
053        : new ImmutableNetwork<N, E>(graph);
054  }
055
056  /**
057   * Simply returns its argument.
058   *
059   * @deprecated no need to use this
060   */
061  @Deprecated
062  public static <N, E> ImmutableNetwork<N, E> copyOf(ImmutableNetwork<N, E> graph) {
063    return checkNotNull(graph);
064  }
065
066  @Override
067  public Set<E> edgesConnecting(Object node1, Object node2) {
068    // This set is calculated as the intersection of two sets, and is likely to be small.
069    // As an optimization, copy it to an ImmutableSet so re-iterating is fast.
070    return ImmutableSet.copyOf(super.edgesConnecting(node1, node2));
071  }
072
073  private static <N, E> Map<N, NodeConnections<N, E>> getNodeConnections(Network<N, E> graph) {
074    ImmutableMap.Builder<N, NodeConnections<N, E>> nodeConnections = ImmutableMap.builder();
075    for (N node : graph.nodes()) {
076      nodeConnections.put(node, nodeConnectionsOf(graph, node));
077    }
078    return nodeConnections.build();
079  }
080
081  private static <N, E> Map<E, N> getEdgeToReferenceNode(Network<N, E> graph) {
082    ImmutableMap.Builder<E, N> edgeToReferenceNode = ImmutableMap.builder();
083    for (E edge : graph.edges()) {
084      edgeToReferenceNode.put(edge, graph.incidentNodes(edge).iterator().next());
085    }
086    return edgeToReferenceNode.build();
087  }
088
089  private static <N, E> NodeConnections<N, E> nodeConnectionsOf(Network<N, E> graph, N node) {
090    if (graph.isDirected()) {
091      Map<E, N> inEdgeMap = Maps.asMap(graph.inEdges(node), sourceNodeFn(graph));
092      Map<E, N> outEdgeMap = Maps.asMap(graph.outEdges(node), targetNodeFn(graph));
093      return graph.allowsParallelEdges()
094           ? DirectedMultiNodeConnections.ofImmutable(inEdgeMap, outEdgeMap)
095           : DirectedNodeConnections.ofImmutable(inEdgeMap, outEdgeMap);
096    } else {
097      Map<E, N> incidentEdgeMap =
098          Maps.asMap(graph.incidentEdges(node), oppositeNodeFn(graph, node));
099      return graph.allowsParallelEdges()
100          ? UndirectedMultiNodeConnections.ofImmutable(incidentEdgeMap)
101          : UndirectedNodeConnections.ofImmutable(incidentEdgeMap);
102    }
103  }
104
105  private static <N, E> Function<E, N> sourceNodeFn(final Network<N, E> graph) {
106    return new Function<E, N>() {
107      @Override
108      public N apply(E edge) {
109        return graph.source(edge);
110      }
111    };
112  }
113
114  private static <N, E> Function<E, N> targetNodeFn(final Network<N, E> graph) {
115    return new Function<E, N>() {
116      @Override
117      public N apply(E edge) {
118        return graph.target(edge);
119      }
120    };
121  }
122
123  private static <N, E> Function<E, N> oppositeNodeFn(final Network<N, E> graph, final N node) {
124    return new Function<E, N>() {
125      @Override
126      public N apply(E edge) {
127        return Graphs.oppositeNode(graph, edge, node);
128      }
129    };
130  }
131}
