045/**
046 * A double-ended priority queue, which provides constant-time access to both
047 * its least element and its greatest element, as determined by the queue's
048 * specified comparator. If no comparator is given at creation time, the
049 * natural order of elements is used. If no maximum size is given at creation time,
050 * the queue is unbounded.
051 *
052 * <p>Usage example: <pre>   {@code
053 *
054 *   MinMaxPriorityQueue<User> users = MinMaxPriorityQueue.orderedBy(userComparator)
055 *       .maximumSize(1000)
056 *       .create();}</pre>
057 *
058 * <p>As a {@link Queue} it functions exactly as a {@link PriorityQueue}: its
059 * head element -- the implicit target of the methods {@link #peek()}, {@link
060 * #poll()} and {@link #remove()} -- is defined as the <i>least</i> element in
061 * the queue according to the queue's comparator. But unlike a regular priority
062 * queue, the methods {@link #peekLast}, {@link #pollLast} and
063 * {@link #removeLast} are also provided, to act on the <i>greatest</i> element
064 * in the queue instead.
065 *
066 * <p>A min-max priority queue can be configured with a maximum size. If so,
067 * each time the size of the queue exceeds that value, the queue automatically
068 * removes its greatest element according to its comparator (which might be the
069 * element that was just added). This is different from conventional bounded
070 * queues, which either block or reject new elements when full.
071 *
072 * <p>This implementation is based on the
073 * <a href="http://portal.acm.org/citation.cfm?id=6621">min-max heap</a>
074 * developed by Atkinson, et al. Unlike many other double-ended priority queues,
075 * it stores elements in a single array, as compact as the traditional heap data
076 * structure used in {@link PriorityQueue}.
077 *
078 * <p>This class is not thread-safe, and does not accept null elements.
079 *
080 * <p><i>Performance notes:</i>
081 *
082 * <ul>
083 * <li>If you only access one end of the queue, and do use a maximum size,
084 *     this class will perform significantly worse than a {@code PriorityQueue}
085 *     with manual eviction above the maximum size.  In many cases
086 *     {@link Ordering#leastOf} may work for your use case with significantly
087 *     improved (and asymptotically superior) performance.
088 * <li>The retrieval operations {@link #peek}, {@link #peekFirst}, {@link
089 *     #peekLast}, {@link #element}, and {@link #size} are constant-time.
090 * <li>The enqueing and dequeing operations ({@link #offer}, {@link #add}, and
091 *     all the forms of {@link #poll} and {@link #remove()}) run in {@code
092 *     O(log n) time}.
093 * <li>The {@link #remove(Object)} and {@link #contains} operations require
094 *     linear ({@code O(n)}) time.
095 * <li>If you only access one end of the queue, and don't use a maximum size,
096 *     this class is functionally equivalent to {@link PriorityQueue}, but
097 *     significantly slower.
098 * </ul>
099 *
100 * @author Sverre Sundsdal
101 * @author Torbjorn Gannholm
102 * @since 8.0
103 */
104// TODO(kevinb): GWT compatibility

241  private MinMaxPriorityQueue(Builder<? super E> builder, int queueSize) {
242    Ordering<E> ordering = builder.ordering();
243    this.minHeap = new Heap(ordering);
244    this.maxHeap = new Heap(ordering.reverse());
245    minHeap.otherHeap = maxHeap;
246    maxHeap.otherHeap = minHeap;
247
248    this.maximumSize = builder.maximumSize;
249    // TODO(kevinb): pad?
250    this.queue = new Object[queueSize];
251  }

282  /**
283   * Adds the given element to this queue. If this queue has a maximum size,
284   * after adding {@code element} the queue will automatically evict its
285   * greatest element (according to its comparator), which may be {@code
286   * element} itself.
287   */
288  @Override
289  public boolean offer(E element) {
290    checkNotNull(element);
291    modCount++;
292    int insertIndex = size++;
293
294    growIfNeeded();
295
296    // Adds the element to the end of the heap and bubbles it up to the correct
297    // position.
298    heapForIndex(insertIndex).bubbleUp(insertIndex, element);
299    return size <= maximumSize || pollLast() != element;
300  }

302  @Override
303  public E poll() {
304    return isEmpty() ? null : removeAndGet(0);
305  }

317  /**
318   * Returns the index of the max element.
319   */
320  private int getMaxElementIndex() {
321    switch (size) {
322      case 1:
323        return 0; // The lone element in the queue is the maximum.
324      case 2:
325        return 1; // The lone element in the maxHeap is the maximum.
326      default:
327        // The max element must sit on the first level of the maxHeap. It is
328        // actually the *lesser* of the two from the maxHeap's perspective.
329        return (maxHeap.compareElements(1, 2) <= 0) ? 1 : 2;
330    }
331  }

386  /**
387   * Removes the element at position {@code index}.
388   *
389   * <p>Normally this method leaves the elements at up to {@code index - 1},
390   * inclusive, untouched.  Under these circumstances, it returns {@code null}.
391   *
392   * <p>Occasionally, in order to maintain the heap invariant, it must swap a
393   * later element of the list with one before {@code index}. Under these
394   * circumstances it returns a pair of elements as a {@link MoveDesc}. The
395   * first one is the element that was previously at the end of the heap and is
396   * now at some position before {@code index}. The second element is the one
397   * that was swapped down to replace the element at {@code index}. This fact is
398   * used by iterator.remove so as to visit elements during a traversal once and
399   * only once.
400   */
401  @VisibleForTesting
402  MoveDesc<E> removeAt(int index) {
403    checkPositionIndex(index, size);
404    modCount++;
405    size--;
406    if (size == index) {
407      queue[size] = null;
408      return null;
409    }
410    E actualLastElement = elementData(size);
411    int lastElementAt = heapForIndex(size).getCorrectLastElement(actualLastElement);
412    E toTrickle = elementData(size);
413    queue[size] = null;
414    MoveDesc<E> changes = fillHole(index, toTrickle);
415    if (lastElementAt < index) {
416      // Last element is moved to before index, swapped with trickled element.
417      if (changes == null) {
418        // The trickled element is still after index.
419        return new MoveDesc<E>(actualLastElement, toTrickle);
420      } else {
421        // The trickled element is back before index, but the replaced element
422        // has now been moved after index.
423        return new MoveDesc<E>(actualLastElement, changes.replaced);
424      }
425    }
426    // Trickled element was after index to begin with, no adjustment needed.
427    return changes;
428  }
429
430  private MoveDesc<E> fillHole(int index, E toTrickle) {
431    Heap heap = heapForIndex(index);
432    // We consider elementData(index) a "hole", and we want to fill it
433    // with the last element of the heap, toTrickle.
434    // Since the last element of the heap is from the bottom level, we
435    // optimistically fill index position with elements from lower levels,
436    // moving the hole down. In most cases this reduces the number of
437    // comparisons with toTrickle, but in some cases we will need to bubble it
438    // all the way up again.
439    int vacated = heap.fillHoleAt(index);
440    // Try to see if toTrickle can be bubbled up min levels.
441    int bubbledTo = heap.bubbleUpAlternatingLevels(vacated, toTrickle);
442    if (bubbledTo == vacated) {
443      // Could not bubble toTrickle up min levels, try moving
444      // it from min level to max level (or max to min level) and bubble up
445      // there.
446      return heap.tryCrossOverAndBubbleUp(index, vacated, toTrickle);
447    } else {
448      return (bubbledTo < index) ? new MoveDesc<E>(toTrickle, elementData(index)) : null;
449    }
450  }

452  // Returned from removeAt() to iterator.remove()
453  static class MoveDesc<E> {
454    final E toTrickle;
455    final E replaced;
456
457    MoveDesc(E toTrickle, E replaced) {
458      this.toTrickle = toTrickle;
459      this.replaced = replaced;
460    }
461  }

467    E value = elementData(index);

472  private Heap heapForIndex(int i) {
473    return isEvenLevel(i) ? minHeap : maxHeap;
474  }

476  private static final int EVEN_POWERS_OF_TWO = 0x55555555;
477  private static final int ODD_POWERS_OF_TWO = 0xaaaaaaaa;
478
479  @VisibleForTesting
480  static boolean isEvenLevel(int index) {
481    int oneBased = index + 1;
482    checkState(oneBased > 0, "negative index");
483    return (oneBased & EVEN_POWERS_OF_TWO) > (oneBased & ODD_POWERS_OF_TWO);
484  }

502  /**
503   * Each instance of MinMaxPriortyQueue encapsulates two instances of Heap:
504   * a min-heap and a max-heap. Conceptually, these might each have their own
505   * array for storage, but for efficiency's sake they are stored interleaved on
506   * alternate heap levels in the same array (MMPQ.queue).
507   */
508  @WeakOuter
509  private class Heap {
510    final Ordering<E> ordering;
511    @Weak Heap otherHeap;
512
513    Heap(Ordering<E> ordering) {
514      this.ordering = ordering;
515    }
516
517    int compareElements(int a, int b) {
518      return ordering.compare(elementData(a), elementData(b));
519    }
520
521    /**
522     * Tries to move {@code toTrickle} from a min to a max level and
523     * bubble up there. If it moved before {@code removeIndex} this method
524     * returns a pair as described in {@link #removeAt}.
525     */
526    MoveDesc<E> tryCrossOverAndBubbleUp(int removeIndex, int vacated, E toTrickle) {
527      int crossOver = crossOver(vacated, toTrickle);
528      if (crossOver == vacated) {
529        return null;
530      }
531      // Successfully crossed over from min to max.
532      // Bubble up max levels.
533      E parent;
534      // If toTrickle is moved up to a parent of removeIndex, the parent is
535      // placed in removeIndex position. We must return that to the iterator so
536      // that it knows to skip it.
537      if (crossOver < removeIndex) {
538        // We crossed over to the parent level in crossOver, so the parent
539        // has already been moved.
540        parent = elementData(removeIndex);
541      } else {
542        parent = elementData(getParentIndex(removeIndex));
543      }
544      // bubble it up the opposite heap
545      if (otherHeap.bubbleUpAlternatingLevels(crossOver, toTrickle) < removeIndex) {
546        return new MoveDesc<E>(toTrickle, parent);
547      } else {
548        return null;
549      }
550    }
551
552    /**
553     * Bubbles a value from {@code index} up the appropriate heap if required.
554     */
555    void bubbleUp(int index, E x) {
556      int crossOver = crossOverUp(index, x);
557
558      Heap heap;
559      if (crossOver == index) {
560        heap = this;
561      } else {
562        index = crossOver;
563        heap = otherHeap;
564      }
565      heap.bubbleUpAlternatingLevels(index, x);
566    }
567
568    /**
569     * Bubbles a value from {@code index} up the levels of this heap, and
570     * returns the index the element ended up at.
571     */
572    int bubbleUpAlternatingLevels(int index, E x) {
573      while (index > 2) {
574        int grandParentIndex = getGrandparentIndex(index);
575        E e = elementData(grandParentIndex);
576        if (ordering.compare(e, x) <= 0) {
577          break;
578        }
579        queue[index] = e;
580        index = grandParentIndex;
581      }
582      queue[index] = x;
583      return index;
584    }
585
586    /**
587     * Returns the index of minimum value between {@code index} and
588     * {@code index + len}, or {@code -1} if {@code index} is greater than
589     * {@code size}.
590     */
591    int findMin(int index, int len) {
592      if (index >= size) {
593        return -1;
594      }
595      checkState(index > 0);
596      int limit = Math.min(index, size - len) + len;
597      int minIndex = index;
598      for (int i = index + 1; i < limit; i++) {
599        if (compareElements(i, minIndex) < 0) {
600          minIndex = i;
601        }
602      }
603      return minIndex;
604    }
605
606    /**
607     * Returns the minimum child or {@code -1} if no child exists.
608     */
609    int findMinChild(int index) {
610      return findMin(getLeftChildIndex(index), 2);
611    }
612
613    /**
614     * Returns the minimum grand child or -1 if no grand child exists.
615     */
616    int findMinGrandChild(int index) {
617      int leftChildIndex = getLeftChildIndex(index);
618      if (leftChildIndex < 0) {
619        return -1;
620      }
621      return findMin(getLeftChildIndex(leftChildIndex), 4);
622    }

624    /**
625     * Moves an element one level up from a min level to a max level
626     * (or vice versa).
627     * Returns the new position of the element.
628     */
629    int crossOverUp(int index, E x) {
630      if (index == 0) {
631        queue[0] = x;
632        return 0;
633      }
634      int parentIndex = getParentIndex(index);
635      E parentElement = elementData(parentIndex);
636      if (parentIndex != 0) {
637        // This is a guard for the case of the childless uncle.
638        // Since the end of the array is actually the middle of the heap,
639        // a smaller childless uncle can become a child of x when we
640        // bubble up alternate levels, violating the invariant.
641        int grandparentIndex = getParentIndex(parentIndex);
642        int uncleIndex = getRightChildIndex(grandparentIndex);
643        if (uncleIndex != parentIndex && getLeftChildIndex(uncleIndex) >= size) {
644          E uncleElement = elementData(uncleIndex);
645          if (ordering.compare(uncleElement, parentElement) < 0) {
646            parentIndex = uncleIndex;
647            parentElement = uncleElement;
648          }
649        }
650      }
651      if (ordering.compare(parentElement, x) < 0) {
652        queue[index] = parentElement;
653        queue[parentIndex] = x;
654        return parentIndex;
655      }
656      queue[index] = x;
657      return index;
658    }
659
660    /**
661     * Returns the conceptually correct last element of the heap.
662     *
663     * <p>Since the last element of the array is actually in the
664     * middle of the sorted structure, a childless uncle node could be
665     * smaller, which would corrupt the invariant if this element
666     * becomes the new parent of the uncle. In that case, we first
667     * switch the last element with its uncle, before returning.
668     */
669    int getCorrectLastElement(E actualLastElement) {
670      int parentIndex = getParentIndex(size);
671      if (parentIndex != 0) {
672        int grandparentIndex = getParentIndex(parentIndex);
673        int uncleIndex = getRightChildIndex(grandparentIndex);
674        if (uncleIndex != parentIndex && getLeftChildIndex(uncleIndex) >= size) {
675          E uncleElement = elementData(uncleIndex);
676          if (ordering.compare(uncleElement, actualLastElement) < 0) {
677            queue[uncleIndex] = actualLastElement;
678            queue[size] = uncleElement;
679            return uncleIndex;
680          }
681        }
682      }
683      return size;
684    }

686    /**
687     * Crosses an element over to the opposite heap by moving it one level down
688     * (or up if there are no elements below it).
689     *
690     * Returns the new position of the element.
691     */
692    int crossOver(int index, E x) {
693      int minChildIndex = findMinChild(index);
694      // TODO(kevinb): split the && into two if's and move crossOverUp so it's
695      // only called when there's no child.
696      if ((minChildIndex > 0) && (ordering.compare(elementData(minChildIndex), x) < 0)) {
697        queue[index] = elementData(minChildIndex);
698        queue[minChildIndex] = x;
699        return minChildIndex;
700      }
701      return crossOverUp(index, x);
702    }
703
704    /**
705     * Fills the hole at {@code index} by moving in the least of its
706     * grandchildren to this position, then recursively filling the new hole
707     * created.
708     *
709     * @return the position of the new hole (where the lowest grandchild moved
710     *     from, that had no grandchild to replace it)
711     */
712    int fillHoleAt(int index) {
713      int minGrandchildIndex;
714      while ((minGrandchildIndex = findMinGrandChild(index)) > 0) {
715        queue[index] = elementData(minGrandchildIndex);
716        index = minGrandchildIndex;
717      }
718      return index;
719    }
720
721    private boolean verifyIndex(int i) {
722      if ((getLeftChildIndex(i) < size) && (compareElements(i, getLeftChildIndex(i)) > 0)) {
723        return false;
724      }
725      if ((getRightChildIndex(i) < size) && (compareElements(i, getRightChildIndex(i)) > 0)) {
726        return false;
727      }
728      if ((i > 0) && (compareElements(i, getParentIndex(i)) > 0)) {
729        return false;
730      }
731      if ((i > 2) && (compareElements(getGrandparentIndex(i), i) > 0)) {
732        return false;
733      }
734      return true;
735    }
736
737    // These would be static if inner classes could have static members.
738
739    private int getLeftChildIndex(int i) {
740      return i * 2 + 1;
741    }
742
743    private int getRightChildIndex(int i) {
744      return i * 2 + 2;
745    }
746
747    private int getParentIndex(int i) {
748      return (i - 1) / 2;
749    }
750
751    private int getGrandparentIndex(int i) {
752      return getParentIndex(getParentIndex(i)); // (i - 3) / 4
753    }
754  }

756  /**
757   * Iterates the elements of the queue in no particular order.
758   *
759   * If the underlying queue is modified during iteration an exception will be
760   * thrown.
761   */
762  private class QueueIterator implements Iterator<E> {
763    private int cursor = -1;
764    private int expectedModCount = modCount;
765    private Queue<E> forgetMeNot;
766    private List<E> skipMe;
767    private E lastFromForgetMeNot;
768    private boolean canRemove;
769
770    @Override
771    public boolean hasNext() {
772      checkModCount();
773      return (nextNotInSkipMe(cursor + 1) < size())
774          || ((forgetMeNot != null) && !forgetMeNot.isEmpty());
775    }
776
777    @Override
778    public E next() {
779      checkModCount();
780      int tempCursor = nextNotInSkipMe(cursor + 1);
781      if (tempCursor < size()) {
782        cursor = tempCursor;
783        canRemove = true;
784        return elementData(cursor);
785      } else if (forgetMeNot != null) {
786        cursor = size();
787        lastFromForgetMeNot = forgetMeNot.poll();
788        if (lastFromForgetMeNot != null) {
789          canRemove = true;
790          return lastFromForgetMeNot;
791        }
792      }
793      throw new NoSuchElementException("iterator moved past last element in queue.");
794    }
795
796    @Override
797    public void remove() {
798      checkRemove(canRemove);
799      checkModCount();
800      canRemove = false;
801      expectedModCount++;
802      if (cursor < size()) {
803        MoveDesc<E> moved = removeAt(cursor);
804        if (moved != null) {
805          if (forgetMeNot == null) {
806            forgetMeNot = new ArrayDeque<E>();
807            skipMe = new ArrayList<E>(3);
808          }
809          forgetMeNot.add(moved.toTrickle);
810          skipMe.add(moved.replaced);
811        }
812        cursor--;
813      } else { // we must have set lastFromForgetMeNot in next()
814        checkState(removeExact(lastFromForgetMeNot));
815        lastFromForgetMeNot = null;
816      }
817    }
818
819    // Finds only this exact instance, not others that are equals()
820    private boolean containsExact(Iterable<E> elements, E target) {
821      for (E element : elements) {
822        if (element == target) {
823          return true;
824        }
825      }
826      return false;
827    }
828
829    // Removes only this exact instance, not others that are equals()
830    boolean removeExact(Object target) {
831      for (int i = 0; i < size; i++) {
832        if (queue[i] == target) {
833          removeAt(i);
834          return true;
835        }
836      }
837      return false;
838    }
839
840    void checkModCount() {
841      if (modCount != expectedModCount) {
842        throw new ConcurrentModificationException();
843      }
844    }
845
846    /**
847     * Returns the index of the first element after {@code c} that is not in
848     * {@code skipMe} and returns {@code size()} if there is no such element.
849     */
850    private int nextNotInSkipMe(int c) {
851      if (skipMe != null) {
852        while (c < size() && containsExact(skipMe, elementData(c))) {
853          c++;
854        }
855      }
856      return c;
857    }
858  }

860  /**
861   * Returns an iterator over the elements contained in this collection,
862   * <i>in no particular order</i>.
863   *
864   * <p>The iterator is <i>fail-fast</i>: If the MinMaxPriorityQueue is modified
865   * at any time after the iterator is created, in any way except through the
866   * iterator's own remove method, the iterator will generally throw a
867   * {@link ConcurrentModificationException}. Thus, in the face of concurrent
868   * modification, the iterator fails quickly and cleanly, rather than risking
869   * arbitrary, non-deterministic behavior at an undetermined time in the
870   * future.
871   *
872   * <p>Note that the fail-fast behavior of an iterator cannot be guaranteed
873   * as it is, generally speaking, impossible to make any hard guarantees in the
874   * presence of unsynchronized concurrent modification.  Fail-fast iterators
875   * throw {@code ConcurrentModificationException} on a best-effort basis.
876   * Therefore, it would be wrong to write a program that depended on this
877   * exception for its correctness: <i>the fail-fast behavior of iterators
878   * should be used only to detect bugs.</i>
879   *
880   * @return an iterator over the elements contained in this collection
881   */
882  @Override
883  public Iterator<E> iterator() {
884    return new QueueIterator();
885  }

887  @Override
888  public void clear() {
889    for (int i = 0; i < size; i++) {
890      queue[i] = null;
891    }
892    size = 0;
893  }

918  private static final int DEFAULT_CAPACITY = 11;
919
920  @VisibleForTesting
921  static int initialQueueSize(
922      int configuredExpectedSize, int maximumSize, Iterable<?> initialContents) {
923    // Start with what they said, if they said it, otherwise DEFAULT_CAPACITY
924    int result =
925        (configuredExpectedSize == Builder.UNSET_EXPECTED_SIZE)
926            ? DEFAULT_CAPACITY
927            : configuredExpectedSize;
928
929    // Enlarge to contain initial contents
930    if (initialContents instanceof Collection) {
931      int initialSize = ((Collection<?>) initialContents).size();
932      result = Math.max(result, initialSize);
933    }
934
935    // Now cap it at maxSize + 1
936    return capAtMaximumSize(result, maximumSize);
937  }

939  private void growIfNeeded() {
940    if (size > queue.length) {
941      int newCapacity = calculateNewCapacity();
942      Object[] newQueue = new Object[newCapacity];
943      System.arraycopy(queue, 0, newQueue, 0, queue.length);
944      queue = newQueue;
945    }
946  }
947
948  /** Returns ~2x the old capacity if small; ~1.5x otherwise. */
949  private int calculateNewCapacity() {
950    int oldCapacity = queue.length;
951    int newCapacity =
952        (oldCapacity < 64)
953            ? (oldCapacity + 1) * 2
954            : IntMath.checkedMultiply(oldCapacity / 2, 3);
955    return capAtMaximumSize(newCapacity, maximumSize);
956  }
957
958  /** There's no reason for the queueSize to ever be more than maxSize + 1 */
959  private static int capAtMaximumSize(int queueSize, int maximumSize) {
960    return Math.min(queueSize - 1, maximumSize) + 1; // don't overflow
961  }
