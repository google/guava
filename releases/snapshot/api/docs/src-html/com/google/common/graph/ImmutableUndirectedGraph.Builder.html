017package com.google.common.graph;
018
019import com.google.common.annotations.Beta;
020import com.google.errorprone.annotations.CanIgnoreReturnValue;
021
022/**
023 * Implementation of an immutable undirected graph consisting of nodes of type N
024 * and edges of type E.
025 *
026 * <p>Some invariants/assumptions are maintained in this implementation:
027 * <ul>
028 * <li>An edge has exactly two end-points, which may or may not be distinct.
029 * <li>By default, this is not a multigraph, that is, parallel edges (multiple
030 *     edges between n1 and n2) are not allowed.  If you want a multigraph,
031 *     build the graph with the 'multigraph' option:
032 *     <pre>ImmutableUndirectedGraph.builder(Graphs.config().multigraph()).build();</pre>
033 * <li>Edges are not adjacent to themselves by definition. In the case of a
034 *     self-loop, a node can be adjacent to itself, but an edge will never be adjacent to itself.
035 * </ul>
036 *
037 * @author James Sexton
038 * @author Joshua O'Madadhain
039 * @author Omar Darwish
040 * @param <N> Node parameter type
041 * @param <E> Edge parameter type
042 * @see AbstractConfigurableGraph
043 * @see AbstractImmutableGraph
044 * @since 20.0
045 */
046@Beta
047public final class ImmutableUndirectedGraph<N, E> extends AbstractImmutableGraph<N, E>
048    implements UndirectedGraph<N, E> {
049
050  private ImmutableUndirectedGraph(Builder<N, E> builder) {
051    super(builder);
052  }
053
054  /**
055   * Returns a new builder. The generated builder is equivalent to the builder
056   * created by the {@code Builder} constructor.
057   */
058  public static <N, E> Builder<N, E> builder() {
059    return new Builder<N, E>();
060  }
061
062  /**
063   * Returns a new builder. The generated builder is equivalent to the builder
064   * created by the {@code Builder} constructor.
065   *
066   * @param config an instance of {@code GraphConfig} with the intended
067   *        graph configuration.
068   */
069  public static <N, E> Builder<N, E> builder(GraphConfig config) {
070    return new Builder<N, E>(config);
071  }
072
073  /**
074   * Returns an immutable copy of the input graph.
075   */
076  public static <N, E> ImmutableUndirectedGraph<N, E> copyOf(UndirectedGraph<N, E> graph) {
077    return new Builder<N, E>(graph).build();
078  }
079
080  /**
081   * A builder for creating immutable undirected graph instances.
082   *
083   * @param <N> Node parameter type
084   * @param <E> edge parameter type
085   * @see GraphConfig
086   */
087  public static final class Builder<N, E> extends AbstractImmutableGraph.Builder<N, E> {
088
089    /**
090     * Creates a new builder with the default graph configuration.
091     */
092    public Builder() {
093      super(Graphs.<N, E>createUndirected());
094    }
095
096    /**
097     * Creates a new builder with the specified configuration.
098     */
099    public Builder(GraphConfig config) {
100      super(Graphs.<N, E>createUndirected(config));
101    }
102
103    /**
104     * Creates a new builder whose internal state is that of {@code graph}.
105     *
106     * <p>NOTE: This constructor should only be used in the case where it will be immediately
107     * followed by a call to {@code build}, so that the input graph will not be modified.
108     * Currently the only such contexts are {@code Immutable*Graph.copyOf()}, which use these
109     * constructors to avoid making an extra copy of the graph state.
110     * @see ImmutableUndirectedGraph#copyOf(UndirectedGraph)
111     */
112    private Builder(UndirectedGraph<N, E> graph) {
113      super(graph);
114    }
115
116    @Override
117    NodeConnections<N, E> nodeConnectionsOf(N node) {
118      return UndirectedNodeConnections.ofImmutable(
119          graph.adjacentNodes(node), graph.incidentEdges(node));
120    }
121
122    @Override
123    public ImmutableUndirectedGraph<N, E> build() {
124      return new ImmutableUndirectedGraph<N, E>(this);
125    }
126
127    @Override
128    @CanIgnoreReturnValue
129    public Builder<N, E> addNode(N node) {
130      return (Builder<N, E>) super.addNode(node); // Refine the return type
131    }
132
133    @Override
134    @CanIgnoreReturnValue
135    public Builder<N, E> addEdge(E edge, N node1, N node2) {
136      return (Builder<N, E>) super.addEdge(edge, node1, node2); // Refine the return type
137    }
138
139    @Override
140    @CanIgnoreReturnValue
141    public Builder<N, E> addGraph(Graph<N, E> graphToAdd) {
142      return (Builder<N, E>) super.addGraph(graphToAdd); // Refine the return type
143    }
144  }
145}
