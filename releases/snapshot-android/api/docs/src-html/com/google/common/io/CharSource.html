041/**
042 * A readable source of characters, such as a text file. Unlike a {@link Reader}, a {@code
043 * CharSource} is not an open, stateful stream of characters that can be read and closed. Instead,
044 * it is an immutable <i>supplier</i> of {@code Reader} instances.
045 *
046 * <p>{@code CharSource} provides two kinds of methods:
047 *
048 * <ul>
049 *   <li><b>Methods that return a reader:</b> These methods should return a <i>new</i>, independent
050 *       instance each time they are called. The caller is responsible for ensuring that the
051 *       returned reader is closed.
052 *   <li><b>Convenience methods:</b> These are implementations of common operations that are
053 *       typically implemented by opening a reader using one of the methods in the first category,
054 *       doing something and finally closing the reader that was opened.
055 * </ul>
056 *
057 * <p>Several methods in this class, such as {@link #readLines()}, break the contents of the source
058 * into lines. Like {@link BufferedReader}, these methods break lines on any of {@code \n}, {@code
059 * \r} or {@code \r\n}, do not include the line separator in each line and do not consider there to
060 * be an empty line at the end if the contents are terminated with a line separator.
061 *
062 * <p>Any {@link ByteSource} containing text encoded with a specific {@linkplain Charset character
063 * encoding} may be viewed as a {@code CharSource} using {@link ByteSource#asCharSource(Charset)}.
064 *
065 * <p><b>Note:</b> In general, {@code CharSource} is intended to be used for "file-like" sources
066 * that provide readers that are:
067 *
068 * <ul>
069 *   <li><b>Finite:</b> Many operations, such as {@link #length()} and {@link #read()}, will either
070 *       block indefinitely or fail if the source creates an infinite reader.
071 *   <li><b>Non-destructive:</b> A <i>destructive</i> reader will consume or otherwise alter the
072 *       source as they are read from it. A source that provides such readers will not be reusable,
073 *       and operations that read from the stream (including {@link #length()}, in some
074 *       implementations) will prevent further operations from completing as expected.
075 * </ul>
076 *
077 * @since 14.0
078 * @author Colin Decker
079 */
080@J2ktIncompatible
081@GwtIncompatible
082@ElementTypesAreNonnullByDefault
083public abstract class CharSource {
084
085  /** Constructor for use by subclasses. */
086  protected CharSource() {}

088  /**
089   * Returns a {@link ByteSource} view of this char source that encodes chars read from this source
090   * as bytes using the given {@link Charset}.
091   *
092   * <p>If {@link ByteSource#asCharSource} is called on the returned source with the same charset,
093   * the default implementation of this method will ensure that the original {@code CharSource} is
094   * returned, rather than round-trip encoding. Subclasses that override this method should behave
095   * the same way.
096   *
097   * @since 20.0
098   */
099  @Beta
100  public ByteSource asByteSource(Charset charset) {
101    return new AsByteSource(charset);
102  }

104  /**
105   * Opens a new {@link Reader} for reading from this source. This method returns a new, independent
106   * reader each time it is called.
107   *
108   * <p>The caller is responsible for ensuring that the returned reader is closed.
109   *
110   * @throws IOException if an I/O error occurs while opening the reader
111   */
112  public abstract Reader openStream() throws IOException;
113
114  /**
115   * Opens a new {@link BufferedReader} for reading from this source. This method returns a new,
116   * independent reader each time it is called.
117   *
118   * <p>The caller is responsible for ensuring that the returned reader is closed.
119   *
120   * @throws IOException if an I/O error occurs while of opening the reader
121   */
122  public BufferedReader openBufferedStream() throws IOException {
123    Reader reader = openStream();
124    return (reader instanceof BufferedReader)
125        ? (BufferedReader) reader
126        : new BufferedReader(reader);
127  }

129  /**
130   * Returns the size of this source in chars, if the size can be easily determined without actually
131   * opening the data stream.
132   *
133   * <p>The default implementation returns {@link Optional#absent}. Some sources, such as a {@code
134   * CharSequence}, may return a non-absent value. Note that in such cases, it is <i>possible</i>
135   * that this method will return a different number of chars than would be returned by reading all
136   * of the chars.
137   *
138   * <p>Additionally, for mutable sources such as {@code StringBuilder}s, a subsequent read may
139   * return a different number of chars if the contents are changed.
140   *
141   * @since 19.0
142   */
143  @Beta
144  public Optional<Long> lengthIfKnown() {
145    return Optional.absent();
146  }
147
148  /**
149   * Returns the length of this source in chars, even if doing so requires opening and traversing an
150   * entire stream. To avoid a potentially expensive operation, see {@link #lengthIfKnown}.
151   *
152   * <p>The default implementation calls {@link #lengthIfKnown} and returns the value if present. If
153   * absent, it will fall back to a heavyweight operation that will open a stream, {@link
154   * Reader#skip(long) skip} to the end of the stream, and return the total number of chars that
155   * were skipped.
156   *
157   * <p>Note that for sources that implement {@link #lengthIfKnown} to provide a more efficient
158   * implementation, it is <i>possible</i> that this method will return a different number of chars
159   * than would be returned by reading all of the chars.
160   *
161   * <p>In either case, for mutable sources such as files, a subsequent read may return a different
162   * number of chars if the contents are changed.
163   *
164   * @throws IOException if an I/O error occurs while reading the length of this source
165   * @since 19.0
166   */
167  @Beta
168  public long length() throws IOException {
169    Optional<Long> lengthIfKnown = lengthIfKnown();
170    if (lengthIfKnown.isPresent()) {
171      return lengthIfKnown.get();
172    }
173
174    Closer closer = Closer.create();
175    try {
176      Reader reader = closer.register(openStream());
177      return countBySkipping(reader);
178    } catch (Throwable e) {
179      throw closer.rethrow(e);
180    } finally {
181      closer.close();
182    }
183  }
184
185  private long countBySkipping(Reader reader) throws IOException {
186    long count = 0;
187    long read;
188    while ((read = reader.skip(Long.MAX_VALUE)) != 0) {
189      count += read;
190    }
191    return count;
192  }

194  /**
195   * Appends the contents of this source to the given {@link Appendable} (such as a {@link Writer}).
196   * Does not close {@code appendable} if it is {@code Closeable}.
197   *
198   * @return the number of characters copied
199   * @throws IOException if an I/O error occurs while reading from this source or writing to {@code
200   *     appendable}
201   */
202  @CanIgnoreReturnValue
203  public long copyTo(Appendable appendable) throws IOException {
204    checkNotNull(appendable);
205
206    Closer closer = Closer.create();
207    try {
208      Reader reader = closer.register(openStream());
209      return CharStreams.copy(reader, appendable);
210    } catch (Throwable e) {
211      throw closer.rethrow(e);
212    } finally {
213      closer.close();
214    }
215  }

218   * Copies the contents of this source to the given sink.
219   *
220   * @return the number of characters copied
221   * @throws IOException if an I/O error occurs while reading from this source or writing to {@code
222   *     sink}
223   */
224  @CanIgnoreReturnValue
225  public long copyTo(CharSink sink) throws IOException {
226    checkNotNull(sink);
227
228    Closer closer = Closer.create();
229    try {
230      Reader reader = closer.register(openStream());
231      Writer writer = closer.register(sink.openStream());
232      return CharStreams.copy(reader, writer);
233    } catch (Throwable e) {

236      closer.close();
237    }

257  /**
258   * Reads the first line of this source as a string. Returns {@code null} if this source is empty.
259   *
260   * <p>Like {@link BufferedReader#readLine()}, this method considers a line to be a sequence of
261   * text that is terminated by (but does not include) one of {@code \r\n}, {@code \r} or {@code
262   * \n}. If the source's content does not end in a line termination sequence, it is treated as if
263   * it does.
264   *
265   * @throws IOException if an I/O error occurs while reading from this source
266   */
267  @CheckForNull
268  public String readFirstLine() throws IOException {
269    Closer closer = Closer.create();
270    try {
271      BufferedReader reader = closer.register(openBufferedStream());
272      return reader.readLine();
273    } catch (Throwable e) {
274      throw closer.rethrow(e);
275    } finally {
276      closer.close();
277    }
278  }

280  /**
281   * Reads all the lines of this source as a list of strings. The returned list will be empty if
282   * this source is empty.
283   *
284   * <p>Like {@link BufferedReader#readLine()}, this method considers a line to be a sequence of
285   * text that is terminated by (but does not include) one of {@code \r\n}, {@code \r} or {@code
286   * \n}. If the source's content does not end in a line termination sequence, it is treated as if
287   * it does.
288   *
289   * @throws IOException if an I/O error occurs while reading from this source
290   */
291  public ImmutableList<String> readLines() throws IOException {
292    Closer closer = Closer.create();
293    try {
294      BufferedReader reader = closer.register(openBufferedStream());
295      List<String> result = Lists.newArrayList();
296      String line;
297      while ((line = reader.readLine()) != null) {
298        result.add(line);
299      }
300      return ImmutableList.copyOf(result);
301    } catch (Throwable e) {
302      throw closer.rethrow(e);
303    } finally {
304      closer.close();
305    }
306  }

308  /**
309   * Reads lines of text from this source, processing each line as it is read using the given {@link
310   * LineProcessor processor}. Stops when all lines have been processed or the processor returns
311   * {@code false} and returns the result produced by the processor.
312   *
313   * <p>Like {@link BufferedReader#readLine()}, this method considers a line to be a sequence of
314   * text that is terminated by (but does not include) one of {@code \r\n}, {@code \r} or {@code
315   * \n}. If the source's content does not end in a line termination sequence, it is treated as if
316   * it does.
317   *
318   * @throws IOException if an I/O error occurs while reading from this source or if {@code
319   *     processor} throws an {@code IOException}
320   * @since 16.0
321   */
322  @Beta
323  @CanIgnoreReturnValue // some processors won't return a useful result
324  @ParametricNullness
325  public <T extends @Nullable Object> T readLines(LineProcessor<T> processor) throws IOException {
326    checkNotNull(processor);
327
328    Closer closer = Closer.create();
329    try {
330      Reader reader = closer.register(openStream());
331      return CharStreams.readLines(reader, processor);
332    } catch (Throwable e) {
333      throw closer.rethrow(e);
334    } finally {
335      closer.close();
336    }
337  }

339  /**
340   * Returns whether the source has zero chars. The default implementation first checks {@link
341   * #lengthIfKnown}, returning true if it's known to be zero and false if it's known to be
342   * non-zero. If the length is not known, it falls back to opening a stream and checking for EOF.
343   *
344   * <p>Note that, in cases where {@code lengthIfKnown} returns zero, it is <i>possible</i> that
345   * chars are actually available for reading. This means that a source may return {@code true} from
346   * {@code isEmpty()} despite having readable content.
347   *
348   * @throws IOException if an I/O error occurs
349   * @since 15.0
350   */
351  public boolean isEmpty() throws IOException {
352    Optional<Long> lengthIfKnown = lengthIfKnown();
353    if (lengthIfKnown.isPresent()) {
354      return lengthIfKnown.get() == 0L;
355    }
356    Closer closer = Closer.create();
357    try {
358      Reader reader = closer.register(openStream());
359      return reader.read() == -1;
360    } catch (Throwable e) {
361      throw closer.rethrow(e);
362    } finally {
363      closer.close();
364    }
365  }

442  /** A byte source that reads chars from this source and encodes them as bytes using a charset. */
443  private final class AsByteSource extends ByteSource {
444
445    final Charset charset;
446
447    AsByteSource(Charset charset) {
448      this.charset = checkNotNull(charset);
449    }
450
451    @Override
452    public CharSource asCharSource(Charset charset) {
453      if (charset.equals(this.charset)) {
454        return CharSource.this;
455      }
456      return super.asCharSource(charset);
457    }
458
459    @Override
460    public InputStream openStream() throws IOException {
461      return new ReaderInputStream(CharSource.this.openStream(), charset, 8192);
462    }
463
464    @Override
465    public String toString() {
466      return CharSource.this.toString() + ".asByteSource(" + charset + ")";
467    }
468  }

470  private static class CharSequenceCharSource extends CharSource {
471
472    private static final Splitter LINE_SPLITTER = Splitter.onPattern("\r\n|\n|\r");
473
474    protected final CharSequence seq;
475
476    protected CharSequenceCharSource(CharSequence seq) {
477      this.seq = checkNotNull(seq);
478    }
479
480    @Override
481    public Reader openStream() {
482      return new CharSequenceReader(seq);
483    }
484
485    @Override
486    public String read() {
487      return seq.toString();
488    }
489
490    @Override
491    public boolean isEmpty() {
492      return seq.length() == 0;
493    }
494
495    @Override
496    public long length() {
497      return seq.length();
498    }
499
500    @Override
501    public Optional<Long> lengthIfKnown() {
502      return Optional.of((long) seq.length());
503    }
504
505    /**
506     * Returns an iterator over the lines in the string. If the string ends in a newline, a final
507     * empty string is not included, to match the behavior of BufferedReader/LineReader.readLine().
508     */
509    private Iterator<String> linesIterator() {
510      return new AbstractIterator<String>() {
511        Iterator<String> lines = LINE_SPLITTER.split(seq).iterator();
512
513        @Override
514        @CheckForNull
515        protected String computeNext() {
516          if (lines.hasNext()) {
517            String next = lines.next();
518            // skip last line if it's empty
519            if (lines.hasNext() || !next.isEmpty()) {
520              return next;
521            }
522          }
523          return endOfData();
524        }
525      };
526    }
527
528    @Override
529    @CheckForNull
530    public String readFirstLine() {
531      Iterator<String> lines = linesIterator();
532      return lines.hasNext() ? lines.next() : null;
533    }
534
535    @Override
536    public ImmutableList<String> readLines() {
537      return ImmutableList.copyOf(linesIterator());
538    }
539
540    @Override
541    @ParametricNullness
542    public <T extends @Nullable Object> T readLines(LineProcessor<T> processor) throws IOException {
543      Iterator<String> lines = linesIterator();
544      while (lines.hasNext()) {
545        if (!processor.processLine(lines.next())) {
546          break;
547        }
548      }
549      return processor.getResult();
550    }
551
552    @Override
553    public String toString() {
554      return "CharSource.wrap(" + Ascii.truncate(seq, 30, "...") + ")";
555    }
556  }

558  /**
559   * Subclass specialized for string instances.
560   *
561   * <p>Since Strings are immutable and built into the jdk we can optimize some operations
562   *
563   * <ul>
564   *   <li>use {@link StringReader} instead of {@link CharSequenceReader}. It is faster since it can
565   *       use {@link String#getChars(int, int, char[], int)} instead of copying characters one by
566   *       one with {@link CharSequence#charAt(int)}.
567   *   <li>use {@link Appendable#append(CharSequence)} in {@link #copyTo(Appendable)} and {@link
568   *       #copyTo(CharSink)}. We know this is correct since strings are immutable and so the length
569   *       can't change, and it is faster because many writers and appendables are optimized for
570   *       appending string instances.
571   * </ul>
572   */
573  private static class StringCharSource extends CharSequenceCharSource {
574    protected StringCharSource(String seq) {
575      super(seq);
576    }
577
578    @Override
579    public Reader openStream() {
580      return new StringReader((String) seq);
581    }
582
583    @Override
584    public long copyTo(Appendable appendable) throws IOException {
585      appendable.append(seq);
586      return seq.length();
587    }
588
589    @Override
590    public long copyTo(CharSink sink) throws IOException {
591      checkNotNull(sink);
592      Closer closer = Closer.create();
593      try {
594        Writer writer = closer.register(sink.openStream());
595        writer.write((String) seq);
596        return seq.length();
597      } catch (Throwable e) {
598        throw closer.rethrow(e);
599      } finally {
600        closer.close();
601      }
602    }
603  }

619  private static final class ConcatenatedCharSource extends CharSource {
620
621    private final Iterable<? extends CharSource> sources;
622
623    ConcatenatedCharSource(Iterable<? extends CharSource> sources) {
624      this.sources = checkNotNull(sources);
625    }
626
627    @Override
628    public Reader openStream() throws IOException {
629      return new MultiReader(sources.iterator());
630    }
631
632    @Override
633    public boolean isEmpty() throws IOException {
634      for (CharSource source : sources) {
635        if (!source.isEmpty()) {
636          return false;
637        }
638      }
639      return true;
640    }
641
642    @Override
643    public Optional<Long> lengthIfKnown() {
644      long result = 0L;
645      for (CharSource source : sources) {
646        Optional<Long> lengthIfKnown = source.lengthIfKnown();
647        if (!lengthIfKnown.isPresent()) {
648          return Optional.absent();
649        }
650        result += lengthIfKnown.get();
651      }
652      return Optional.of(result);
653    }
654
655    @Override
656    public long length() throws IOException {
657      long result = 0L;
658      for (CharSource source : sources) {
659        result += source.length();
660      }
661      return result;
662    }
663
664    @Override
665    public String toString() {
666      return "CharSource.concat(" + sources + ")";
667    }
668  }
