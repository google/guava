046/**
047 * A {@link NavigableMap} whose contents will never change, with many other important properties
048 * detailed at {@link ImmutableCollection}.
049 *
050 * <p><b>Warning:</b> as with any sorted collection, you are strongly advised not to use a {@link
051 * Comparator} or {@link Comparable} type whose comparison behavior is <i>inconsistent with
052 * equals</i>. That is, {@code a.compareTo(b)} or {@code comparator.compare(a, b)} should equal zero
053 * <i>if and only if</i> {@code a.equals(b)}. If this advice is not followed, the resulting map will
054 * not correctly obey its specification.
055 *
056 * <p>See the Guava User Guide article on <a href=
057 * "https://github.com/google/guava/wiki/ImmutableCollectionsExplained">immutable collections</a>.
058 *
059 * @author Jared Levy
060 * @author Louis Wasserman
061 * @since 2.0 (implements {@code NavigableMap} since 12.0)
062 */
063@GwtCompatible(serializable = true, emulated = true)
064@ElementTypesAreNonnullByDefault
065public final class ImmutableSortedMap<K, V> extends ImmutableMap<K, V>
066    implements NavigableMap<K, V> {

422  /**
423   * Returns an immutable map containing the same entries as the provided sorted map, with the same
424   * ordering.
425   *
426   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
427   * safe to do so. The exact circumstances under which a copy will or will not be performed are
428   * undocumented and subject to change.
429   *
430   * @throws NullPointerException if any key or value in {@code map} is null
431   */
432  @SuppressWarnings("unchecked")
433  public static <K, V> ImmutableSortedMap<K, V> copyOfSorted(SortedMap<K, ? extends V> map) {
434    Comparator<? super K> comparator = map.comparator();
435    if (comparator == null) {
436      // If map has a null comparator, the keys should have a natural ordering,
437      // even though K doesn't explicitly implement Comparable.
438      comparator = (Comparator<? super K>) NATURAL_ORDER;
439    }
440    if (map instanceof ImmutableSortedMap) {
441      // TODO(kevinb): Prove that this cast is safe, even though
442      // Collections.unmodifiableSortedMap requires the same key type.
443      @SuppressWarnings("unchecked")
444      ImmutableSortedMap<K, V> kvMap = (ImmutableSortedMap<K, V>) map;
445      if (!kvMap.isPartialView()) {
446        return kvMap;
447      }
448    }
449    return fromEntries(comparator, true, map.entrySet());
450  }

452  private static <K, V> ImmutableSortedMap<K, V> copyOfInternal(
453      Map<? extends K, ? extends V> map, Comparator<? super K> comparator) {
454    boolean sameComparator = false;
455    if (map instanceof SortedMap) {
456      SortedMap<?, ?> sortedMap = (SortedMap<?, ?>) map;
457      Comparator<?> comparator2 = sortedMap.comparator();
458      sameComparator =
459          (comparator2 == null) ? comparator == NATURAL_ORDER : comparator.equals(comparator2);
460    }
461
462    if (sameComparator && (map instanceof ImmutableSortedMap)) {
463      // TODO(kevinb): Prove that this cast is safe, even though
464      // Collections.unmodifiableSortedMap requires the same key type.
465      @SuppressWarnings("unchecked")
466      ImmutableSortedMap<K, V> kvMap = (ImmutableSortedMap<K, V>) map;
467      if (!kvMap.isPartialView()) {
468        return kvMap;
469      }
470    }
471    return fromEntries(comparator, sameComparator, map.entrySet());
472  }

479  /**
480   * Accepts a collection of possibly-null entries. If {@code sameComparator}, then it is assumed
481   * that they do not need to be sorted or checked for dupes.
482   */
483  private static <K, V> ImmutableSortedMap<K, V> fromEntries(
484      Comparator<? super K> comparator,
485      boolean sameComparator,
486      Iterable<? extends Entry<? extends K, ? extends V>> entries) {
487    // "adding" type params to an array of a raw type should be safe as
488    // long as no one can ever cast that same array instance back to a
489    // raw type.
490    @SuppressWarnings("unchecked")
491    Entry<K, V>[] entryArray = (Entry[]) Iterables.toArray(entries, EMPTY_ENTRY_ARRAY);
492    return fromEntries(comparator, sameComparator, entryArray, entryArray.length);
493  }

495  @SuppressWarnings("nullness") // TODO: b/316358623 - Remove after checker fix.
496  private static <K, V> ImmutableSortedMap<K, V> fromEntries(
497      final Comparator<? super K> comparator,
498      boolean sameComparator,
499      @Nullable Entry<K, V>[] entryArray,
500      int size) {
501    switch (size) {
502      case 0:
503        return emptyMap(comparator);
504      case 1:
505        // requireNonNull is safe because the first `size` elements have been filled in.
506        Entry<K, V> onlyEntry = requireNonNull(entryArray[0]);
507        return of(comparator, onlyEntry.getKey(), onlyEntry.getValue());
508      default:
509        Object[] keys = new Object[size];
510        Object[] values = new Object[size];
511        if (sameComparator) {
512          // Need to check for nulls, but don't need to sort or validate.
513          for (int i = 0; i < size; i++) {
514            // requireNonNull is safe because the first `size` elements have been filled in.
515            Entry<K, V> entry = requireNonNull(entryArray[i]);
516            Object key = entry.getKey();
517            Object value = entry.getValue();
518            checkEntryNotNull(key, value);
519            keys[i] = key;
520            values[i] = value;
521          }
522        } else {
523          // Need to sort and check for nulls and dupes.
524          // Inline the Comparator implementation rather than transforming with a Function
525          // to save code size.
526          Arrays.sort(
527              entryArray,
528              0,
529              size,
530              (e1, e2) -> {
531                // requireNonNull is safe because the first `size` elements have been filled in.
532                requireNonNull(e1);
533                requireNonNull(e2);
534                return comparator.compare(e1.getKey(), e2.getKey());
535              });
536          // requireNonNull is safe because the first `size` elements have been filled in.
537          Entry<K, V> firstEntry = requireNonNull(entryArray[0]);
538          K prevKey = firstEntry.getKey();
539          keys[0] = prevKey;
540          values[0] = firstEntry.getValue();
541          checkEntryNotNull(keys[0], values[0]);
542          for (int i = 1; i < size; i++) {
543            // requireNonNull is safe because the first `size` elements have been filled in.
544            Entry<K, V> prevEntry = requireNonNull(entryArray[i - 1]);
545            Entry<K, V> entry = requireNonNull(entryArray[i]);
546            K key = entry.getKey();
547            V value = entry.getValue();
548            checkEntryNotNull(key, value);
549            keys[i] = key;
550            values[i] = value;
551            checkNoConflict(comparator.compare(prevKey, key) != 0, "key", prevEntry, entry);
552            prevKey = key;
553          }
554        }
555        return new ImmutableSortedMap<>(
556            new RegularImmutableSortedSet<K>(ImmutableList.<K>asImmutableList(keys), comparator),
557            ImmutableList.<V>asImmutableList(values));
558    }
559  }

589  /**
590   * A builder for creating immutable sorted map instances, especially {@code public static final}
591   * maps ("constant maps"). Example:
592   *
593   * <pre>{@code
594   * static final ImmutableSortedMap<Integer, String> INT_TO_WORD =
595   *     new ImmutableSortedMap.Builder<Integer, String>(Ordering.natural())
596   *         .put(1, "one")
597   *         .put(2, "two")
598   *         .put(3, "three")
599   *         .buildOrThrow();
600   * }</pre>
601   *
602   * <p>For <i>small</i> immutable sorted maps, the {@code ImmutableSortedMap.of()} methods are even
603   * more convenient.
604   *

606   * build multiple maps in series. Each map is a superset of the maps created before it.

611    private transient @Nullable Object[] keys;
612    private transient @Nullable Object[] values;
613    private final Comparator<? super K> comparator;

615    /**
616     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
617     * ImmutableSortedMap#orderedBy}.

620      this(comparator, ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY);

622
623    private Builder(Comparator<? super K> comparator, int initialCapacity) {
624      this.comparator = checkNotNull(comparator);
625      this.keys = new @Nullable Object[initialCapacity];
626      this.values = new @Nullable Object[initialCapacity];

629    private void ensureCapacity(int minCapacity) {
630      if (minCapacity > keys.length) {
631        int newCapacity = ImmutableCollection.Builder.expandedCapacity(keys.length, minCapacity);
632        this.keys = Arrays.copyOf(keys, newCapacity);
633        this.values = Arrays.copyOf(values, newCapacity);
634      }
635    }

637    /**
638     * Associates {@code key} with {@code value} in the built map. Duplicate keys, according to the
639     * comparator (which might be the keys' natural order), are not allowed, and will cause {@link
640     * #build} to fail.
641     */
642    @CanIgnoreReturnValue
643    @Override
644    public Builder<K, V> put(K key, V value) {
645      ensureCapacity(size + 1);
646      checkEntryNotNull(key, value);
647      keys[size] = key;
648      values[size] = value;
649      size++;
650      return this;
651    }

710    @CanIgnoreReturnValue
711    Builder<K, V> combine(ImmutableSortedMap.Builder<K, V> other) {
712      ensureCapacity(size + other.size);
713      System.arraycopy(other.keys, 0, this.keys, this.size, other.size);
714      System.arraycopy(other.values, 0, this.values, this.size, other.size);

719    /**
720     * Returns a newly-created immutable sorted map.
721     *
722     * <p>Prefer the equivalent method {@link #buildOrThrow()} to make it explicit that the method
723     * will throw an exception if there are duplicate keys. The {@code build()} method will soon be
724     * deprecated.
725     *
726     * @throws IllegalArgumentException if any two keys are equal according to the comparator (which
727     *     might be the keys' natural order)
728     */
729    @Override
730    public ImmutableSortedMap<K, V> build() {
731      return buildOrThrow();
732    }
733
734    /**
735     * Returns a newly-created immutable sorted map, or throws an exception if any two keys are
736     * equal.
737     *
738     * @throws IllegalArgumentException if any two keys are equal according to the comparator (which
739     *     might be the keys' natural order)
740     * @since 31.0
741     */
742    @Override
743    public ImmutableSortedMap<K, V> buildOrThrow() {
744      switch (size) {
745        case 0:
746          return emptyMap(comparator);
747        case 1:
748          // requireNonNull is safe because the first `size` elements have been filled in.
749          return of(comparator, (K) requireNonNull(keys[0]), (V) requireNonNull(values[0]));
750        default:
751          Object[] sortedKeys = Arrays.copyOf(keys, size);
752          Arrays.sort((K[]) sortedKeys, comparator);
753          Object[] sortedValues = new Object[size];
754
755          // We might, somehow, be able to reorder values in-place.  But it doesn't seem like
756          // there's a way around creating the separate sortedKeys array, and if we're allocating
757          // one array of size n, we might as well allocate two -- to say nothing of the allocation
758          // done in Arrays.sort.
759          for (int i = 0; i < size; i++) {
760            if (i > 0 && comparator.compare((K) sortedKeys[i - 1], (K) sortedKeys[i]) == 0) {
761              throw new IllegalArgumentException(
762                  "keys required to be distinct but compared as equal: "
763                      + sortedKeys[i - 1]
764                      + " and "
765                      + sortedKeys[i]);
766            }
767            // requireNonNull is safe because the first `size` elements have been filled in.
768            int index =
769                Arrays.binarySearch((K[]) sortedKeys, (K) requireNonNull(keys[i]), comparator);
770            sortedValues[index] = requireNonNull(values[i]);
771          }
772          return new ImmutableSortedMap<K, V>(
773              new RegularImmutableSortedSet<K>(
774                  ImmutableList.<K>asImmutableList(sortedKeys), comparator),
775              ImmutableList.<V>asImmutableList(sortedValues));
776      }
777    }
