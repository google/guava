032/**

034 *
035 * <p>All methods return serializable predicates as long as they're given serializable parameters.

038 * href="https://github.com/google/guava/wiki/FunctionalExplained">the use of {@code Predicate}</a>.

089  /**
090   * Returns a predicate that evaluates to {@code true} if each of its components evaluates to
091   * {@code true}. The components are evaluated in order, and evaluation will be "short-circuited"
092   * as soon as a false predicate is found. It defensively copies the iterable passed in, so future
093   * changes to it won't alter the behavior of this predicate. If {@code components} is empty, the
094   * returned predicate will always evaluate to {@code true}.
095   */
096  public static <T extends @Nullable Object> Predicate<T> and(
097      Iterable<? extends Predicate<? super T>> components) {
098    return new AndPredicate<>(defensiveCopy(components));
099  }

104   * as soon as a false predicate is found. It defensively copies the array passed in, so future
105   * changes to it won't alter the behavior of this predicate. If {@code components} is empty, the
106   * returned predicate will always evaluate to {@code true}.

167  /**
168   * Returns a predicate that evaluates to {@code true} if the object being tested is an instance of
169   * the given class. If the object being tested is {@code null} this predicate evaluates to {@code
170   * false}.
171   *
172   * <p>If you want to filter an {@code Iterable} to narrow its type, consider using {@link
173   * com.google.common.collect.Iterables#filter(Iterable, Class)} in preference.
174   *
175   * <p><b>Warning:</b> contrary to the typical assumptions about predicates (as documented at
176   * {@link Predicate#apply}), the returned predicate may not be <i>consistent with equals</i>. For
177   * example, {@code instanceOf(ArrayList.class)} will yield different results for the two equal
178   * instances {@code Lists.newArrayList(1)} and {@code Arrays.asList(1)}.
179   */
180  @GwtIncompatible // Class.isInstance
181  public static <T extends @Nullable Object> Predicate<T> instanceOf(Class<?> clazz) {
182    return new InstanceOfPredicate<>(clazz);
183  }

186   * Returns a predicate that evaluates to {@code true} if the class being tested is assignable to
187   * (is a subtype of) {@code clazz}. Example:
188   *
189   * <pre>{@code
190   * List<Class<?>> classes = Arrays.asList(
191   *     Object.class, String.class, Number.class, Long.class);
192   * return Iterables.filter(classes, subtypeOf(Number.class));
193   * }</pre>
194   *
195   * The code above returns an iterable containing {@code Number.class} and {@code Long.class}.
196   *
197   * @since 20.0 (since 10.0 under the incorrect name {@code assignableFrom})
198   */

200  @GwtIncompatible // Class.isAssignableFrom
201  @Beta
202  public static Predicate<Class<?>> subtypeOf(Class<?> clazz) {
203    return new SubtypeOfPredicate(clazz);
204  }

318  /** @see Predicates#not(Predicate) */
319  private static class NotPredicate<T extends @Nullable Object>
320      implements Predicate<T>, Serializable {
321    final Predicate<T> predicate;
322
323    NotPredicate(Predicate<T> predicate) {
324      this.predicate = checkNotNull(predicate);
325    }
326
327    @Override
328    public boolean apply(@ParametricNullness T t) {
329      return !predicate.apply(t);
330    }
331
332    @Override
333    public int hashCode() {
334      return ~predicate.hashCode();
335    }
336
337    @Override
338    public boolean equals(@CheckForNull Object obj) {
339      if (obj instanceof NotPredicate) {
340        NotPredicate<?> that = (NotPredicate<?>) obj;
341        return predicate.equals(that.predicate);
342      }
343      return false;
344    }
345
346    @Override
347    public String toString() {
348      return "Predicates.not(" + predicate + ")";
349    }
350
351    private static final long serialVersionUID = 0;
352  }

354  /** @see Predicates#and(Iterable) */
355  private static class AndPredicate<T extends @Nullable Object>
356      implements Predicate<T>, Serializable {
357    private final List<? extends Predicate<? super T>> components;
358
359    private AndPredicate(List<? extends Predicate<? super T>> components) {
360      this.components = components;
361    }
362
363    @Override
364    public boolean apply(@ParametricNullness T t) {
365      // Avoid using the Iterator to avoid generating garbage (issue 820).
366      for (int i = 0; i < components.size(); i++) {
367        if (!components.get(i).apply(t)) {
368          return false;
369        }
370      }
371      return true;
372    }
373
374    @Override
375    public int hashCode() {
376      // add a random number to avoid collisions with OrPredicate
377      return components.hashCode() + 0x12472c2c;
378    }
379
380    @Override
381    public boolean equals(@CheckForNull Object obj) {
382      if (obj instanceof AndPredicate) {
383        AndPredicate<?> that = (AndPredicate<?>) obj;
384        return components.equals(that.components);
385      }
386      return false;
387    }
388
389    @Override
390    public String toString() {
391      return toStringHelper("and", components);
392    }
393
394    private static final long serialVersionUID = 0;
395  }

440  private static String toStringHelper(String methodName, Iterable<?> components) {
441    StringBuilder builder = new StringBuilder("Predicates.").append(methodName).append('(');
442    boolean first = true;
443    for (Object o : components) {
444      if (!first) {
445        builder.append(',');
446      }
447      builder.append(o);
448      first = false;
449    }
450    return builder.append(')').toString();
451  }

611  /** @see Predicates#compose(Predicate, Function) */
612  private static class CompositionPredicate<A extends @Nullable Object, B extends @Nullable Object>
613      implements Predicate<A>, Serializable {
614    final Predicate<B> p;
615    final Function<A, ? extends B> f;
616
617    private CompositionPredicate(Predicate<B> p, Function<A, ? extends B> f) {
618      this.p = checkNotNull(p);
619      this.f = checkNotNull(f);
620    }
621
622    @Override
623    public boolean apply(@ParametricNullness A a) {
624      return p.apply(f.apply(a));
625    }
626
627    @Override
628    public boolean equals(@CheckForNull Object obj) {
629      if (obj instanceof CompositionPredicate) {
630        CompositionPredicate<?, ?> that = (CompositionPredicate<?, ?>) obj;
631        return f.equals(that.f) && p.equals(that.p);
632      }
633      return false;
634    }
635
636    @Override
637    public int hashCode() {
638      return f.hashCode() ^ p.hashCode();
639    }
640
641    @Override
642    public String toString() {
643      // TODO(cpovirk): maybe make this look like the method call does ("Predicates.compose(...)")
644      return p + "(" + f + ")";
645    }
646
647    private static final long serialVersionUID = 0;
648  }

650  /**
651   * @see Predicates#contains(Pattern)
652   */
653  @J2ktIncompatible
654  @GwtIncompatible // Only used by other GWT-incompatible code.
655  private static class ContainsPatternPredicate implements Predicate<CharSequence>, Serializable {
656    final CommonPattern pattern;
657
658    ContainsPatternPredicate(CommonPattern pattern) {
659      this.pattern = checkNotNull(pattern);
660    }
661
662    @Override
663    public boolean apply(CharSequence t) {
664      return pattern.matcher(t).find();
665    }
666
667    @Override
668    public int hashCode() {
669      // Pattern uses Object.hashCode, so we have to reach
670      // inside to build a hashCode consistent with equals.
671
672      return Objects.hashCode(pattern.pattern(), pattern.flags());
673    }
674
675    @Override
676    public boolean equals(@CheckForNull Object obj) {
677      if (obj instanceof ContainsPatternPredicate) {
678        ContainsPatternPredicate that = (ContainsPatternPredicate) obj;
679
680        // Pattern uses Object (identity) equality, so we have to reach
681        // inside to compare individual fields.
682        return Objects.equal(pattern.pattern(), that.pattern.pattern())
683            && pattern.flags() == that.pattern.flags();
684      }
685      return false;
686    }
687
688    @Override
689    public String toString() {
690      String patternString =
691          MoreObjects.toStringHelper(pattern)
692              .add("pattern", pattern.pattern())
693              .add("pattern.flags", pattern.flags())
694              .toString();
695      return "Predicates.contains(" + patternString + ")";
696    }
697
698    private static final long serialVersionUID = 0;
699  }

726  private static <T> List<T> defensiveCopy(T... array) {
727    return defensiveCopy(Arrays.asList(array));

730  static <T> List<T> defensiveCopy(Iterable<T> iterable) {
731    ArrayList<T> list = new ArrayList<>();
732    for (T element : iterable) {
733      list.add(checkNotNull(element));
734    }
735    return list;
