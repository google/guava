183   * Returns an immutable list containing the given elements, in order.
184   *
185   * <p>The array {@code others} must not be longer than {@code Integer.MAX_VALUE - 12}.
186   *
187   * @throws NullPointerException if any element is null
188   * @since 3.0 (source-compatible since 2.0)

190  @SafeVarargs // For Eclipse. For internal javac we have disabled this pointless type of warning.
191  public static <E> ImmutableList<E> of(
192      E e1, E e2, E e3, E e4, E e5, E e6, E e7, E e8, E e9, E e10, E e11, E e12, E... others) {
193    checkArgument(
194        others.length <= Integer.MAX_VALUE - 12, "the total number of elements must fit in an int");
195    Object[] array = new Object[12 + others.length];
196    array[0] = e1;
197    array[1] = e2;
198    array[2] = e3;
199    array[3] = e4;
200    array[4] = e5;
201    array[5] = e6;
202    array[6] = e7;
203    array[7] = e8;
204    array[8] = e9;
205    array[9] = e10;
206    array[10] = e11;
207    array[11] = e12;
208    System.arraycopy(others, 0, array, 12, others.length);
209    return construct(array);
210  }

213   * Returns an immutable list containing the given elements, in order. If {@code elements} is a
214   * {@link Collection}, this method behaves exactly as {@link #copyOf(Collection)}; otherwise, it
215   * behaves exactly as {@code copyOf(elements.iterator()}.
216   *
217   * @throws NullPointerException if {@code elements} contains a null element
218   */
219  public static <E> ImmutableList<E> copyOf(Iterable<? extends E> elements) {
220    checkNotNull(elements); // TODO(kevinb): is this here only for GWT?
221    return (elements instanceof Collection)
222        ? copyOf((Collection<? extends E>) elements)
223        : copyOf(elements.iterator());
224  }

226  /**
227   * Returns an immutable list containing the given elements, in order.
228   *
229   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
230   * safe to do so. The exact circumstances under which a copy will or will not be performed are
231   * undocumented and subject to change.
232   *
233   * <p>Note that if {@code list} is a {@code List<String>}, then {@code ImmutableList.copyOf(list)}
234   * returns an {@code ImmutableList<String>} containing each of the strings in {@code list}, while
235   * {@code ImmutableList.of(list)} returns an {@code ImmutableList<List<String>>} containing one
236   * element (the given list itself).
237   *
238   * <p>This method is safe to use even when {@code elements} is a synchronized or concurrent
239   * collection that is currently being modified by another thread.
240   *
241   * @throws NullPointerException if {@code elements} contains a null element
242   */
243  public static <E> ImmutableList<E> copyOf(Collection<? extends E> elements) {
244    if (elements instanceof ImmutableCollection) {
245      @SuppressWarnings("unchecked") // all supported methods are covariant
246      ImmutableList<E> list = ((ImmutableCollection<E>) elements).asList();
247      return list.isPartialView() ? ImmutableList.<E>asImmutableList(list.toArray()) : list;
248    }
249    return construct(elements.toArray());
250  }

252  /**
253   * Returns an immutable list containing the given elements, in order.
254   *
255   * @throws NullPointerException if {@code elements} contains a null element
256   */
257  public static <E> ImmutableList<E> copyOf(Iterator<? extends E> elements) {
258    // We special-case for 0 or 1 elements, but going further is madness.
259    if (!elements.hasNext()) {
260      return of();
261    }
262    E first = elements.next();
263    if (!elements.hasNext()) {
264      return of(first);
265    } else {
266      return new ImmutableList.Builder<E>().add(first).addAll(elements).build();
267    }
268  }

282  /**
283   * Returns an immutable list containing the given elements, sorted according to their natural
284   * order. The sorting algorithm used is stable, so elements that compare as equal will stay in the
285   * order in which they appear in the input.
286   *
287   * <p>If your data has no duplicates, or you wish to deduplicate elements, use {@code
288   * ImmutableSortedSet.copyOf(elements)}; if you want a {@code List} you can use its {@code
289   * asList()} view.
290   *
291   * <p><b>Java 8 users:</b> If you want to convert a {@link java.util.stream.Stream} to a sorted
292   * {@code ImmutableList}, use {@code stream.sorted().collect(toImmutableList())}.
293   *
294   * @throws NullPointerException if any element in the input is null
295   * @since 21.0
296   */
297  public static <E extends Comparable<? super E>> ImmutableList<E> sortedCopyOf(
298      Iterable<? extends E> elements) {
299    Comparable<?>[] array = Iterables.toArray(elements, new Comparable<?>[0]);
300    checkElementsNotNull((Object[]) array);
301    Arrays.sort(array);
302    return asImmutableList(array);
303  }

305  /**
306   * Returns an immutable list containing the given elements, in sorted order relative to the
307   * specified comparator. The sorting algorithm used is stable, so elements that compare as equal
308   * will stay in the order in which they appear in the input.
309   *
310   * <p>If your data has no duplicates, or you wish to deduplicate elements, use {@code
311   * ImmutableSortedSet.copyOf(comparator, elements)}; if you want a {@code List} you can use its
312   * {@code asList()} view.
313   *
314   * <p><b>Java 8 users:</b> If you want to convert a {@link java.util.stream.Stream} to a sorted
315   * {@code ImmutableList}, use {@code stream.sorted(comparator).collect(toImmutableList())}.
316   *
317   * @throws NullPointerException if any element in the input is null
318   * @since 21.0
319   */
320  public static <E> ImmutableList<E> sortedCopyOf(
321      Comparator<? super E> comparator, Iterable<? extends E> elements) {
322    checkNotNull(comparator);
323    @SuppressWarnings("unchecked") // all supported methods are covariant
324    E[] array = (E[]) Iterables.toArray(elements);
325    checkElementsNotNull(array);
326    Arrays.sort(array, comparator);
327    return asImmutableList(array);
328  }

412  /**
413   * Returns an immutable list of the elements between the specified {@code fromIndex}, inclusive,
414   * and {@code toIndex}, exclusive. (If {@code fromIndex} and {@code toIndex} are equal, the empty
415   * immutable list is returned.)
416   *
417   * <p><b>Note:</b> in almost all circumstances, the returned {@link ImmutableList} retains a
418   * strong reference to {@code this}, which may prevent the original list from being garbage
419   * collected. If you want the original list to be eligible for garbage collection, you should
420   * create and use a copy of the sub list (e.g., {@code
421   * ImmutableList.copyOf(originalList.subList(...))}).
422   */
423  @Override
424  public ImmutableList<E> subList(int fromIndex, int toIndex) {
425    checkPositionIndexes(fromIndex, toIndex, size());
426    int length = toIndex - fromIndex;
427    if (length == size()) {
428      return this;
429    } else if (length == 0) {
430      return of();
431    } else {
432      return subListUnchecked(fromIndex, toIndex);
433    }
434  }

436  /**
437   * Called by the default implementation of {@link #subList} when {@code toIndex - fromIndex > 1},
438   * after index validation has already been performed.
439   */
440  ImmutableList<E> subListUnchecked(int fromIndex, int toIndex) {
441    return new SubList(fromIndex, toIndex - fromIndex);
442  }

561  @Override
562  int copyIntoArray(@Nullable Object[] dst, int offset) {
563    // this loop is faster for RandomAccess instances, which ImmutableLists are
564    int size = size();
565    for (int i = 0; i < size; i++) {
566      dst[offset + i] = get(i);
567    }
568    return offset + size;
569  }

572   * Returns a view of this immutable list in reverse order. For example, {@code ImmutableList.of(1,
573   * 2, 3).reverse()} is equivalent to {@code ImmutableList.of(3, 2, 1)}.

575   * @return a view of this immutable list in reverse order

579    return (size() <= 1) ? this : new ReverseImmutableList<E>(this);

582  private static class ReverseImmutableList<E> extends ImmutableList<E> {
583    private final transient ImmutableList<E> forwardList;
584
585    ReverseImmutableList(ImmutableList<E> backingList) {
586      this.forwardList = backingList;
587    }
588
589    private int reverseIndex(int index) {
590      return (size() - 1) - index;
591    }
592
593    private int reversePosition(int index) {
594      return size() - index;
595    }
596
597    @Override
598    public ImmutableList<E> reverse() {
599      return forwardList;
600    }
601
602    @Override
603    public boolean contains(@CheckForNull Object object) {
604      return forwardList.contains(object);
605    }
606
607    @Override
608    public int indexOf(@CheckForNull Object object) {
609      int index = forwardList.lastIndexOf(object);
610      return (index >= 0) ? reverseIndex(index) : -1;
611    }
612
613    @Override
614    public int lastIndexOf(@CheckForNull Object object) {
615      int index = forwardList.indexOf(object);
616      return (index >= 0) ? reverseIndex(index) : -1;
617    }
618
619    @Override
620    public ImmutableList<E> subList(int fromIndex, int toIndex) {
621      checkPositionIndexes(fromIndex, toIndex, size());
622      return forwardList.subList(reversePosition(toIndex), reversePosition(fromIndex)).reverse();
623    }
624
625    @Override
626    public E get(int index) {
627      checkElementIndex(index, size());
628      return forwardList.get(reverseIndex(index));
629    }
630
631    @Override
632    public int size() {
633      return forwardList.size();
634    }
635
636    @Override
637    boolean isPartialView() {
638      return forwardList.isPartialView();
639    }
640  }

647  @Override
648  public int hashCode() {
649    int hashCode = 1;
650    int n = size();
651    for (int i = 0; i < n; i++) {
652      hashCode = 31 * hashCode + get(i).hashCode();
653
654      hashCode = ~~hashCode;
655      // needed to deal with GWT integer overflow
656    }
657    return hashCode;

693   */
694  public static <E> Builder<E> builder() {
695    return new Builder<E>();
696  }

698  /**
699   * Returns a new builder, expecting the specified number of elements to be added.
700   *
701   * <p>If {@code expectedSize} is exactly the number of elements added to the builder before {@link
702   * Builder#build} is called, the builder is likely to perform better than an unsized {@link
703   * #builder()} would have.
704   *
705   * <p>It is not specified if any performance benefits apply if {@code expectedSize} is close to,
706   * but not exactly, the number of elements added to the builder.
707   *
708   * @since 23.1
709   */
710  public static <E> Builder<E> builderWithExpectedSize(int expectedSize) {
711    checkNonnegative(expectedSize, "expectedSize");
712    return new ImmutableList.Builder<E>(expectedSize);
713  }

715  /**
716   * A builder for creating immutable list instances, especially {@code public static final} lists
717   * ("constant lists"). Example:
718   *
719   * <pre>{@code
720   * public static final ImmutableList<Color> GOOGLE_COLORS
721   *     = new ImmutableList.Builder<Color>()
722   *         .addAll(WEBSAFE_COLORS)
723   *         .add(new Color(0, 191, 255))
724   *         .build();
725   * }</pre>
726   *
727   * <p>Elements appear in the resulting list in the same order they were added to the builder.
728   *
729   * <p>Builder instances can be reused; it is safe to call {@link #build} multiple times to build
730   * multiple lists in series. Each new list contains all the elements of the ones created before
731   * it.
732   *
733   * @since 2.0
734   */
735  public static final class Builder<E> extends ImmutableCollection.ArrayBasedBuilder<E> {
736    /**
737     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
738     * ImmutableList#builder}.
739     */
740    public Builder() {
741      this(DEFAULT_INITIAL_CAPACITY);
742    }
743
744    Builder(int capacity) {
745      super(capacity);
746    }
747
748    /**
749     * Adds {@code element} to the {@code ImmutableList}.
750     *
751     * @param element the element to add
752     * @return this {@code Builder} object
753     * @throws NullPointerException if {@code element} is null
754     */
755    @CanIgnoreReturnValue
756    @Override
757    public Builder<E> add(E element) {
758      super.add(element);
759      return this;
760    }
761
762    /**
763     * Adds each element of {@code elements} to the {@code ImmutableList}.
764     *
765     * @param elements the {@code Iterable} to add to the {@code ImmutableList}
766     * @return this {@code Builder} object
767     * @throws NullPointerException if {@code elements} is null or contains a null element
768     */
769    @CanIgnoreReturnValue
770    @Override
771    public Builder<E> add(E... elements) {
772      super.add(elements);
773      return this;
774    }
775
776    /**
777     * Adds each element of {@code elements} to the {@code ImmutableList}.
778     *
779     * @param elements the {@code Iterable} to add to the {@code ImmutableList}
780     * @return this {@code Builder} object
781     * @throws NullPointerException if {@code elements} is null or contains a null element
782     */
783    @CanIgnoreReturnValue
784    @Override
785    public Builder<E> addAll(Iterable<? extends E> elements) {
786      super.addAll(elements);
787      return this;
788    }
789
790    /**
791     * Adds each element of {@code elements} to the {@code ImmutableList}.
792     *
793     * @param elements the {@code Iterator} to add to the {@code ImmutableList}
794     * @return this {@code Builder} object
795     * @throws NullPointerException if {@code elements} is null or contains a null element
796     */
797    @CanIgnoreReturnValue
798    @Override
799    public Builder<E> addAll(Iterator<? extends E> elements) {
800      super.addAll(elements);
801      return this;
802    }
803
804    @CanIgnoreReturnValue
805    Builder<E> combine(Builder<E> other) {
806      addAll(other.contents, other.size);
807      return this;
808    }
809
810    /**
811     * Returns a newly-created {@code ImmutableList} based on the contents of the {@code Builder}.
812     */
813    @Override
814    public ImmutableList<E> build() {
815      forceCopy = true;
816      return asImmutableList(contents, size);
817    }
818  }
