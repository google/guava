184   * Returns an immutable list containing the given elements, in order.
185   *
186   * <p>The array {@code others} must not be longer than {@code Integer.MAX_VALUE - 12}.
187   *
188   * @throws NullPointerException if any element is null
189   * @since 3.0 (source-compatible since 2.0)

191  @SafeVarargs // For Eclipse. For internal javac we have disabled this pointless type of warning.
192  public static <E> ImmutableList<E> of(
193      E e1, E e2, E e3, E e4, E e5, E e6, E e7, E e8, E e9, E e10, E e11, E e12, E... others) {
194    checkArgument(
195        others.length <= Integer.MAX_VALUE - 12, "the total number of elements must fit in an int");
196    Object[] array = new Object[12 + others.length];
197    array[0] = e1;
198    array[1] = e2;
199    array[2] = e3;
200    array[3] = e4;
201    array[4] = e5;
202    array[5] = e6;
203    array[6] = e7;
204    array[7] = e8;
205    array[8] = e9;
206    array[9] = e10;
207    array[10] = e11;
208    array[11] = e12;
209    System.arraycopy(others, 0, array, 12, others.length);
210    return construct(array);
211  }

213  /**
214   * Returns an immutable list containing the given elements, in order. If {@code elements} is a
215   * {@link Collection}, this method behaves exactly as {@link #copyOf(Collection)}; otherwise, it
216   * behaves exactly as {@code copyOf(elements.iterator()}.
217   *
218   * @throws NullPointerException if {@code elements} contains a null element
219   */
220  public static <E> ImmutableList<E> copyOf(Iterable<? extends E> elements) {
221    checkNotNull(elements); // TODO(kevinb): is this here only for GWT?
222    return (elements instanceof Collection)
223        ? copyOf((Collection<? extends E>) elements)
224        : copyOf(elements.iterator());
225  }

227  /**
228   * Returns an immutable list containing the given elements, in order.
229   *
230   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
231   * safe to do so. The exact circumstances under which a copy will or will not be performed are
232   * undocumented and subject to change.
233   *
234   * <p>Note that if {@code list} is a {@code List<String>}, then {@code ImmutableList.copyOf(list)}
235   * returns an {@code ImmutableList<String>} containing each of the strings in {@code list}, while
236   * {@code ImmutableList.of(list)} returns an {@code ImmutableList<List<String>>} containing one
237   * element (the given list itself).
238   *
239   * <p>This method is safe to use even when {@code elements} is a synchronized or concurrent
240   * collection that is currently being modified by another thread.
241   *
242   * @throws NullPointerException if {@code elements} contains a null element
243   */
244  public static <E> ImmutableList<E> copyOf(Collection<? extends E> elements) {
245    if (elements instanceof ImmutableCollection) {
246      @SuppressWarnings("unchecked") // all supported methods are covariant
247      ImmutableList<E> list = ((ImmutableCollection<E>) elements).asList();
248      return list.isPartialView() ? ImmutableList.<E>asImmutableList(list.toArray()) : list;
249    }
250    return construct(elements.toArray());
251  }

253  /**
254   * Returns an immutable list containing the given elements, in order.
255   *
256   * @throws NullPointerException if {@code elements} contains a null element
257   */
258  public static <E> ImmutableList<E> copyOf(Iterator<? extends E> elements) {
259    // We special-case for 0 or 1 elements, but going further is madness.
260    if (!elements.hasNext()) {
261      return of();
262    }
263    E first = elements.next();
264    if (!elements.hasNext()) {
265      return of(first);
266    } else {
267      return new ImmutableList.Builder<E>().add(first).addAll(elements).build();
268    }
269  }

283  /**
284   * Returns an immutable list containing the given elements, sorted according to their natural
285   * order. The sorting algorithm used is stable, so elements that compare as equal will stay in the
286   * order in which they appear in the input.
287   *
288   * <p>If your data has no duplicates, or you wish to deduplicate elements, use {@code
289   * ImmutableSortedSet.copyOf(elements)}; if you want a {@code List} you can use its {@code
290   * asList()} view.
291   *
292   * <p><b>Java 8 users:</b> If you want to convert a {@link java.util.stream.Stream} to a sorted
293   * {@code ImmutableList}, use {@code stream.sorted().collect(toImmutableList())}.
294   *
295   * @throws NullPointerException if any element in the input is null
296   * @since 21.0
297   */
298  public static <E extends Comparable<? super E>> ImmutableList<E> sortedCopyOf(
299      Iterable<? extends E> elements) {
300    Comparable<?>[] array = Iterables.toArray(elements, new Comparable<?>[0]);
301    checkElementsNotNull((Object[]) array);
302    Arrays.sort(array);
303    return asImmutableList(array);
304  }

306  /**
307   * Returns an immutable list containing the given elements, in sorted order relative to the
308   * specified comparator. The sorting algorithm used is stable, so elements that compare as equal
309   * will stay in the order in which they appear in the input.
310   *
311   * <p>If your data has no duplicates, or you wish to deduplicate elements, use {@code
312   * ImmutableSortedSet.copyOf(comparator, elements)}; if you want a {@code List} you can use its
313   * {@code asList()} view.
314   *
315   * <p><b>Java 8 users:</b> If you want to convert a {@link java.util.stream.Stream} to a sorted
316   * {@code ImmutableList}, use {@code stream.sorted(comparator).collect(toImmutableList())}.
317   *
318   * @throws NullPointerException if any element in the input is null
319   * @since 21.0
320   */
321  public static <E> ImmutableList<E> sortedCopyOf(
322      Comparator<? super E> comparator, Iterable<? extends E> elements) {
323    checkNotNull(comparator);
324    @SuppressWarnings("unchecked") // all supported methods are covariant
325    E[] array = (E[]) Iterables.toArray(elements);
326    checkElementsNotNull(array);
327    Arrays.sort(array, comparator);
328    return asImmutableList(array);
329  }

413  /**
414   * Returns an immutable list of the elements between the specified {@code fromIndex}, inclusive,
415   * and {@code toIndex}, exclusive. (If {@code fromIndex} and {@code toIndex} are equal, the empty
416   * immutable list is returned.)
417   *
418   * <p><b>Note:</b> in almost all circumstances, the returned {@link ImmutableList} retains a
419   * strong reference to {@code this}, which may prevent the original list from being garbage
420   * collected. If you want the original list to be eligible for garbage collection, you should
421   * create and use a copy of the sub list (e.g., {@code
422   * ImmutableList.copyOf(originalList.subList(...))}).
423   */
424  @Override
425  public ImmutableList<E> subList(int fromIndex, int toIndex) {
426    checkPositionIndexes(fromIndex, toIndex, size());
427    int length = toIndex - fromIndex;
428    if (length == size()) {
429      return this;
430    } else if (length == 0) {
431      return of();
432    } else {
433      return subListUnchecked(fromIndex, toIndex);
434    }
435  }
436
437  /**
438   * Called by the default implementation of {@link #subList} when {@code toIndex - fromIndex > 1},
439   * after index validation has already been performed.
440   */
441  ImmutableList<E> subListUnchecked(int fromIndex, int toIndex) {
442    return new SubList(fromIndex, toIndex - fromIndex);
443  }

562  @Override
563  int copyIntoArray(@Nullable Object[] dst, int offset) {
564    // this loop is faster for RandomAccess instances, which ImmutableLists are
565    int size = size();
566    for (int i = 0; i < size; i++) {
567      dst[offset + i] = get(i);
568    }
569    return offset + size;
570  }

573   * Returns a view of this immutable list in reverse order. For example, {@code ImmutableList.of(1,
574   * 2, 3).reverse()} is equivalent to {@code ImmutableList.of(3, 2, 1)}.

576   * @return a view of this immutable list in reverse order

580    return (size() <= 1) ? this : new ReverseImmutableList<E>(this);

583  private static class ReverseImmutableList<E> extends ImmutableList<E> {
584    private final transient ImmutableList<E> forwardList;
585
586    ReverseImmutableList(ImmutableList<E> backingList) {
587      this.forwardList = backingList;
588    }
589
590    private int reverseIndex(int index) {
591      return (size() - 1) - index;
592    }
593
594    private int reversePosition(int index) {
595      return size() - index;
596    }
597
598    @Override
599    public ImmutableList<E> reverse() {
600      return forwardList;
601    }
602
603    @Override
604    public boolean contains(@CheckForNull Object object) {
605      return forwardList.contains(object);
606    }
607
608    @Override
609    public int indexOf(@CheckForNull Object object) {
610      int index = forwardList.lastIndexOf(object);
611      return (index >= 0) ? reverseIndex(index) : -1;
612    }
613
614    @Override
615    public int lastIndexOf(@CheckForNull Object object) {
616      int index = forwardList.indexOf(object);
617      return (index >= 0) ? reverseIndex(index) : -1;
618    }
619
620    @Override
621    public ImmutableList<E> subList(int fromIndex, int toIndex) {
622      checkPositionIndexes(fromIndex, toIndex, size());
623      return forwardList.subList(reversePosition(toIndex), reversePosition(fromIndex)).reverse();
624    }
625
626    @Override
627    public E get(int index) {
628      checkElementIndex(index, size());
629      return forwardList.get(reverseIndex(index));
630    }
631
632    @Override
633    public int size() {
634      return forwardList.size();
635    }
636
637    @Override
638    boolean isPartialView() {
639      return forwardList.isPartialView();
640    }
641  }

648  @Override
649  public int hashCode() {
650    int hashCode = 1;
651    int n = size();
652    for (int i = 0; i < n; i++) {
653      hashCode = 31 * hashCode + get(i).hashCode();
654
655      hashCode = ~~hashCode;
656      // needed to deal with GWT integer overflow
657    }
658    return hashCode;
659  }

696    return new Builder<E>();
697  }

699  /**
700   * Returns a new builder, expecting the specified number of elements to be added.
701   *
702   * <p>If {@code expectedSize} is exactly the number of elements added to the builder before {@link
703   * Builder#build} is called, the builder is likely to perform better than an unsized {@link
704   * #builder()} would have.
705   *
706   * <p>It is not specified if any performance benefits apply if {@code expectedSize} is close to,
707   * but not exactly, the number of elements added to the builder.
708   *
709   * @since 23.1
710   */
711  public static <E> Builder<E> builderWithExpectedSize(int expectedSize) {
712    checkNonnegative(expectedSize, "expectedSize");
713    return new ImmutableList.Builder<E>(expectedSize);
714  }

716  /**
717   * A builder for creating immutable list instances, especially {@code public static final} lists
718   * ("constant lists"). Example:
719   *
720   * <pre>{@code
721   * public static final ImmutableList<Color> GOOGLE_COLORS
722   *     = new ImmutableList.Builder<Color>()
723   *         .addAll(WEBSAFE_COLORS)
724   *         .add(new Color(0, 191, 255))
725   *         .build();
726   * }</pre>
727   *
728   * <p>Elements appear in the resulting list in the same order they were added to the builder.
729   *
730   * <p>Builder instances can be reused; it is safe to call {@link #build} multiple times to build
731   * multiple lists in series. Each new list contains all the elements of the ones created before
732   * it.
733   *
734   * @since 2.0
735   */
736  public static final class Builder<E> extends ImmutableCollection.ArrayBasedBuilder<E> {
737    /**
738     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
739     * ImmutableList#builder}.
740     */
741    public Builder() {
742      this(DEFAULT_INITIAL_CAPACITY);
743    }
744
745    Builder(int capacity) {
746      super(capacity);
747    }
748
749    /**
750     * Adds {@code element} to the {@code ImmutableList}.
751     *
752     * @param element the element to add
753     * @return this {@code Builder} object
754     * @throws NullPointerException if {@code element} is null
755     */
756    @CanIgnoreReturnValue
757    @Override
758    public Builder<E> add(E element) {
759      super.add(element);
760      return this;
761    }
762
763    /**
764     * Adds each element of {@code elements} to the {@code ImmutableList}.
765     *
766     * @param elements the {@code Iterable} to add to the {@code ImmutableList}
767     * @return this {@code Builder} object
768     * @throws NullPointerException if {@code elements} is null or contains a null element
769     */
770    @CanIgnoreReturnValue
771    @Override
772    public Builder<E> add(E... elements) {
773      super.add(elements);
774      return this;
775    }
776
777    /**
778     * Adds each element of {@code elements} to the {@code ImmutableList}.
779     *
780     * @param elements the {@code Iterable} to add to the {@code ImmutableList}
781     * @return this {@code Builder} object
782     * @throws NullPointerException if {@code elements} is null or contains a null element
783     */
784    @CanIgnoreReturnValue
785    @Override
786    public Builder<E> addAll(Iterable<? extends E> elements) {
787      super.addAll(elements);
788      return this;
789    }
790
791    /**
792     * Adds each element of {@code elements} to the {@code ImmutableList}.
793     *
794     * @param elements the {@code Iterator} to add to the {@code ImmutableList}
795     * @return this {@code Builder} object
796     * @throws NullPointerException if {@code elements} is null or contains a null element
797     */
798    @CanIgnoreReturnValue
799    @Override
800    public Builder<E> addAll(Iterator<? extends E> elements) {
801      super.addAll(elements);
802      return this;
803    }
804
805    @CanIgnoreReturnValue
806    Builder<E> combine(Builder<E> other) {
807      addAll(other.contents, other.size);
808      return this;
809    }
810
811    /**
812     * Returns a newly-created {@code ImmutableList} based on the contents of the {@code Builder}.
813     */
814    @Override
815    public ImmutableList<E> build() {
816      forceCopy = true;
817      return asImmutableList(contents, size);
818    }
819  }
