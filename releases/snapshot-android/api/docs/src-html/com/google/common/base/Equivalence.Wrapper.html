015package com.google.common.base;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018
019import com.google.common.annotations.GwtCompatible;
020import com.google.errorprone.annotations.ForOverride;
021import java.io.Serializable;
022import javax.annotation.CheckForNull;
023import org.checkerframework.checker.nullness.qual.Nullable;
024
025/**
026 * A strategy for determining whether two instances are considered equivalent, and for computing
027 * hash codes in a manner consistent with that equivalence. Two examples of equivalences are the
028 * {@linkplain #identity() identity equivalence} and the {@linkplain #equals "equals" equivalence}.
029 *
030 * <p><b>For users targeting Android API level 24 or higher:</b> This class will eventually
031 * implement {@code BiPredicate<T, T>} (as it does in the main Guava artifact), but we currently
032 * target a lower API level. In the meantime, if you have support for method references you can use
033 * an equivalence as a bi-predicate like this: {@code myEquivalence::equivalent}.
034 *
035 * @author Bob Lee
036 * @author Ben Yu
037 * @author Gregory Kick
038 * @since 10.0 (<a href="https://github.com/google/guava/wiki/Compatibility">mostly
039 *     source-compatible</a> since 4.0)
040 */
041@GwtCompatible
042@ElementTypesAreNonnullByDefault
043/*
044 * The type parameter is <T> rather than <T extends @Nullable> so that we can use T in the
045 * doEquivalent and doHash methods to indicate that the parameter cannot be null.
046 */
047public abstract class Equivalence<T> {
048  /** Constructor for use by subclasses. */
049  protected Equivalence() {}
050
051  /**
052   * Returns {@code true} if the given objects are considered equivalent.
053   *
054   * <p>This method describes an <i>equivalence relation</i> on object references, meaning that for
055   * all references {@code x}, {@code y}, and {@code z} (any of which may be null):
056   *
057   * <ul>
058   *   <li>{@code equivalent(x, x)} is true (<i>reflexive</i> property)
059   *   <li>{@code equivalent(x, y)} and {@code equivalent(y, x)} each return the same result
060   *       (<i>symmetric</i> property)
061   *   <li>If {@code equivalent(x, y)} and {@code equivalent(y, z)} are both true, then {@code
062   *       equivalent(x, z)} is also true (<i>transitive</i> property)
063   * </ul>
064   *
065   * <p>Note that all calls to {@code equivalent(x, y)} are expected to return the same result as
066   * long as neither {@code x} nor {@code y} is modified.
067   */
068  public final boolean equivalent(@CheckForNull T a, @CheckForNull T b) {
069    if (a == b) {
070      return true;
071    }
072    if (a == null || b == null) {
073      return false;
074    }
075    return doEquivalent(a, b);
076  }
077
078  /**
079   *
080   * @since 10.0 (previously, subclasses would override equivalent())
081   */
082  @ForOverride
083  protected abstract boolean doEquivalent(T a, T b);
084
085  /**
086   * Returns a hash code for {@code t}.
087   *
088   * <p>The {@code hash} has the following properties:
089   *
090   * <ul>
091   *   <li>It is <i>consistent</i>: for any reference {@code x}, multiple invocations of {@code
092   *       hash(x}} consistently return the same value provided {@code x} remains unchanged
093   *       according to the definition of the equivalence. The hash need not remain consistent from
094   *       one execution of an application to another execution of the same application.
095   *   <li>It is <i>distributable across equivalence</i>: for any references {@code x} and {@code
096   *       y}, if {@code equivalent(x, y)}, then {@code hash(x) == hash(y)}. It is <i>not</i>
097   *       necessary that the hash be distributable across <i>inequivalence</i>. If {@code
098   *       equivalence(x, y)} is false, {@code hash(x) == hash(y)} may still be true.
099   *   <li>{@code hash(null)} is {@code 0}.
100   * </ul>
101   */
102  public final int hash(@CheckForNull T t) {
103    if (t == null) {
104      return 0;
105    }
106    return doHash(t);
107  }
108
109  /**
110   * Implemented by the user to return a hash code for {@code t}, subject to the requirements
111   * specified in {@link #hash}.
112   *
113   * <p>This method should not be called except by {@link #hash}. When {@link #hash} calls this
114   * method, {@code t} is guaranteed to be non-null.
115   *
116   * @since 10.0 (previously, subclasses would override hash())
117   */
118  @ForOverride
119  protected abstract int doHash(T t);
120
121  /**
122   * Returns a new equivalence relation for {@code F} which evaluates equivalence by first applying
123   * {@code function} to the argument, then evaluating using {@code this}. That is, for any pair of
124   * non-null objects {@code x} and {@code y}, {@code equivalence.onResultOf(function).equivalent(a,
125   * b)} is true if and only if {@code equivalence.equivalent(function.apply(a), function.apply(b))}
126   * is true.
127   *
128   * <p>For example:
129   *
130   * <pre>{@code
131   * Equivalence<Person> SAME_AGE = Equivalence.equals().onResultOf(GET_PERSON_AGE);
132   * }</pre>
133   *
134   * <p>{@code function} will never be invoked with a null value.
135   *
136   * <p>Note that {@code function} must be consistent according to {@code this} equivalence
137   * relation. That is, invoking {@link Function#apply} multiple times for a given value must return
138   * equivalent results. For example, {@code
139   * Equivalence.identity().onResultOf(Functions.toStringFunction())} is broken because it's not
140   * guaranteed that {@link Object#toString}) always returns the same string instance.
141   *
142   * @since 10.0
143   */
144  public final <F> Equivalence<F> onResultOf(Function<? super F, ? extends @Nullable T> function) {
145    return new FunctionalEquivalence<>(function, this);
146  }
147
148  /**
149   * Returns a wrapper of {@code reference} that implements {@link Wrapper#equals(Object)
150   * Object.equals()} such that {@code wrap(a).equals(wrap(b))} if and only if {@code equivalent(a,
151   * b)}.
152   *
153   * <p>The returned object is serializable if both this {@code Equivalence} and {@code reference}
154   * are serializable (including when {@code reference} is null).
155   *
156   * @since 10.0
157   */
158  public final <S extends @Nullable T> Wrapper<S> wrap(@ParametricNullness S reference) {
159    @SuppressWarnings({"nullness", "unchecked"})
160    Wrapper<S> w = (Wrapper<S>) new Wrapper<T>(this, reference);
161    return w;
162  }
163
164  /**
165   * Wraps an object so that {@link #equals(Object)} and {@link #hashCode()} delegate to an {@link
166   * Equivalence}.
167   *
168   * <p>For example, given an {@link Equivalence} for {@link String strings} named {@code equiv}
169   * that tests equivalence using their lengths:
170   *
171   * <pre>{@code
172   * equiv.wrap("a").equals(equiv.wrap("b")) // true
173   * equiv.wrap("a").equals(equiv.wrap("hello")) // false
174   * }</pre>
175   *
176   * <p>Note in particular that an equivalence wrapper is never equal to the object it wraps.
177   *
178   * <pre>{@code
179   * equiv.wrap(obj).equals(obj) // always false
180   * }</pre>
181   *
182   * @since 10.0
183   */
184  public static final class Wrapper<T extends @Nullable Object> implements Serializable {
185    /*
186     * Equivalence's type argument is always non-nullable: Equivalence<Number>, never
187     * Equivalence<@Nullable Number>. That can still produce wrappers of various types --
188     * Wrapper<Number>, Wrapper<Integer>, Wrapper<@Nullable Integer>, etc. If we used just
189     * Equivalence<? super T> below, no type could satisfy both that bound and T's own
190     * bound.
191     */
192    private final Equivalence<?> equivalence;
193
194    @ParametricNullness private final T reference;
195
196    private Wrapper(Equivalence<?> equivalence, @ParametricNullness T reference) {
197      this.equivalence = checkNotNull(equivalence);
198      this.reference = reference;
199    }
200
201    /** Returns the (possibly null) reference wrapped by this instance. */
202    @ParametricNullness
203    public T get() {
204      return reference;
205    }
206
207    /**
208     * Returns {@code true} if {@link Equivalence#equivalent(Object, Object)} applied to the wrapped
209     * references is {@code true} and both wrappers use the {@link Object#equals(Object) same}
210     * equivalence.
211     */
212    @Override
213    public boolean equals(@CheckForNull Object obj) {
214      if (obj == this) {
215        return true;
216      }
217      if (obj instanceof Wrapper) {
218        Wrapper<?> that = (Wrapper<?>) obj; // note: not necessarily a Wrapper<T>
219
220        if (this.equivalence.equals(that.equivalence)) {
221          /*
222           * We'll accept that as sufficient "proof" that either equivalence should be able to
223           * handle either reference, so it's safe to circumvent compile-time type checking.
224           */
225          @SuppressWarnings("unchecked")
226          Equivalence<Object> equivalence = (Equivalence<Object>) this.equivalence;
227          return equivalence.equivalent(this.reference, that.reference);
228        }
229      }
230      return false;
231    }
232
233    /** Returns the result of {@link Equivalence#hash(Object)} applied to the wrapped reference. */
234    @Override
235    public int hashCode() {
236      @SuppressWarnings("unchecked") // This is really Equivalence<T & @NonNull Object>
237      Equivalence<Object> equivalence = (Equivalence<Object>) this.equivalence;
238      return equivalence.hash(reference);
239    }
240
241    /**
242     * Returns a string representation for this equivalence wrapper. The form of this string
243     * representation is not specified.
244     */
245    @Override
246    public String toString() {
247      return equivalence + ".wrap(" + reference + ")";
248    }
249
250    private static final long serialVersionUID = 0;
251  }
252
253  /**
254   * Returns an equivalence over iterables based on the equivalence of their elements. More
255   * specifically, two iterables are considered equivalent if they both contain the same number of
256   * elements, and each pair of corresponding elements is equivalent according to {@code this}. Null
257   * iterables are equivalent to one another.
258   *
259   * <p>Note that this method performs a similar function for equivalences as {@link
260   * com.google.common.collect.Ordering#lexicographical} does for orderings.
261   *
262   * <p>The returned object is serializable if this object is serializable.
263   *
264   * @since 10.0
265   */
266  @GwtCompatible(serializable = true)
267  public final <S extends @Nullable T> Equivalence<Iterable<S>> pairwise() {
268    // Ideally, the returned equivalence would support Iterable<? extends T>. However,
269    // the need for this is so rare that it's not worth making callers deal with the ugly wildcard.
270    return new PairwiseEquivalence<>(this);
271  }
272
273  /**
274   * Returns a predicate that evaluates to true if and only if the input is equivalent to {@code
275   * target} according to this equivalence relation.
276   *
277   * @since 10.0
278   */
279  public final Predicate<@Nullable T> equivalentTo(@CheckForNull T target) {
280    return new EquivalentToPredicate<T>(this, target);
281  }
282
283  private static final class EquivalentToPredicate<T>
284      implements Predicate<@Nullable T>, Serializable {
285
286    private final Equivalence<T> equivalence;
287    @CheckForNull private final T target;
288
289    EquivalentToPredicate(Equivalence<T> equivalence, @CheckForNull T target) {
290      this.equivalence = checkNotNull(equivalence);
291      this.target = target;
292    }
293
294    @Override
295    public boolean apply(@CheckForNull T input) {
296      return equivalence.equivalent(input, target);
297    }
298
299    @Override
300    public boolean equals(@CheckForNull Object obj) {
301      if (this == obj) {
302        return true;
303      }
304      if (obj instanceof EquivalentToPredicate) {
305        EquivalentToPredicate<?> that = (EquivalentToPredicate<?>) obj;
306        return equivalence.equals(that.equivalence) && Objects.equal(target, that.target);
307      }
308      return false;
309    }
310
311    @Override
312    public int hashCode() {
313      return Objects.hashCode(equivalence, target);
314    }
315
316    @Override
317    public String toString() {
318      return equivalence + ".equivalentTo(" + target + ")";
319    }
320
321    private static final long serialVersionUID = 0;
322  }
323
324  /**
325   * Returns an equivalence that delegates to {@link Object#equals} and {@link Object#hashCode}.
326   * {@link Equivalence#equivalent} returns {@code true} if both values are null, or if neither
327   * value is null and {@link Object#equals} returns {@code true}. {@link Equivalence#hash} returns
328   * {@code 0} if passed a null value.
329   *
330   * @since 13.0
331   * @since 8.0 (in Equivalences with null-friendly behavior)
332   * @since 4.0 (in Equivalences)
333   */
334  public static Equivalence<Object> equals() {
335    return Equals.INSTANCE;
336  }
337
338  /**
339   * Returns an equivalence that uses {@code ==} to compare values and {@link
340   * System#identityHashCode(Object)} to compute the hash code. {@link Equivalence#equivalent}
341   * returns {@code true} if {@code a == b}, including in the case that a and b are both null.
342   *
343   * @since 13.0
344   * @since 4.0 (in Equivalences)
345   */
346  public static Equivalence<Object> identity() {
347    return Identity.INSTANCE;
348  }
349
350  static final class Equals extends Equivalence<Object> implements Serializable {
351
352    static final Equals INSTANCE = new Equals();
353
354    @Override
355    protected boolean doEquivalent(Object a, Object b) {
356      return a.equals(b);
357    }
358
359    @Override
360    protected int doHash(Object o) {
361      return o.hashCode();
362    }
363
364    private Object readResolve() {
365      return INSTANCE;
366    }
367
368    private static final long serialVersionUID = 1;
369  }
370
371  static final class Identity extends Equivalence<Object> implements Serializable {
372
373    static final Identity INSTANCE = new Identity();
374
375    @Override
376    protected boolean doEquivalent(Object a, Object b) {
377      return false;
378    }
379
380    @Override
381    protected int doHash(Object o) {
382      return System.identityHashCode(o);
383    }
384
385    private Object readResolve() {
386      return INSTANCE;
387    }
388
389    private static final long serialVersionUID = 1;
390  }
