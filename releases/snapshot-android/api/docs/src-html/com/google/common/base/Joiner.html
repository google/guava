032/**
033 * An object which joins pieces of text (specified as an array, {@link Iterable}, varargs or even a
034 * {@link Map}) with a separator. It either appends the results to an {@link Appendable} or returns
035 * them as a {@link String}. Example:
036 *
037 * <pre>{@code
038 * Joiner joiner = Joiner.on("; ").skipNulls();
039 *  . . .
040 * return joiner.join("Harry", null, "Ron", "Hermione");
041 * }</pre>
042 *
043 * <p>This returns the string {@code "Harry; Ron; Hermione"}. Note that all input elements are
044 * converted to strings using {@link Object#toString()} before being appended.
045 *
046 * <p>If neither {@link #skipNulls()} nor {@link #useForNull(String)} is specified, the joining
047 * methods will throw {@link NullPointerException} if any given element is null.
048 *
049 * <p><b>Warning: joiner instances are always immutable</b>; a configuration method such as {@code
050 * useForNull} has no effect on the instance it is invoked on! You must store and use the new joiner
051 * instance returned by the method. This makes joiners thread-safe, and safe to store as {@code
052 * static final} constants.
053 *
054 * <pre>{@code
055 * // Bad! Do not do this!
056 * Joiner joiner = Joiner.on(',');
057 * joiner.skipNulls(); // does nothing!
058 * return joiner.join("wrong", null, "wrong");
059 * }</pre>
060 *
061 * <p>See the Guava User Guide article on <a
062 * href="https://github.com/google/guava/wiki/StringsExplained#joiner">{@code Joiner}</a>.
063 *
064 * @author Kevin Bourrillion
065 * @since 2.0
066 */
067@GwtCompatible
068public class Joiner {

105  /**
106   * Appends the string representation of each of {@code parts}, using the previously configured
107   * separator between each, to {@code appendable}.
108   *
109   * @since 11.0
110   */
111  @CanIgnoreReturnValue
112  public <A extends Appendable> A appendTo(A appendable, Iterator<? extends @Nullable Object> parts)
113      throws IOException {
114    checkNotNull(appendable);
115    if (parts.hasNext()) {
116      appendable.append(toString(parts.next()));
117      while (parts.hasNext()) {
118        appendable.append(separator);
119        appendable.append(toString(parts.next()));
120      }
121    }
122    return appendable;
123  }

250  /**
251   * Returns a joiner with the same behavior as this one, except automatically substituting {@code
252   * nullText} for any provided null elements.
253   */
254  public Joiner useForNull(String nullText) {
255    checkNotNull(nullText);
256    return new Joiner(this) {
257      @Override
258      CharSequence toString(@CheckForNull Object part) {
259        return (part == null) ? nullText : Joiner.this.toString(part);
260      }
261
262      @Override
263      public Joiner useForNull(String nullText) {
264        throw new UnsupportedOperationException("already specified useForNull");
265      }
266
267      @Override
268      public Joiner skipNulls() {
269        throw new UnsupportedOperationException("already specified useForNull");
270      }
271    };
272  }

274  /**
275   * Returns a joiner with the same behavior as this joiner, except automatically skipping over any
276   * provided null elements.
277   */
278  public Joiner skipNulls() {
279    return new Joiner(this) {
280      @Override
281      @SuppressWarnings("JoinIterableIterator") // suggests infinite recursion
282      public String join(Iterable<? extends @Nullable Object> parts) {
283        return join(parts.iterator());
284      }
285
286      @Override
287      public <A extends Appendable> A appendTo(
288          A appendable, Iterator<? extends @Nullable Object> parts) throws IOException {
289        checkNotNull(appendable, "appendable");
290        checkNotNull(parts, "parts");
291        while (parts.hasNext()) {
292          Object part = parts.next();
293          if (part != null) {
294            appendable.append(Joiner.this.toString(part));
295            break;
296          }
297        }
298        while (parts.hasNext()) {
299          Object part = parts.next();
300          if (part != null) {
301            appendable.append(separator);
302            appendable.append(Joiner.this.toString(part));
303          }
304        }
305        return appendable;
306      }
307
308      @Override
309      public Joiner useForNull(String nullText) {
310        throw new UnsupportedOperationException("already specified skipNulls");
311      }
312
313      @Override
314      public MapJoiner withKeyValueSeparator(String kvs) {
315        throw new UnsupportedOperationException("can't use .skipNulls() with maps");
316      }
317    };
318  }

338  /**
339   * An object that joins map entries in the same manner as {@code Joiner} joins iterables and
340   * arrays. Like {@code Joiner}, it is thread-safe and immutable.
341   *
342   * <p>In addition to operating on {@code Map} instances, {@code MapJoiner} can operate on {@code
343   * Multimap} entries in two distinct modes:
344   *
345   * <ul>
346   *   <li>To output a separate entry for each key-value pair, pass {@code multimap.entries()} to a
347   *       {@code MapJoiner} method that accepts entries as input, and receive output of the form
348   *       {@code key1=A&key1=B&key2=C}.
349   *   <li>To output a single entry for each key, pass {@code multimap.asMap()} to a {@code
350   *       MapJoiner} method that accepts a map as input, and receive output of the form {@code
351   *       key1=[A, B]&key2=C}.
352   * </ul>
353   *
354   * @since 2.0
355   */
356  public static final class MapJoiner {
357    private final Joiner joiner;
358    private final String keyValueSeparator;
359
360    private MapJoiner(Joiner joiner, String keyValueSeparator) {
361      this.joiner = joiner; // only "this" is ever passed, so don't checkNotNull
362      this.keyValueSeparator = checkNotNull(keyValueSeparator);
363    }
364
365    /**
366     * Appends the string representation of each entry of {@code map}, using the previously
367     * configured separator and key-value separator, to {@code appendable}.
368     */
369    @CanIgnoreReturnValue
370    public <A extends Appendable> A appendTo(A appendable, Map<?, ?> map) throws IOException {
371      return appendTo(appendable, map.entrySet());
372    }
373
374    /**
375     * Appends the string representation of each entry of {@code map}, using the previously
376     * configured separator and key-value separator, to {@code builder}. Identical to {@link
377     * #appendTo(Appendable, Map)}, except that it does not throw {@link IOException}.
378     */
379    @CanIgnoreReturnValue
380    public StringBuilder appendTo(StringBuilder builder, Map<?, ?> map) {
381      return appendTo(builder, map.entrySet());
382    }
383
384    /**
385     * Appends the string representation of each entry in {@code entries}, using the previously
386     * configured separator and key-value separator, to {@code appendable}.
387     *
388     * @since 10.0
389     */
390    @CanIgnoreReturnValue
391    public <A extends Appendable> A appendTo(A appendable, Iterable<? extends Entry<?, ?>> entries)
392        throws IOException {
393      return appendTo(appendable, entries.iterator());
394    }
395
396    /**
397     * Appends the string representation of each entry in {@code entries}, using the previously
398     * configured separator and key-value separator, to {@code appendable}.
399     *
400     * @since 11.0
401     */
402    @CanIgnoreReturnValue
403    public <A extends Appendable> A appendTo(A appendable, Iterator<? extends Entry<?, ?>> parts)
404        throws IOException {
405      checkNotNull(appendable);
406      if (parts.hasNext()) {
407        Entry<?, ?> entry = parts.next();
408        appendable.append(joiner.toString(entry.getKey()));
409        appendable.append(keyValueSeparator);
410        appendable.append(joiner.toString(entry.getValue()));
411        while (parts.hasNext()) {
412          appendable.append(joiner.separator);
413          Entry<?, ?> e = parts.next();
414          appendable.append(joiner.toString(e.getKey()));
415          appendable.append(keyValueSeparator);
416          appendable.append(joiner.toString(e.getValue()));
417        }
418      }
419      return appendable;
420    }
421
422    /**
423     * Appends the string representation of each entry in {@code entries}, using the previously
424     * configured separator and key-value separator, to {@code builder}. Identical to {@link
425     * #appendTo(Appendable, Iterable)}, except that it does not throw {@link IOException}.
426     *
427     * @since 10.0
428     */
429    @CanIgnoreReturnValue
430    public StringBuilder appendTo(StringBuilder builder, Iterable<? extends Entry<?, ?>> entries) {
431      return appendTo(builder, entries.iterator());
432    }
433
434    /**
435     * Appends the string representation of each entry in {@code entries}, using the previously
436     * configured separator and key-value separator, to {@code builder}. Identical to {@link
437     * #appendTo(Appendable, Iterable)}, except that it does not throw {@link IOException}.
438     *
439     * @since 11.0
440     */
441    @CanIgnoreReturnValue
442    public StringBuilder appendTo(StringBuilder builder, Iterator<? extends Entry<?, ?>> entries) {
443      try {
444        appendTo((Appendable) builder, entries);
445      } catch (IOException impossible) {
446        throw new AssertionError(impossible);
447      }
448      return builder;
449    }

488  // TODO(cpovirk): Rename to "toCharSequence."
489  CharSequence toString(@CheckForNull Object part) {
490    /*
491     * requireNonNull is not safe: Joiner.on(...).join(somethingThatContainsNull) will indeed throw.
492     * However, Joiner.on(...).useForNull(...).join(somethingThatContainsNull) *is* safe -- because
493     * it returns a subclass of Joiner that overrides this method to tolerate null inputs.
494     *
495     * Unfortunately, we don't distinguish between these two cases in our public API: Joiner.on(...)
496     * and Joiner.on(...).useForNull(...) both declare the same return type: plain Joiner. To ensure
497     * that users *can* pass null arguments to Joiner, we annotate it as if it always tolerates null
498     * inputs, rather than as if it never tolerates them.
499     *
500     * We rely on checkers to implement special cases to catch dangerous calls to join(), etc. based
501     * on what they know about the particular Joiner instances the calls are performed on.
502     *
503     * (In addition to useForNull, we also offer skipNulls. It, too, tolerates null inputs, but its
504     * tolerance is implemented differently: Its implementation avoids calling this toString(Object)
505     * method in the first place.)
506     */
507    requireNonNull(part);
508    return (part instanceof CharSequence) ? (CharSequence) part : part.toString();
509  }

511  private static Iterable<@Nullable Object> iterable(
512      @CheckForNull Object first, @CheckForNull Object second, @Nullable Object[] rest) {
513    checkNotNull(rest);
514    return new AbstractList<@Nullable Object>() {
515      @Override
516      public int size() {
517        return rest.length + 2;
518      }
519
520      @Override
521      @CheckForNull
522      public Object get(int index) {
523        switch (index) {
524          case 0:
525            return first;
526          case 1:
527            return second;
528          default:
529            return rest[index - 2];
530        }
531      }
532    };
533  }

535  // cloned from ImmutableCollection
536  private static int expandedCapacity(int oldCapacity, int minCapacity) {
537    if (minCapacity < 0) {
538      throw new IllegalArgumentException("cannot store more than Integer.MAX_VALUE elements");
539    } else if (minCapacity <= oldCapacity) {
540      return oldCapacity;
541    }
542    // careful of overflow!
543    int newCapacity = oldCapacity + (oldCapacity >> 1) + 1;
544    if (newCapacity < minCapacity) {
545      newCapacity = Integer.highestOneBit(minCapacity - 1) << 1;
546    }
547    if (newCapacity < 0) {
548      newCapacity = Integer.MAX_VALUE;
549      // guaranteed to be >= newCapacity
550    }
551    return newCapacity;
552  }
