015package com.google.common.base;
016
017import com.google.common.annotations.GwtIncompatible;
018import com.google.common.annotations.J2ktIncompatible;
019import com.google.errorprone.annotations.DoNotMock;
020
021/**
022 * Implemented by references that have code to run after garbage collection of their referents.
023 *
024 * @see FinalizableReferenceQueue
025 * @author Bob Lee
026 * @since 2.0
027 */
028@DoNotMock("Use an instance of one of the Finalizable*Reference classes")
029@J2ktIncompatible
030@GwtIncompatible
031public interface FinalizableReference {
032  /**
033   * Invoked on a background thread after the referent has been garbage collected unless security
034   * restrictions prevented starting a background thread, in which case this method is invoked when
035   * new references are created.
036   */
037  void finalizeReferent();
038}
