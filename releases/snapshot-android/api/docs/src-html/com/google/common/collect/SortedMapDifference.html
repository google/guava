017package com.google.common.collect;
018
019import com.google.common.annotations.GwtCompatible;
020import java.util.SortedMap;
021import org.jspecify.annotations.Nullable;
022
023/**
024 * An object representing the differences between two sorted maps.
025 *
026 * @author Louis Wasserman
027 * @since 8.0
028 */
029@GwtCompatible
030public interface SortedMapDifference<K extends @Nullable Object, V extends @Nullable Object>
031    extends MapDifference<K, V> {
032
033  @Override
034  SortedMap<K, V> entriesOnlyOnLeft();
035
036  @Override
037  SortedMap<K, V> entriesOnlyOnRight();
038
039  @Override
040  SortedMap<K, V> entriesInCommon();
041
042  @Override
043  SortedMap<K, ValueDifference<V>> entriesDiffering();
044}
