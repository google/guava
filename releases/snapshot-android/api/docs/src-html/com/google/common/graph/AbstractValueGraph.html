017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkArgument;
020import static com.google.common.graph.GraphConstants.NODE_NOT_IN_GRAPH;
021
022import com.google.common.annotations.Beta;
023import com.google.common.base.Function;
024import com.google.common.collect.Maps;
025import java.util.Map;
026import java.util.Set;
027import javax.annotation.Nullable;
028
029/**
030 * This class provides a skeletal implementation of {@link ValueGraph}. It is recommended to extend
031 * this class rather than implement {@link ValueGraph} directly.
032 *
033 * <p>The methods implemented in this class should not be overridden unless the subclass admits a
034 * more efficient implementation.
035 *
036 * @author James Sexton
037 * @param <N> Node parameter type
038 * @param <V> Value parameter type
039 * @since 20.0
040 */
041@Beta
042public abstract class AbstractValueGraph<N, V> extends AbstractBaseGraph<N>
043    implements ValueGraph<N, V> {
044
045  @Override
046  public Graph<N> asGraph() {
047    return new AbstractGraph<N>() {
048      @Override
049      public Set<N> nodes() {
050        return AbstractValueGraph.this.nodes();
051      }
052
053      @Override
054      public Set<EndpointPair<N>> edges() {
055        return AbstractValueGraph.this.edges();
056      }
057
058      @Override
059      public boolean isDirected() {
060        return AbstractValueGraph.this.isDirected();
061      }
062
063      @Override
064      public boolean allowsSelfLoops() {
065        return AbstractValueGraph.this.allowsSelfLoops();
066      }
067
068      @Override
069      public ElementOrder<N> nodeOrder() {
070        return AbstractValueGraph.this.nodeOrder();
071      }
072
073      @Override
074      public Set<N> adjacentNodes(N node) {
075        return AbstractValueGraph.this.adjacentNodes(node);
076      }
077
078      @Override
079      public Set<N> predecessors(N node) {
080        return AbstractValueGraph.this.predecessors(node);
081      }
082
083      @Override
084      public Set<N> successors(N node) {
085        return AbstractValueGraph.this.successors(node);
086      }
087
088      @Override
089      public int degree(N node) {
090        return AbstractValueGraph.this.degree(node);
091      }
092
093      @Override
094      public int inDegree(N node) {
095        return AbstractValueGraph.this.inDegree(node);
096      }
097
098      @Override
099      public int outDegree(N node) {
100        return AbstractValueGraph.this.outDegree(node);
101      }
102    };
103  }
104
105  @Override
106  public V edgeValue(N nodeU, N nodeV) {
107    V value = edgeValueOrDefault(nodeU, nodeV, null);
108    if (value == null) {
109      checkArgument(nodes().contains(nodeU), NODE_NOT_IN_GRAPH, nodeU);
110      checkArgument(nodes().contains(nodeV), NODE_NOT_IN_GRAPH, nodeV);
111      throw new IllegalArgumentException(
112          "Edge connecting " + nodeU + " to " + nodeV + " is not present in this graph.");
113    }
114    return value;
115  }
116
117  @Override
118  public final boolean equals(@Nullable Object obj) {
119    if (obj == this) {
120      return true;
121    }
122    if (!(obj instanceof ValueGraph)) {
123      return false;
124    }
125    ValueGraph<?, ?> other = (ValueGraph<?, ?>) obj;
126
127    return isDirected() == other.isDirected()
128        && nodes().equals(other.nodes())
129        && edgeValueMap(this).equals(edgeValueMap(other));
130  }
131
132  @Override
133  public final int hashCode() {
134    return edgeValueMap(this).hashCode();
135  }
136
137  /** Returns a string representation of this graph. */
138  @Override
139  public String toString() {
140    return "isDirected: "
141        + isDirected()
142        + ", allowsSelfLoops: "
143        + allowsSelfLoops()
144        + ", nodes: "
145        + nodes()
146        + ", edges: "
147        + edgeValueMap(this);
148  }
149
150  private static <N, V> Map<EndpointPair<N>, V> edgeValueMap(final ValueGraph<N, V> graph) {
151    Function<EndpointPair<N>, V> edgeToValueFn =
152        new Function<EndpointPair<N>, V>() {
153          @Override
154          public V apply(EndpointPair<N> edge) {
155            return graph.edgeValue(edge.nodeU(), edge.nodeV());
156          }
157        };
158    return Maps.asMap(graph.edges(), edgeToValueFn);
159  }
160}
