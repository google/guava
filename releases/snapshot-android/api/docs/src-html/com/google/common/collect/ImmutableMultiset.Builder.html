039/**
040 * A {@link Multiset} whose contents will never change, with many other important properties
041 * detailed at {@link ImmutableCollection}.
042 *
043 * <p><b>Grouped iteration.</b> In all current implementations, duplicate elements always appear
044 * consecutively when iterating. Elements iterate in order by the <i>first</i> appearance of that
045 * element when the multiset was created.
046 *
047 * <p>See the Guava User Guide article on <a href=
048 * "https://github.com/google/guava/wiki/ImmutableCollectionsExplained">immutable collections</a>.
049 *
050 * @author Jared Levy
051 * @author Louis Wasserman
052 * @since 2.0
053 */
054@GwtCompatible(serializable = true, emulated = true)
055@SuppressWarnings("serial") // we're overriding default serialization
056@ElementTypesAreNonnullByDefault
057public abstract class ImmutableMultiset<E> extends ImmutableMultisetGwtSerializationDependencies<E>
058    implements Multiset<E> {

146   * Returns an immutable multiset containing the given elements, in the "grouped iteration order"
147   * described in the class documentation.
148   *
149   * @throws NullPointerException if any of {@code elements} is null
150   */
151  public static <E> ImmutableMultiset<E> copyOf(Iterable<? extends E> elements) {
152    if (elements instanceof ImmutableMultiset) {
153      @SuppressWarnings("unchecked") // all supported methods are covariant
154      ImmutableMultiset<E> result = (ImmutableMultiset<E>) elements;
155      if (!result.isPartialView()) {
156        return result;
157      }
158    }
159    ImmutableMultiset.Builder<E> builder =
160        new ImmutableMultiset.Builder<E>(Multisets.inferDistinctElements(elements));
161    builder.addAll(elements);
162    return builder.build();
163  }

172    return new ImmutableMultiset.Builder<E>().addAll(elements).build();

175  private static <E> ImmutableMultiset<E> copyFromElements(E... elements) {
176    return new ImmutableMultiset.Builder<E>().add(elements).build();
177  }

179  static <E> ImmutableMultiset<E> copyFromEntries(
180      Collection<? extends Entry<? extends E>> entries) {
181    ImmutableMultiset.Builder<E> builder = new ImmutableMultiset.Builder<E>(entries.size());
182    for (Entry<? extends E> entry : entries) {
183      builder.addCopies(entry.getElement(), entry.getCount());
184    }
185    return builder.build();
186  }

190  @Override
191  public UnmodifiableIterator<E> iterator() {
192    final Iterator<Entry<E>> entryIterator = entrySet().iterator();
193    return new UnmodifiableIterator<E>() {
194      int remaining;
195      @CheckForNull E element;
196
197      @Override
198      public boolean hasNext() {
199        return (remaining > 0) || entryIterator.hasNext();
200      }
201
202      @Override
203      public E next() {
204        if (remaining <= 0) {
205          Entry<E> entry = entryIterator.next();
206          element = entry.getElement();
207          remaining = entry.getCount();
208        }
209        remaining--;
210        /*
211         * requireNonNull is safe because `remaining` starts at 0, forcing us to initialize
212         * `element` above. After that, we never clear it.
213         */
214        return requireNonNull(element);
215      }
216    };
217  }

415  /**
416   * A builder for creating immutable multiset instances, especially {@code public static final}
417   * multisets ("constant multisets"). Example:
418   *
419   * <pre>{@code
420   * public static final ImmutableMultiset<Bean> BEANS =
421   *     new ImmutableMultiset.Builder<Bean>()
422   *         .addCopies(Bean.COCOA, 4)
423   *         .addCopies(Bean.GARDEN, 6)
424   *         .addCopies(Bean.RED, 8)
425   *         .addCopies(Bean.BLACK_EYED, 10)
426   *         .build();
427   * }</pre>
428   *
429   * <p>Builder instances can be reused; it is safe to call {@link #build} multiple times to build
430   * multiple multisets in series.
431   *
432   * @since 2.0
433   */
434  public static class Builder<E> extends ImmutableCollection.Builder<E> {
435    /*
436     * `contents` is null only for instances of the subclass, ImmutableSortedMultiset.Builder. That
437     * subclass overrides all the methods that access it here. Thus, all the methods here can safely
438     * assume that this field is non-null.
439     */
440    @CheckForNull ObjectCountHashMap<E> contents;
441
442    /**
443     * If build() has been called on the current contents multiset, we need to copy it on any future
444     * modifications, or we'll modify the already-built ImmutableMultiset.
445     */
446    boolean buildInvoked = false;
447    /**
448     * In the event of a setCount(elem, 0) call, we may need to remove elements, which destroys the
449     * insertion order property of ObjectCountHashMap. In that event, we need to convert to a
450     * ObjectCountLinkedHashMap, but we need to know we did that so we can convert back.
451     */
452    boolean isLinkedHash = false;
453
454    /**
455     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
456     * ImmutableMultiset#builder}.
457     */
458    public Builder() {
459      this(4);
460    }
461
462    Builder(int estimatedDistinct) {
463      this.contents = ObjectCountHashMap.createWithExpectedSize(estimatedDistinct);
464    }
465
466    Builder(boolean forSubtype) {
467      // for ImmutableSortedMultiset not to allocate data structures not used there
468      this.contents = null;
469    }
470
471    /**
472     * Adds {@code element} to the {@code ImmutableMultiset}.
473     *
474     * @param element the element to add
475     * @return this {@code Builder} object
476     * @throws NullPointerException if {@code element} is null
477     */
478    @CanIgnoreReturnValue
479    @Override
480    public Builder<E> add(E element) {
481      return addCopies(element, 1);
482    }
483
484    /**
485     * Adds each element of {@code elements} to the {@code ImmutableMultiset}.
486     *
487     * @param elements the elements to add
488     * @return this {@code Builder} object
489     * @throws NullPointerException if {@code elements} is null or contains a null element
490     */
491    @CanIgnoreReturnValue
492    @Override
493    public Builder<E> add(E... elements) {
494      super.add(elements);
495      return this;
496    }
497
498    /**
499     * Adds a number of occurrences of an element to this {@code ImmutableMultiset}.
500     *
501     * @param element the element to add
502     * @param occurrences the number of occurrences of the element to add. May be zero, in which
503     *     case no change will be made.
504     * @return this {@code Builder} object
505     * @throws NullPointerException if {@code element} is null
506     * @throws IllegalArgumentException if {@code occurrences} is negative, or if this operation
507     *     would result in more than {@link Integer#MAX_VALUE} occurrences of the element
508     */
509    @CanIgnoreReturnValue
510    public Builder<E> addCopies(E element, int occurrences) {
511      requireNonNull(contents); // see the comment on the field
512      if (occurrences == 0) {
513        return this;
514      }
515      if (buildInvoked) {
516        contents = new ObjectCountHashMap<E>(contents);
517        isLinkedHash = false;
518      }
519      buildInvoked = false;
520      checkNotNull(element);
521      contents.put(element, occurrences + contents.get(element));
522      return this;
523    }
524
525    /**
526     * Adds or removes the necessary occurrences of an element such that the element attains the
527     * desired count.
528     *
529     * @param element the element to add or remove occurrences of
530     * @param count the desired count of the element in this multiset
531     * @return this {@code Builder} object
532     * @throws NullPointerException if {@code element} is null
533     * @throws IllegalArgumentException if {@code count} is negative
534     */
535    @CanIgnoreReturnValue
536    public Builder<E> setCount(E element, int count) {
537      requireNonNull(contents); // see the comment on the field
538      if (count == 0 && !isLinkedHash) {
539        contents = new ObjectCountLinkedHashMap<E>(contents);
540        isLinkedHash = true;
541        // to preserve insertion order through deletions, we have to switch to an actual linked
542        // implementation at least for now, but this should be a super rare case
543      } else if (buildInvoked) {
544        contents = new ObjectCountHashMap<E>(contents);
545        isLinkedHash = false;
546      }
547      buildInvoked = false;
548      checkNotNull(element);
549      if (count == 0) {
550        contents.remove(element);
551      } else {
552        contents.put(checkNotNull(element), count);
553      }
554      return this;
555    }
556
557    /**
558     * Adds each element of {@code elements} to the {@code ImmutableMultiset}.
559     *
560     * @param elements the {@code Iterable} to add to the {@code ImmutableMultiset}
561     * @return this {@code Builder} object
562     * @throws NullPointerException if {@code elements} is null or contains a null element
563     */
564    @CanIgnoreReturnValue
565    @Override
566    public Builder<E> addAll(Iterable<? extends E> elements) {
567      requireNonNull(contents); // see the comment on the field
568      if (elements instanceof Multiset) {
569        Multiset<? extends E> multiset = Multisets.cast(elements);
570        ObjectCountHashMap<? extends E> backingMap = tryGetMap(multiset);
571        if (backingMap != null) {
572          contents.ensureCapacity(Math.max(contents.size(), backingMap.size()));
573          for (int i = backingMap.firstIndex(); i >= 0; i = backingMap.nextIndex(i)) {
574            addCopies(backingMap.getKey(i), backingMap.getValue(i));
575          }
576        } else {
577          Set<? extends Entry<? extends E>> entries = multiset.entrySet();
578          contents.ensureCapacity(Math.max(contents.size(), entries.size())); // might overlap
579          for (Entry<? extends E> entry : multiset.entrySet()) {
580            addCopies(entry.getElement(), entry.getCount());
581          }
582        }
583      } else {
584        super.addAll(elements);
585      }
586      return this;
587    }
588
589    /**
590     * Adds each element of {@code elements} to the {@code ImmutableMultiset}.
591     *
592     * @param elements the elements to add to the {@code ImmutableMultiset}
593     * @return this {@code Builder} object
594     * @throws NullPointerException if {@code elements} is null or contains a null element
595     */
596    @CanIgnoreReturnValue
597    @Override
598    public Builder<E> addAll(Iterator<? extends E> elements) {
599      super.addAll(elements);
600      return this;
601    }
602
603    /**
604     * If the specified collection is backed by an ObjectCountHashMap, it will be much more
605     * efficient to iterate over it by index rather than an entry iterator, which will need to
606     * allocate an object for each entry, so we check for that.
607     */
608    @CheckForNull
609    static <T> ObjectCountHashMap<T> tryGetMap(Iterable<T> multiset) {
610      if (multiset instanceof RegularImmutableMultiset) {
611        return ((RegularImmutableMultiset<T>) multiset).contents;
612      } else if (multiset instanceof AbstractMapBasedMultiset) {
613        return ((AbstractMapBasedMultiset<T>) multiset).backingMap;
614      } else {
615        return null;
616      }
617    }
618
619    /**
620     * Returns a newly-created {@code ImmutableMultiset} based on the contents of the {@code
621     * Builder}.
622     */
623    @Override
624    public ImmutableMultiset<E> build() {
625      requireNonNull(contents); // see the comment on the field
626      if (contents.size() == 0) {
627        return of();
628      }
629      if (isLinkedHash) {
630        // we need ObjectCountHashMap-backed contents, with its keys and values array in direct
631        // insertion order
632        contents = new ObjectCountHashMap<E>(contents);
633        isLinkedHash = false;
634      }
635      buildInvoked = true;
636      // contents is now ObjectCountHashMap, but still guaranteed to be in insertion order!
637      return new RegularImmutableMultiset<E>(contents);
638    }
639  }
