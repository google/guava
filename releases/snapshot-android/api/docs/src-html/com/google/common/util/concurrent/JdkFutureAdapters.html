015package com.google.common.util.concurrent;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018import static com.google.common.util.concurrent.Uninterruptibles.getUninterruptibly;
019
020import com.google.common.annotations.Beta;
021import com.google.common.annotations.GwtIncompatible;
022import com.google.common.annotations.J2ktIncompatible;
023import java.util.concurrent.ExecutionException;
024import java.util.concurrent.Executor;
025import java.util.concurrent.Executors;
026import java.util.concurrent.Future;
027import java.util.concurrent.ThreadFactory;
028import java.util.concurrent.atomic.AtomicBoolean;
029import org.checkerframework.checker.nullness.qual.Nullable;
030
031/**
032 * Utilities necessary for working with libraries that supply plain {@link Future} instances. Note
033 * that, whenever possible, it is strongly preferred to modify those libraries to return {@code
034 * ListenableFuture} directly.
035 *
036 * <p>For interoperability between {@code ListenableFuture} and <b>{@code CompletableFuture}</b>,
037 * consider <a href="https://github.com/lukas-krecan/future-converter">Future Converter</a>.
038 *
039 * @author Sven Mawson
040 * @since 10.0 (replacing {@code Futures.makeListenable}, which existed in 1.0)
041 */
042@Beta
043@J2ktIncompatible
044@GwtIncompatible
045@ElementTypesAreNonnullByDefault
046public final class JdkFutureAdapters {
047  /**
048   * Assigns a thread to the given {@link Future} to provide {@link ListenableFuture} functionality.
049   *
050   * <p><b>Warning:</b> If the input future does not already implement {@code ListenableFuture}, the
051   * returned future will emulate {@link ListenableFuture#addListener} by taking a thread from an
052   * internal, unbounded pool at the first call to {@code addListener} and holding it until the
053   * future is {@linkplain Future#isDone() done}.
054   *
055   * <p>Prefer to create {@code ListenableFuture} instances with {@link SettableFuture}, {@link
056   * MoreExecutors#listeningDecorator( java.util.concurrent.ExecutorService)}, {@link
057   * ListenableFutureTask}, {@link AbstractFuture}, and other utilities over creating plain {@code
058   * Future} instances to be upgraded to {@code ListenableFuture} after the fact.
059   */
060  public static <V extends @Nullable Object> ListenableFuture<V> listenInPoolThread(
061      Future<V> future) {
062    if (future instanceof ListenableFuture) {
063      return (ListenableFuture<V>) future;
064    }
065    return new ListenableFutureAdapter<>(future);
066  }
067
068  /**
069   * Submits a blocking task for the given {@link Future} to provide {@link ListenableFuture}
070   * functionality.
071   *
072   * <p><b>Warning:</b> If the input future does not already implement {@code ListenableFuture}, the
073   * returned future will emulate {@link ListenableFuture#addListener} by submitting a task to the
074   * given executor at the first call to {@code addListener}. The task must be started by the
075   * executor promptly, or else the returned {@code ListenableFuture} may fail to work. The task's
076   * execution consists of blocking until the input future is {@linkplain Future#isDone() done}, so
077   * each call to this method may claim and hold a thread for an arbitrary length of time. Use of
078   * bounded executors or other executors that may fail to execute a task promptly may result in
079   * deadlocks.
080   *
081   * <p>Prefer to create {@code ListenableFuture} instances with {@link SettableFuture}, {@link
082   * MoreExecutors#listeningDecorator( java.util.concurrent.ExecutorService)}, {@link
083   * ListenableFutureTask}, {@link AbstractFuture}, and other utilities over creating plain {@code
084   * Future} instances to be upgraded to {@code ListenableFuture} after the fact.
085   *
086   * @since 12.0
087   */
088  public static <V extends @Nullable Object> ListenableFuture<V> listenInPoolThread(
089      Future<V> future, Executor executor) {
090    checkNotNull(executor);
091    if (future instanceof ListenableFuture) {
092      return (ListenableFuture<V>) future;
093    }
094    return new ListenableFutureAdapter<>(future, executor);
095  }
096
097  /**
098   * An adapter to turn a {@link Future} into a {@link ListenableFuture}. This will wait on the
099   * future to finish, and when it completes, run the listeners. This implementation will wait on
100   * the source future indefinitely, so if the source future never completes, the adapter will never
101   * complete either.
102   *
103   * <p>If the delegate future is interrupted or throws an unexpected unchecked exception, the
104   * listeners will not be invoked.
105   */
106  private static class ListenableFutureAdapter<V extends @Nullable Object>
107      extends ForwardingFuture<V> implements ListenableFuture<V> {
108
109    private static final ThreadFactory threadFactory =
110        new ThreadFactoryBuilder()
111            .setDaemon(true)
112            .setNameFormat("ListenableFutureAdapter-thread-%d")
113            .build();
114    private static final Executor defaultAdapterExecutor =
115        Executors.newCachedThreadPool(threadFactory);
116
117    private final Executor adapterExecutor;
118
119    // The execution list to hold our listeners.
120    private final ExecutionList executionList = new ExecutionList();
121
122    // This allows us to only start up a thread waiting on the delegate future when the first
123    // listener is added.
124    private final AtomicBoolean hasListeners = new AtomicBoolean(false);
125
126    // The delegate future.
127    private final Future<V> delegate;
128
129    ListenableFutureAdapter(Future<V> delegate) {
130      this(delegate, defaultAdapterExecutor);
131    }
132
133    ListenableFutureAdapter(Future<V> delegate, Executor adapterExecutor) {
134      this.delegate = checkNotNull(delegate);
135      this.adapterExecutor = checkNotNull(adapterExecutor);
136    }
137
138    @Override
139    protected Future<V> delegate() {
140      return delegate;
141    }
142
143    @Override
144    public void addListener(Runnable listener, Executor exec) {
145      executionList.add(listener, exec);
146
147      // When a listener is first added, we run a task that will wait for the delegate to finish,
148      // and when it is done will run the listeners.
149      if (hasListeners.compareAndSet(false, true)) {
150        if (delegate.isDone()) {
151          // If the delegate is already done, run the execution list immediately on the current
152          // thread.
153          executionList.execute();
154          return;
155        }
156
157        // TODO(lukes): handle RejectedExecutionException
158        adapterExecutor.execute(
159            () -> {
160              try {
161                /*
162                 * Threads from our private pool are never interrupted. Threads from a
163                 * user-supplied executor might be, but... what can we do? This is another reason
164                 * to return a proper ListenableFuture instead of using listenInPoolThread.
165                 */
166                getUninterruptibly(delegate);
167              } catch (ExecutionException | RuntimeException | Error e) {
168                // (including CancellationException)
169                // The task is presumably done, run the listeners.
170                // TODO(cpovirk): Do *something* in case of Error (and maybe RuntimeException)?
171              }
172              executionList.execute();
173            });
174      }
175    }
176  }
177
178  private JdkFutureAdapters() {}
179}
