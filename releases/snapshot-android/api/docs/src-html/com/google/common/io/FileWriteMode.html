015package com.google.common.io;
016
017import com.google.common.annotations.GwtIncompatible;
018
019/**
020 * Modes for opening a file for writing. The default when mode when none is specified is to truncate
021 * the file before writing.
022 *
023 * @author Colin Decker
024 */
025@GwtIncompatible
026@ElementTypesAreNonnullByDefault
027public enum FileWriteMode {
028  /** Specifies that writes to the opened file should append to the end of the file. */
029  APPEND
030}
