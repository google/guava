048/**
049 * Represents an <a href="http://en.wikipedia.org/wiki/Internet_media_type">Internet Media Type</a>
050 * (also known as a MIME Type or Content Type). This class also supports the concept of media ranges
051 * <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1">defined by HTTP/1.1</a>.
052 * As such, the {@code *} character is treated as a wildcard and is used to represent any acceptable
053 * type or subtype value. A media type may not have wildcard type with a declared subtype. The
054 * {@code *} character has no special meaning as part of a parameter. All values for type, subtype,
055 * parameter attributes or parameter values must be valid according to RFCs <a
056 * href="https://tools.ietf.org/html/rfc2045">2045</a> and <a
057 * href="https://tools.ietf.org/html/rfc2046">2046</a>.
058 *
059 * <p>All portions of the media type that are case-insensitive (type, subtype, parameter attributes)
060 * are normalized to lowercase. The value of the {@code charset} parameter is normalized to
061 * lowercase, but all others are left as-is.
062 *
063 * <p>Note that this specifically does <strong>not</strong> represent the value of the MIME {@code
064 * Content-Type} header and as such has no support for header-specific considerations such as line
065 * folding and comments.
066 *
067 * <p>For media types that take a charset the predefined constants default to UTF-8 and have a
068 * "_UTF_8" suffix. To get a version without a character set, use {@link #withoutParameters}.
069 *
070 * @since 12.0
071 * @author Gregory Kick
072 */
073@GwtCompatible
074@Immutable
075@ElementTypesAreNonnullByDefault
076public final class MediaType {
077  private static final String CHARSET_ATTRIBUTE = "charset";
078  private static final ImmutableListMultimap<String, String> UTF_8_CONSTANT_PARAMETERS =
079      ImmutableListMultimap.of(CHARSET_ATTRIBUTE, Ascii.toLowerCase(UTF_8.name()));

082  private static final CharMatcher TOKEN_MATCHER =

084          .and(javaIsoControl().negate())
085          .and(CharMatcher.isNot(' '))

106  private static final Map<MediaType, MediaType> KNOWN_TYPES = Maps.newHashMap();
107
108  private static MediaType createConstant(String type, String subtype) {
109    MediaType mediaType =
110        addKnownType(new MediaType(type, subtype, ImmutableListMultimap.<String, String>of()));
111    mediaType.parsedCharset = Optional.absent();
112    return mediaType;
113  }
114
115  private static MediaType createConstantUtf8(String type, String subtype) {
116    MediaType mediaType = addKnownType(new MediaType(type, subtype, UTF_8_CONSTANT_PARAMETERS));
117    mediaType.parsedCharset = Optional.of(UTF_8);
118    return mediaType;
119  }
120
121  private static MediaType addKnownType(MediaType mediaType) {
122    KNOWN_TYPES.put(mediaType, mediaType);
123    return mediaType;
124  }

819  /**
820   * Returns an optional charset for the value of the charset parameter if it is specified.
821   *
822   * @throws IllegalStateException if multiple charset values have been set for this media type
823   * @throws IllegalCharsetNameException if a charset value is present, but illegal
824   * @throws UnsupportedCharsetException if a charset value is present, but no support is available
825   *     in this instance of the Java virtual machine
826   */
827  public Optional<Charset> charset() {
828    // racy single-check idiom, this is safe because Optional is immutable.
829    Optional<Charset> local = parsedCharset;
830    if (local == null) {
831      String value = null;
832      local = Optional.absent();
833      for (String currentValue : parameters.get(CHARSET_ATTRIBUTE)) {
834        if (value == null) {
835          value = currentValue;
836          local = Optional.of(Charset.forName(value));
837        } else if (!value.equals(currentValue)) {
838          throw new IllegalStateException(
839              "Multiple charset values defined: " + value + ", " + currentValue);
840        }
841      }
842      parsedCharset = local;
843    }
844    return local;
845  }

864  /**
865   * <em>Replaces</em> all parameters with the given attribute with parameters using the given
866   * values. If there are no values, any existing parameters with the given attribute are removed.
867   *
868   * @throws IllegalArgumentException if either {@code attribute} or {@code values} is invalid
869   * @since 24.0
870   */
871  public MediaType withParameters(String attribute, Iterable<String> values) {
872    checkNotNull(attribute);
873    checkNotNull(values);
874    String normalizedAttribute = normalizeToken(attribute);
875    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
876    for (Entry<String, String> entry : parameters.entries()) {
877      String key = entry.getKey();
878      if (!normalizedAttribute.equals(key)) {
879        builder.put(key, entry.getValue());
880      }
881    }
882    for (String value : values) {
883      builder.put(normalizedAttribute, normalizeParameterValue(normalizedAttribute, value));
884    }
885    MediaType mediaType = new MediaType(type, subtype, builder.build());
886    // if the attribute isn't charset, we can just inherit the current parsedCharset
887    if (!normalizedAttribute.equals(CHARSET_ATTRIBUTE)) {
888      mediaType.parsedCharset = this.parsedCharset;
889    }
890    // Return one of the constants if the media type is a known type.
891    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
892  }

896   * given value. If multiple parameters with the same attributes are necessary use {@link
897   * #withParameters(String, Iterable)}. Prefer {@link #withCharset} for setting the {@code charset}
898   * parameter when using a {@link Charset} object.
899   *
900   * @throws IllegalArgumentException if either {@code attribute} or {@code value} is invalid

902  public MediaType withParameter(String attribute, String value) {

908   * charset} parameter set to the {@link Charset#name name} of the given charset. Only one {@code
909   * charset} parameter will be present on the new instance regardless of the number set on this
910   * one.

928  /**
929   * Returns {@code true} if this instance falls within the range (as defined by <a
930   * href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html">the HTTP Accept header</a>) given
931   * by the argument according to three criteria:
932   *
933   * <ol>
934   *   <li>The type of the argument is the wildcard or equal to the type of this instance.
935   *   <li>The subtype of the argument is the wildcard or equal to the subtype of this instance.
936   *   <li>All of the parameters present in the argument are present in this instance.
937   * </ol>
938   *
939   * <p>For example:
940   *
941   * <pre>{@code
942   * PLAIN_TEXT_UTF_8.is(PLAIN_TEXT_UTF_8) // true
943   * PLAIN_TEXT_UTF_8.is(HTML_UTF_8) // false
944   * PLAIN_TEXT_UTF_8.is(ANY_TYPE) // true
945   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE) // true
946   * PLAIN_TEXT_UTF_8.is(ANY_IMAGE_TYPE) // false
947   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_8)) // true
948   * PLAIN_TEXT_UTF_8.withoutParameters().is(ANY_TEXT_TYPE.withCharset(UTF_8)) // false
949   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_16)) // false
950   * }</pre>
951   *
952   * <p>Note that while it is possible to have the same parameter declared multiple times within a
953   * media type this method does not consider the number of occurrences of a parameter. For example,
954   * {@code "text/plain; charset=UTF-8"} satisfies {@code "text/plain; charset=UTF-8;
955   * charset=UTF-8"}.
956   */
957  public boolean is(MediaType mediaTypeRange) {
958    return (mediaTypeRange.type.equals(WILDCARD) || mediaTypeRange.type.equals(this.type))
959        && (mediaTypeRange.subtype.equals(WILDCARD) || mediaTypeRange.subtype.equals(this.subtype))
960        && this.parameters.entries().containsAll(mediaTypeRange.parameters.entries());
961  }

963  /**
964   * Creates a new media type with the given type and subtype.
965   *
966   * @throws IllegalArgumentException if type or subtype is invalid or if a wildcard is used for the
967   *     type, but not the subtype.
968   */
969  public static MediaType create(String type, String subtype) {
970    MediaType mediaType = create(type, subtype, ImmutableListMultimap.<String, String>of());
971    mediaType.parsedCharset = Optional.absent();
972    return mediaType;
973  }

975  private static MediaType create(
976      String type, String subtype, Multimap<String, String> parameters) {
977    checkNotNull(type);
978    checkNotNull(subtype);
979    checkNotNull(parameters);
980    String normalizedType = normalizeToken(type);
981    String normalizedSubtype = normalizeToken(subtype);
982    checkArgument(
983        !WILDCARD.equals(normalizedType) || WILDCARD.equals(normalizedSubtype),
984        "A wildcard type cannot be used with a non-wildcard subtype");
985    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
986    for (Entry<String, String> entry : parameters.entries()) {
987      String attribute = normalizeToken(entry.getKey());
988      builder.put(attribute, normalizeParameterValue(attribute, entry.getValue()));
989    }
990    MediaType mediaType = new MediaType(normalizedType, normalizedSubtype, builder.build());
991    // Return one of the constants if the media type is a known type.
992    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
993  }

1049  private static String normalizeToken(String token) {
1050    checkArgument(TOKEN_MATCHER.matchesAllOf(token));
1051    checkArgument(!token.isEmpty());
1052    return Ascii.toLowerCase(token);
1053  }
1054
1055  private static String normalizeParameterValue(String attribute, String value) {
1056    checkNotNull(value); // for GWT
1057    checkArgument(ascii().matchesAllOf(value), "parameter values must be ASCII: %s", value);
1058    return CHARSET_ATTRIBUTE.equals(attribute) ? Ascii.toLowerCase(value) : value;
1059  }

1061  /**
1062   * Parses a media type from its string representation.
1063   *
1064   * @throws IllegalArgumentException if the input is not parsable
1065   */
1066  @CanIgnoreReturnValue // TODO(b/219820829): consider removing
1067  public static MediaType parse(String input) {
1068    checkNotNull(input);
1069    Tokenizer tokenizer = new Tokenizer(input);
1070    try {
1071      String type = tokenizer.consumeToken(TOKEN_MATCHER);
1072      consumeSeparator(tokenizer, '/');
1073      String subtype = tokenizer.consumeToken(TOKEN_MATCHER);
1074      ImmutableListMultimap.Builder<String, String> parameters = ImmutableListMultimap.builder();
1075      while (tokenizer.hasMore()) {
1076        consumeSeparator(tokenizer, ';');
1077        String attribute = tokenizer.consumeToken(TOKEN_MATCHER);
1078        consumeSeparator(tokenizer, '=');
1079        String value;
1080        if ('"' == tokenizer.previewChar()) {
1081          tokenizer.consumeCharacter('"');
1082          StringBuilder valueBuilder = new StringBuilder();
1083          while ('"' != tokenizer.previewChar()) {
1084            if ('\\' == tokenizer.previewChar()) {
1085              tokenizer.consumeCharacter('\\');
1086              valueBuilder.append(tokenizer.consumeCharacter(ascii()));
1087            } else {
1088              valueBuilder.append(tokenizer.consumeToken(QUOTED_TEXT_MATCHER));
1089            }
1090          }
1091          value = valueBuilder.toString();
1092          tokenizer.consumeCharacter('"');
1093        } else {
1094          value = tokenizer.consumeToken(TOKEN_MATCHER);
1095        }
1096        parameters.put(attribute, value);
1097      }
1098      return create(type, subtype, parameters.build());
1099    } catch (IllegalStateException e) {
1100      throw new IllegalArgumentException("Could not parse '" + input + "'", e);
1101    }
1102  }

1105    tokenizer.consumeTokenIfPresent(LINEAR_WHITE_SPACE);
1106    tokenizer.consumeCharacter(c);
1107    tokenizer.consumeTokenIfPresent(LINEAR_WHITE_SPACE);

1110  private static final class Tokenizer {
1111    final String input;
1112    int position = 0;
1113
1114    Tokenizer(String input) {
1115      this.input = input;
1116    }
1117
1118    @CanIgnoreReturnValue
1119    String consumeTokenIfPresent(CharMatcher matcher) {
1120      checkState(hasMore());
1121      int startPosition = position;
1122      position = matcher.negate().indexIn(input, startPosition);
1123      return hasMore() ? input.substring(startPosition, position) : input.substring(startPosition);
1124    }
1125
1126    String consumeToken(CharMatcher matcher) {
1127      int startPosition = position;
1128      String token = consumeTokenIfPresent(matcher);
1129      checkState(position != startPosition);
1130      return token;
1131    }
1132
1133    char consumeCharacter(CharMatcher matcher) {
1134      checkState(hasMore());
1135      char c = previewChar();
1136      checkState(matcher.matches(c));
1137      position++;
1138      return c;
1139    }
1140
1141    @CanIgnoreReturnValue
1142    char consumeCharacter(char c) {
1143      checkState(hasMore());
1144      checkState(previewChar() == c);
1145      position++;
1146      return c;
1147    }
1148
1149    char previewChar() {
1150      checkState(hasMore());
1151      return input.charAt(position);
1152    }
1153
1154    boolean hasMore() {
1155      return (position >= 0) && (position < input.length());
1156    }
1157  }

1185  private static final MapJoiner PARAMETER_JOINER = Joiner.on("; ").withKeyValueSeparator("=");
1186
1187  /**
1188   * Returns the string representation of this media type in the format described in <a
1189   * href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045</a>.
1190   */
1191  @Override
1192  public String toString() {
1193    // racy single-check idiom, safe because String is immutable
1194    String result = toString;
1195    if (result == null) {
1196      result = computeToString();
1197      toString = result;
1198    }
1199    return result;
1200  }
1201
1202  private String computeToString() {
1203    StringBuilder builder = new StringBuilder().append(type).append('/').append(subtype);
1204    if (!parameters.isEmpty()) {
1205      builder.append("; ");
1206      Multimap<String, String> quotedParameters =
1207          Multimaps.transformValues(
1208              parameters,
1209              (String value) ->
1210                  (TOKEN_MATCHER.matchesAllOf(value) && !value.isEmpty())
1211                      ? value
1212                      : escapeAndQuote(value));
1213      PARAMETER_JOINER.appendTo(builder, quotedParameters.entries());
1214    }
1215    return builder.toString();
1216  }
1217
1218  private static String escapeAndQuote(String value) {
1219    StringBuilder escaped = new StringBuilder(value.length() + 16).append('"');
1220    for (int i = 0; i < value.length(); i++) {
1221      char ch = value.charAt(i);
1222      if (ch == '\r' || ch == '\\' || ch == '"') {
1223        escaped.append('\\');
1224      }
1225      escaped.append(ch);
1226    }
1227    return escaped.append('"').toString();
1228  }
