030/**
031 * A mutable object which accumulates double values and tracks some basic statistics over all the
032 * values added so far. The values may be added singly or in groups. This class is not thread safe.
033 *
034 * @author Pete Gillin
035 * @author Kevin Bourrillion
036 * @since 20.0
037 */
038@J2ktIncompatible
039@GwtIncompatible
040@ElementTypesAreNonnullByDefault
041public final class StatsAccumulator {
042
043  // These fields must satisfy the requirements of Stats' constructor as well as those of the stat
044  // methods of this class.
045  private long count = 0;
046  private double mean = 0.0; // any finite value will do, we only use it to multiply by zero for sum
047  private double sumOfSquaresOfDeltas = 0.0;
048  private double min = NaN; // any value will do
049  private double max = NaN; // any value will do
050
051  /** Adds the given value to the dataset. */
052  public void add(double value) {
053    if (count == 0) {
054      count = 1;
055      mean = value;
056      min = value;
057      max = value;
058      if (!isFinite(value)) {
059        sumOfSquaresOfDeltas = NaN;
060      }
061    } else {
062      count++;
063      if (isFinite(value) && isFinite(mean)) {
064        // Art of Computer Programming vol. 2, Knuth, 4.2.2, (15) and (16)
065        double delta = value - mean;
066        mean += delta / count;
067        sumOfSquaresOfDeltas += delta * (value - mean);
068      } else {
069        mean = calculateNewMeanNonFinite(mean, value);
070        sumOfSquaresOfDeltas = NaN;
071      }
072      min = Math.min(min, value);
073      max = Math.max(max, value);
074    }
075  }

077  /**
078   * Adds the given values to the dataset.
079   *
080   * @param values a series of values, which will be converted to {@code double} values (this may
081   *     cause loss of precision)
082   */
083  public void addAll(Iterable<? extends Number> values) {
084    for (Number value : values) {
085      add(value.doubleValue());
086    }
087  }

172  /**
173   * Adds the given statistics to the dataset, as if the individual values used to compute the
174   * statistics had been added directly.
175   */
176  public void addAll(Stats values) {
177    if (values.count() == 0) {
178      return;
179    }
180    merge(values.count(), values.mean(), values.sumOfSquaresOfDeltas(), values.min(), values.max());
181  }
182
183  /**
184   * Adds the given statistics to the dataset, as if the individual values used to compute the
185   * statistics had been added directly.
186   *
187   * @since 28.2
188   */
189  public void addAll(StatsAccumulator values) {
190    if (values.count() == 0) {
191      return;
192    }
193    merge(values.count(), values.mean(), values.sumOfSquaresOfDeltas(), values.min(), values.max());
194  }
195
196  private void merge(
197      long otherCount,
198      double otherMean,
199      double otherSumOfSquaresOfDeltas,
200      double otherMin,
201      double otherMax) {
202    if (count == 0) {
203      count = otherCount;
204      mean = otherMean;
205      sumOfSquaresOfDeltas = otherSumOfSquaresOfDeltas;
206      min = otherMin;
207      max = otherMax;
208    } else {
209      count += otherCount;
210      if (isFinite(mean) && isFinite(otherMean)) {
211        // This is a generalized version of the calculation in add(double) above.
212        double delta = otherMean - mean;
213        mean += delta * otherCount / count;
214        sumOfSquaresOfDeltas += otherSumOfSquaresOfDeltas + delta * (otherMean - mean) * otherCount;
215      } else {
216        mean = calculateNewMeanNonFinite(mean, otherMean);
217        sumOfSquaresOfDeltas = NaN;
218      }
219      min = Math.min(min, otherMin);
220      max = Math.max(max, otherMax);
221    }
222  }
223
224  /** Returns an immutable snapshot of the current statistics. */
225  public Stats snapshot() {
226    return new Stats(count, mean, sumOfSquaresOfDeltas, min, max);
227  }

234  /**
235   * Returns the <a href="http://en.wikipedia.org/wiki/Arithmetic_mean">arithmetic mean</a> of the
236   * values. The count must be non-zero.
237   *
238   * <p>If these values are a sample drawn from a population, this is also an unbiased estimator of
239   * the arithmetic mean of the population.
240   *
241   * <h3>Non-finite values</h3>
242   *
243   * <p>If the dataset contains {@link Double#NaN} then the result is {@link Double#NaN}. If it
244   * contains both {@link Double#POSITIVE_INFINITY} and {@link Double#NEGATIVE_INFINITY} then the
245   * result is {@link Double#NaN}. If it contains {@link Double#POSITIVE_INFINITY} and finite values
246   * only or {@link Double#POSITIVE_INFINITY} only, the result is {@link Double#POSITIVE_INFINITY}.
247   * If it contains {@link Double#NEGATIVE_INFINITY} and finite values only or {@link
248   * Double#NEGATIVE_INFINITY} only, the result is {@link Double#NEGATIVE_INFINITY}.
249   *
250   * @throws IllegalStateException if the dataset is empty
251   */
252  public double mean() {
253    checkState(count != 0);
254    return mean;
255  }
256
257  /**
258   * Returns the sum of the values.
259   *
260   * <h3>Non-finite values</h3>
261   *
262   * <p>If the dataset contains {@link Double#NaN} then the result is {@link Double#NaN}. If it
263   * contains both {@link Double#POSITIVE_INFINITY} and {@link Double#NEGATIVE_INFINITY} then the
264   * result is {@link Double#NaN}. If it contains {@link Double#POSITIVE_INFINITY} and finite values
265   * only or {@link Double#POSITIVE_INFINITY} only, the result is {@link Double#POSITIVE_INFINITY}.
266   * If it contains {@link Double#NEGATIVE_INFINITY} and finite values only or {@link
267   * Double#NEGATIVE_INFINITY} only, the result is {@link Double#NEGATIVE_INFINITY}.
268   */
269  public final double sum() {
270    return mean * count;
271  }

273  /**
274   * Returns the <a href="http://en.wikipedia.org/wiki/Variance#Population_variance">population
275   * variance</a> of the values. The count must be non-zero.
276   *
277   * <p>This is guaranteed to return zero if the dataset contains only exactly one finite value. It
278   * is not guaranteed to return zero when the dataset consists of the same value multiple times,
279   * due to numerical errors. However, it is guaranteed never to return a negative result.
280   *
281   * <h3>Non-finite values</h3>
282   *
283   * <p>If the dataset contains any non-finite values ({@link Double#POSITIVE_INFINITY}, {@link
284   * Double#NEGATIVE_INFINITY}, or {@link Double#NaN}) then the result is {@link Double#NaN}.
285   *
286   * @throws IllegalStateException if the dataset is empty
287   */
288  public final double populationVariance() {
289    checkState(count != 0);
290    if (isNaN(sumOfSquaresOfDeltas)) {
291      return NaN;
292    }
293    if (count == 1) {
294      return 0.0;
295    }
296    return ensureNonNegative(sumOfSquaresOfDeltas) / count;
297  }

299  /**
300   * Returns the <a
301   * href="http://en.wikipedia.org/wiki/Standard_deviation#Definition_of_population_values">
302   * population standard deviation</a> of the values. The count must be non-zero.
303   *
304   * <p>This is guaranteed to return zero if the dataset contains only exactly one finite value. It
305   * is not guaranteed to return zero when the dataset consists of the same value multiple times,
306   * due to numerical errors. However, it is guaranteed never to return a negative result.
307   *
308   * <h3>Non-finite values</h3>
309   *
310   * <p>If the dataset contains any non-finite values ({@link Double#POSITIVE_INFINITY}, {@link
311   * Double#NEGATIVE_INFINITY}, or {@link Double#NaN}) then the result is {@link Double#NaN}.
312   *
313   * @throws IllegalStateException if the dataset is empty
314   */
315  public final double populationStandardDeviation() {
316    return Math.sqrt(populationVariance());
317  }

319  /**
320   * Returns the <a href="http://en.wikipedia.org/wiki/Variance#Sample_variance">unbiased sample
321   * variance</a> of the values. If this dataset is a sample drawn from a population, this is an
322   * unbiased estimator of the population variance of the population. The count must be greater than
323   * one.
324   *
325   * <p>This is not guaranteed to return zero when the dataset consists of the same value multiple
326   * times, due to numerical errors. However, it is guaranteed never to return a negative result.
327   *
328   * <h3>Non-finite values</h3>
329   *
330   * <p>If the dataset contains any non-finite values ({@link Double#POSITIVE_INFINITY}, {@link
331   * Double#NEGATIVE_INFINITY}, or {@link Double#NaN}) then the result is {@link Double#NaN}.
332   *
333   * @throws IllegalStateException if the dataset is empty or contains a single value
334   */
335  public final double sampleVariance() {
336    checkState(count > 1);
337    if (isNaN(sumOfSquaresOfDeltas)) {
338      return NaN;
339    }
340    return ensureNonNegative(sumOfSquaresOfDeltas) / (count - 1);
341  }

405  /**
406   * Calculates the new value for the accumulated mean when a value is added, in the case where at
407   * least one of the previous mean and the value is non-finite.
408   */
409  static double calculateNewMeanNonFinite(double previousMean, double value) {
410    /*
411     * Desired behaviour is to match the results of applying the naive mean formula. In particular,
412     * the update formula can subtract infinities in cases where the naive formula would add them.
413     *
414     * Consequently:
415     * 1. If the previous mean is finite and the new value is non-finite then the new mean is that
416     *    value (whether it is NaN or infinity).
417     * 2. If the new value is finite and the previous mean is non-finite then the mean is unchanged
418     *    (whether it is NaN or infinity).
419     * 3. If both the previous mean and the new value are non-finite and...
420     * 3a. ...either or both is NaN (so mean != value) then the new mean is NaN.
421     * 3b. ...they are both the same infinities (so mean == value) then the mean is unchanged.
422     * 3c. ...they are different infinities (so mean != value) then the new mean is NaN.
423     */
424    if (isFinite(previousMean)) {
425      // This is case 1.
426      return value;
427    } else if (isFinite(value) || previousMean == value) {
428      // This is case 2. or 3b.
429      return previousMean;
430    } else {
431      // This is case 3a. or 3c.
432      return NaN;
433    }
434  }
