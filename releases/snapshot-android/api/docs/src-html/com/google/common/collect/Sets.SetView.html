109  /**
110   * Returns an immutable set instance containing the given enum elements. Internally, the returned
111   * set will be backed by an {@link EnumSet}.
112   *
113   * <p>The iteration order of the returned set follows the enum's iteration order, not the order in
114   * which the elements appear in the given collection.
115   *
116   * @param elements the elements, all of the same {@code enum} type, that the set should contain
117   * @return an immutable set containing those elements, minus duplicates
118   */
119  // http://code.google.com/p/google-web-toolkit/issues/detail?id=3028
120  @GwtCompatible(serializable = true)
121  public static <E extends Enum<E>> ImmutableSet<E> immutableEnumSet(Iterable<E> elements) {
122    if (elements instanceof ImmutableEnumSet) {
123      return (ImmutableEnumSet<E>) elements;
124    } else if (elements instanceof Collection) {
125      Collection<E> collection = (Collection<E>) elements;
126      if (collection.isEmpty()) {
127        return ImmutableSet.of();
128      } else {
129        return ImmutableEnumSet.asImmutable(EnumSet.copyOf(collection));
130      }
131    } else {
132      Iterator<E> itr = elements.iterator();
133      if (itr.hasNext()) {
134        EnumSet<E> enumSet = EnumSet.of(itr.next());
135        Iterators.addAll(enumSet, itr);
136        return ImmutableEnumSet.asImmutable(enumSet);
137      } else {
138        return ImmutableSet.of();
139      }
140    }
141  }

716          }
717        }

719      }

724      }

1613   * combinations(ImmutableSet.of(1, 2, 3), 2)} returns the set {@code {{1, 2}, {1, 3}, {2, 3}}}.

1615   * <p>Elements appear in these subsets in the same iteration order as they appeared in the input
1616   * set. The order in which these subsets appear in the outer set is undefined.
1617   *
1618   * <p>The returned set and its constituent sets use {@code equals} to decide whether two elements
1619   * are identical, even if the input set uses a different concept of equivalence.
1620   *
1621   * <p><i>Performance notes:</i> the memory usage of the returned set is only {@code O(n)}. When
1622   * the result set is constructed, the input set is merely copied. Only as the result set is
1623   * iterated are the individual subsets created. Each of these subsets occupies an additional O(n)
1624   * memory but only for as long as the user retains a reference to it. That is, the set returned by
1625   * {@code combinations} does not retain the individual subsets.
1626   *
1627   * @param set the set of elements to take combinations of
1628   * @param size the number of elements per combination
1629   * @return the set of all combinations of {@code size} elements from {@code set}
1630   * @throws IllegalArgumentException if {@code size} is not between 0 and {@code set.size()}
1631   *     inclusive
1632   * @throws NullPointerException if {@code set} is or contains {@code null}
1633   * @since 23.0
1634   */
1635  public static <E> Set<Set<E>> combinations(Set<E> set, final int size) {
1636    final ImmutableMap<E, Integer> index = Maps.indexMap(set);
1637    checkNonnegative(size, "size");
1638    checkArgument(size <= index.size(), "size (%s) must be <= set.size() (%s)", size, index.size());
1639    if (size == 0) {
1640      return ImmutableSet.<Set<E>>of(ImmutableSet.<E>of());
1641    } else if (size == index.size()) {
1642      return ImmutableSet.<Set<E>>of(index.keySet());
1643    }
1644    return new AbstractSet<Set<E>>() {
1645      @Override
1646      public boolean contains(@CheckForNull Object o) {
1647        if (o instanceof Set) {
1648          Set<?> s = (Set<?>) o;
1649          return s.size() == size && index.keySet().containsAll(s);
1650        }
1651        return false;
1652      }
1653
1654      @Override
1655      public Iterator<Set<E>> iterator() {
1656        return new AbstractIterator<Set<E>>() {
1657          final BitSet bits = new BitSet(index.size());
1658
1659          @Override
1660          @CheckForNull
1661          protected Set<E> computeNext() {
1662            if (bits.isEmpty()) {
1663              bits.set(0, size);
1664            } else {
1665              int firstSetBit = bits.nextSetBit(0);
1666              int bitToFlip = bits.nextClearBit(firstSetBit);
1667
1668              if (bitToFlip == index.size()) {
1669                return endOfData();
1670              }
1671
1672              /*
1673               * The current set in sorted order looks like
1674               * {firstSetBit, firstSetBit + 1, ..., bitToFlip - 1, ...}
1675               * where it does *not* contain bitToFlip.
1676               *
1677               * The next combination is
1678               *
1679               * {0, 1, ..., bitToFlip - firstSetBit - 2, bitToFlip, ...}
1680               *
1681               * This is lexicographically next if you look at the combinations in descending order
1682               * e.g. {2, 1, 0}, {3, 1, 0}, {3, 2, 0}, {3, 2, 1}, {4, 1, 0}...
1683               */
1684
1685              bits.set(0, bitToFlip - firstSetBit - 1);
1686              bits.clear(bitToFlip - firstSetBit - 1, bitToFlip);
1687              bits.set(bitToFlip);
1688            }
1689            final BitSet copy = (BitSet) bits.clone();
1690            return new AbstractSet<E>() {
1691              @Override
1692              public boolean contains(@CheckForNull Object o) {
1693                Integer i = index.get(o);
1694                return i != null && copy.get(i);
1695              }
1696
1697              @Override
1698              public Iterator<E> iterator() {
1699                return new AbstractIterator<E>() {
1700                  int i = -1;
1701
1702                  @Override
1703                  @CheckForNull
1704                  protected E computeNext() {
1705                    i = copy.nextSetBit(i + 1);
1706                    if (i == -1) {
1707                      return endOfData();
1708                    }
1709                    return index.keySet().asList().get(i);
1710                  }
1711                };
1712              }
1713
1714              @Override
1715              public int size() {
1716                return size;
1717              }
1718            };
1719          }
1720        };
1721      }
1722
1723      @Override
1724      public int size() {
1725        return IntMath.binomial(index.size(), size);
1726      }
1727
1728      @Override
1729      public String toString() {
1730        return "Sets.combinations(" + index.keySet() + ", " + size + ")";
1731      }
1732    };
1733  }

1879   * Returns a synchronized (thread-safe) navigable set backed by the specified navigable set. In
1880   * order to guarantee serial access, it is critical that <b>all</b> access to the backing
1881   * navigable set is accomplished through the returned navigable set (or its views).
1882   *
1883   * <p>It is imperative that the user manually synchronize on the returned sorted set when
1884   * iterating over it or any of its {@code descendingSet}, {@code subSet}, {@code headSet}, or

1886   *
1887   * <pre>{@code
1888   * NavigableSet<E> set = synchronizedNavigableSet(new TreeSet<E>());
1889   *  ...
1890   * synchronized (set) {
1891   *   // Must be in the synchronized block
1892   *   Iterator<E> it = set.iterator();
1893   *   while (it.hasNext()) {
1894   *     foo(it.next());

1896   * }
1897   * }</pre>

1900   *

1906   *   // Must be in the synchronized block
1907   *   Iterator<E> it = set2.descendingIterator();

1932  /** Remove each element in an iterable from a set. */
1933  static boolean removeAllImpl(Set<?> set, Iterator<?> iterator) {
1934    boolean changed = false;
1935    while (iterator.hasNext()) {
1936      changed |= set.remove(iterator.next());
1937    }
1938    return changed;
1939  }

1941  static boolean removeAllImpl(Set<?> set, Collection<?> collection) {
1942    checkNotNull(collection); // for GWT
1943    if (collection instanceof Multiset) {
1944      collection = ((Multiset<?>) collection).elementSet();
1945    }
1946    /*
1947     * AbstractSet.removeAll(List) has quadratic behavior if the list size
1948     * is just more than the set's size.  We augment the test by
1949     * assuming that sets have fast contains() performance, and other
1950     * collections don't.  See
1951     * http://code.google.com/p/guava-libraries/issues/detail?id=1013
1952     */
1953    if (collection instanceof Set && collection.size() > set.size()) {
1954      return Iterators.removeAll(set.iterator(), collection);
1955    } else {
1956      return removeAllImpl(set, collection.iterator());
1957    }

2103  /**
2104   * Returns a view of the portion of {@code set} whose elements are contained by {@code range}.
2105   *
2106   * <p>This method delegates to the appropriate methods of {@link NavigableSet} (namely {@link
2107   * NavigableSet#subSet(Object, boolean, Object, boolean) subSet()}, {@link
2108   * NavigableSet#tailSet(Object, boolean) tailSet()}, and {@link NavigableSet#headSet(Object,
2109   * boolean) headSet()}) to actually construct the view. Consult these methods for a full
2110   * description of the returned view's behavior.
2111   *
2112   * <p><b>Warning:</b> {@code Range}s always represent a range of values using the values' natural
2113   * ordering. {@code NavigableSet} on the other hand can specify a custom ordering via a {@link
2114   * Comparator}, which can violate the natural ordering. Using this method (or in general using
2115   * {@code Range}) with unnaturally-ordered sets can lead to unexpected and undefined behavior.
2116   *
2117   * @since 20.0
2118   */
2119  @GwtIncompatible // NavigableSet
2120  public static <K extends Comparable<? super K>> NavigableSet<K> subSet(
2121      NavigableSet<K> set, Range<K> range) {
2122    if (set.comparator() != null
2123        && set.comparator() != Ordering.natural()
2124        && range.hasLowerBound()
2125        && range.hasUpperBound()) {
2126      checkArgument(
2127          set.comparator().compare(range.lowerEndpoint(), range.upperEndpoint()) <= 0,
2128          "set is using a custom comparator which is inconsistent with the natural ordering.");
2129    }
2130    if (range.hasLowerBound() && range.hasUpperBound()) {
2131      return set.subSet(
2132          range.lowerEndpoint(),
2133          range.lowerBoundType() == BoundType.CLOSED,
2134          range.upperEndpoint(),
2135          range.upperBoundType() == BoundType.CLOSED);
2136    } else if (range.hasLowerBound()) {
2137      return set.tailSet(range.lowerEndpoint(), range.lowerBoundType() == BoundType.CLOSED);
2138    } else if (range.hasUpperBound()) {
2139      return set.headSet(range.upperEndpoint(), range.upperBoundType() == BoundType.CLOSED);
2140    }
2141    return checkNotNull(set);
2142  }
