042/**
043 * A {@link NavigableSet} whose contents will never change, with many other important properties
044 * detailed at {@link ImmutableCollection}.
045 *
046 * <p><b>Warning:</b> as with any sorted collection, you are strongly advised not to use a {@link
047 * Comparator} or {@link Comparable} type whose comparison behavior is <i>inconsistent with
048 * equals</i>. That is, {@code a.compareTo(b)} or {@code comparator.compare(a, b)} should equal zero
049 * <i>if and only if</i> {@code a.equals(b)}. If this advice is not followed, the resulting
050 * collection will not correctly obey its specification.
051 *
052 * <p>See the Guava User Guide article on <a href=
053 * "https://github.com/google/guava/wiki/ImmutableCollectionsExplained">immutable collections</a>.
054 *
055 * @author Jared Levy
056 * @author Louis Wasserman
057 * @since 2.0 (implements {@code NavigableSet} since 12.0)
058 */

069    } else {

265  /**
266   * Returns an immutable sorted set containing the given elements sorted by the given {@code
267   * Comparator}. When multiple elements are equivalent according to {@code compare()}, only the
268   * first one specified is included. This method iterates over {@code elements} at most once.
269   *
270   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
271   * safe to do so. The exact circumstances under which a copy will or will not be performed are
272   * undocumented and subject to change.
273   *
274   * @throws NullPointerException if {@code comparator} or any of {@code elements} is null
275   */
276  public static <E> ImmutableSortedSet<E> copyOf(
277      Comparator<? super E> comparator, Iterable<? extends E> elements) {
278    checkNotNull(comparator);
279    boolean hasSameComparator = SortedIterables.hasSameComparator(comparator, elements);
280
281    if (hasSameComparator && (elements instanceof ImmutableSortedSet)) {
282      @SuppressWarnings("unchecked")
283      ImmutableSortedSet<E> original = (ImmutableSortedSet<E>) elements;
284      if (!original.isPartialView()) {
285        return original;
286      }
287    }
288    @SuppressWarnings("unchecked") // elements only contains E's; it's safe.
289    E[] array = (E[]) Iterables.toArray(elements);
290    return construct(comparator, array.length, array);
291  }

313  /**
314   * Returns an immutable sorted set containing the elements of a sorted set, sorted by the same
315   * {@code Comparator}. That behavior differs from {@link #copyOf(Iterable)}, which always uses the
316   * natural ordering of the elements.
317   *
318   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
319   * safe to do so. The exact circumstances under which a copy will or will not be performed are
320   * undocumented and subject to change.
321   *
322   * <p>This method is safe to use even when {@code sortedSet} is a synchronized or concurrent
323   * collection that is currently being modified by another thread.
324   *
325   * @throws NullPointerException if {@code sortedSet} or any of its elements is null
326   */
327  public static <E> ImmutableSortedSet<E> copyOfSorted(SortedSet<E> sortedSet) {
328    Comparator<? super E> comparator = SortedIterables.comparator(sortedSet);
329    ImmutableList<E> list = ImmutableList.copyOf(sortedSet);
330    if (list.isEmpty()) {
331      return emptySet(comparator);
332    } else {
333      return new RegularImmutableSortedSet<E>(list, comparator);
334    }
335  }

337  /**
338   * Constructs an {@code ImmutableSortedSet} from the first {@code n} elements of {@code contents}.
339   * If {@code k} is the size of the returned {@code ImmutableSortedSet}, then the sorted unique
340   * elements are in the first {@code k} positions of {@code contents}, and {@code contents[i] ==
341   * null} for {@code k <= i < n}.
342   *
343   * <p>This method takes ownership of {@code contents}; do not modify {@code contents} after this
344   * returns.
345   *
346   * @throws NullPointerException if any of the first {@code n} elements of {@code contents} is null
347   */
348  static <E> ImmutableSortedSet<E> construct(
349      Comparator<? super E> comparator, int n, E... contents) {
350    if (n == 0) {
351      return emptySet(comparator);
352    }
353    checkElementsNotNull(contents, n);
354    Arrays.sort(contents, 0, n, comparator);
355    int uniques = 1;
356    for (int i = 1; i < n; i++) {
357      E cur = contents[i];
358      E prev = contents[uniques - 1];
359      if (comparator.compare(cur, prev) != 0) {
360        contents[uniques++] = cur;
361      }
362    }
363    Arrays.fill(contents, uniques, n, null);
364    if (uniques < contents.length / 2) {
365      // Deduplication eliminated many of the elements.  We don't want to retain an arbitrarily
366      // large array relative to the number of elements, so we cap the ratio.
367      contents = Arrays.copyOf(contents, uniques);
368    }
369    return new RegularImmutableSortedSet<E>(
370        ImmutableList.<E>asImmutableList(contents, uniques), comparator);
371  }

403  /**
404   * A builder for creating immutable sorted set instances, especially {@code public static final}
405   * sets ("constant sets"), with a given comparator. Example:
406   *
407   * <pre>{@code
408   * public static final ImmutableSortedSet<Number> LUCKY_NUMBERS =
409   *     new ImmutableSortedSet.Builder<Number>(ODDS_FIRST_COMPARATOR)
410   *         .addAll(SINGLE_DIGIT_PRIMES)
411   *         .add(42)
412   *         .build();
413   * }</pre>
414   *
415   * <p>Builder instances can be reused; it is safe to call {@link #build} multiple times to build
416   * multiple sets in series. Each set is a superset of the set created before it.
417   *
418   * @since 2.0
419   */
420  public static final class Builder<E> extends ImmutableSet.Builder<E> {
421    private final Comparator<? super E> comparator;
422
423    /**
424     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
425     * ImmutableSortedSet#orderedBy}.
426     */
427    public Builder(Comparator<? super E> comparator) {
428      this.comparator = checkNotNull(comparator);
429    }
430
431    /**
432     * Adds {@code element} to the {@code ImmutableSortedSet}. If the {@code ImmutableSortedSet}
433     * already contains {@code element}, then {@code add} has no effect. (only the previously added
434     * element is retained).
435     *
436     * @param element the element to add
437     * @return this {@code Builder} object
438     * @throws NullPointerException if {@code element} is null
439     */
440    @CanIgnoreReturnValue
441    @Override
442    public Builder<E> add(E element) {
443      super.add(element);
444      return this;
445    }
446
447    /**
448     * Adds each element of {@code elements} to the {@code ImmutableSortedSet}, ignoring duplicate
449     * elements (only the first duplicate element is added).
450     *
451     * @param elements the elements to add
452     * @return this {@code Builder} object
453     * @throws NullPointerException if {@code elements} contains a null element
454     */
455    @CanIgnoreReturnValue
456    @Override
457    public Builder<E> add(E... elements) {
458      super.add(elements);
459      return this;
460    }
461
462    /**
463     * Adds each element of {@code elements} to the {@code ImmutableSortedSet}, ignoring duplicate
464     * elements (only the first duplicate element is added).
465     *
466     * @param elements the elements to add to the {@code ImmutableSortedSet}
467     * @return this {@code Builder} object
468     * @throws NullPointerException if {@code elements} contains a null element
469     */
470    @CanIgnoreReturnValue
471    @Override
472    public Builder<E> addAll(Iterable<? extends E> elements) {
473      super.addAll(elements);
474      return this;
475    }
476
477    /**
478     * Adds each element of {@code elements} to the {@code ImmutableSortedSet}, ignoring duplicate
479     * elements (only the first duplicate element is added).
480     *
481     * @param elements the elements to add to the {@code ImmutableSortedSet}
482     * @return this {@code Builder} object
483     * @throws NullPointerException if {@code elements} contains a null element
484     */
485    @CanIgnoreReturnValue
486    @Override
487    public Builder<E> addAll(Iterator<? extends E> elements) {
488      super.addAll(elements);
489      return this;
490    }
491
492    @CanIgnoreReturnValue
493    @Override
494    Builder<E> combine(ImmutableSet.Builder<E> builder) {
495      super.combine(builder);
496      return this;
497    }
498
499    /**
500     * Returns a newly-created {@code ImmutableSortedSet} based on the contents of the {@code
501     * Builder} and its comparator.
502     */
503    @Override
504    public ImmutableSortedSet<E> build() {
505      @SuppressWarnings("unchecked") // we're careful to put only E's in here
506      E[] contentsArray = (E[]) contents;
507      ImmutableSortedSet<E> result = construct(comparator, size, contentsArray);
508      this.size = result.size(); // we eliminated duplicates in-place in contentsArray
509      this.forceCopy = true;
510      return result;
511    }
512  }

514  int unsafeCompare(Object a, @CheckForNull Object b) {
515    return unsafeCompare(comparator, a, b);
516  }

518  static int unsafeCompare(Comparator<?> comparator, Object a, @CheckForNull Object b) {
519    // Pretend the comparator can compare anything. If it turns out it can't
520    // compare a and b, we should get a CCE or NPE on the subsequent line. Only methods
521    // that are spec'd to throw CCE and NPE should call this.
522    @SuppressWarnings({"unchecked", "nullness"})
523    Comparator<@Nullable Object> unsafeComparator = (Comparator<@Nullable Object>) comparator;
524    return unsafeComparator.compare(a, b);
525  }

733  /*
734   * This class is used to serialize all ImmutableSortedSet instances,
735   * regardless of implementation type. It captures their "logical contents"
736   * only. This is necessary to ensure that the existence of a particular
737   * implementation type is an implementation detail.
738   */
739  @J2ktIncompatible // serialization
740  private static class SerializedForm<E> implements Serializable {
741    final Comparator<? super E> comparator;
742    final Object[] elements;
743
744    public SerializedForm(Comparator<? super E> comparator, Object[] elements) {
745      this.comparator = comparator;
746      this.elements = elements;
747    }
748
749    @SuppressWarnings("unchecked")
750    Object readResolve() {
751      return new Builder<E>(comparator).add((E[]) elements).build();
752    }
753
754    private static final long serialVersionUID = 0;
755  }

759    throw new InvalidObjectException("Use SerializedForm");
760  }
