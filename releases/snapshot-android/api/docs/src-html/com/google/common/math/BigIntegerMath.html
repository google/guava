078  /** Returns {@code true} if {@code x} represents a power of two. */
079  public static boolean isPowerOfTwo(BigInteger x) {
080    checkNotNull(x);
081    return x.signum() > 0 && x.getLowestSetBit() == x.bitLength() - 1;
082  }

084  /**
085   * Returns the base-2 logarithm of {@code x}, rounded according to the specified rounding mode.
086   *
087   * @throws IllegalArgumentException if {@code x <= 0}
088   * @throws ArithmeticException if {@code mode} is {@link RoundingMode#UNNECESSARY} and {@code x}
089   *     is not a power of two
090   */
091  @SuppressWarnings("fallthrough")
092  // TODO(kevinb): remove after this warning is disabled globally
093  public static int log2(BigInteger x, RoundingMode mode) {
094    checkPositive("x", checkNotNull(x));
095    int logFloor = x.bitLength() - 1;
096    switch (mode) {
097      case UNNECESSARY:
098        checkRoundingUnnecessary(isPowerOfTwo(x)); // fall through
099      case DOWN:
100      case FLOOR:
101        return logFloor;
102
103      case UP:
104      case CEILING:
105        return isPowerOfTwo(x) ? logFloor : logFloor + 1;
106
107      case HALF_DOWN:
108      case HALF_UP:
109      case HALF_EVEN:
110        if (logFloor < SQRT2_PRECOMPUTE_THRESHOLD) {
111          BigInteger halfPower =
112              SQRT2_PRECOMPUTED_BITS.shiftRight(SQRT2_PRECOMPUTE_THRESHOLD - logFloor);
113          if (x.compareTo(halfPower) <= 0) {
114            return logFloor;
115          } else {
116            return logFloor + 1;
117          }
118        }
119        // Since sqrt(2) is irrational, log2(x) - logFloor cannot be exactly 0.5
120        //
121        // To determine which side of logFloor.5 the logarithm is,
122        // we compare x^2 to 2^(2 * logFloor + 1).
123        BigInteger x2 = x.pow(2);
124        int logX2Floor = x2.bitLength() - 1;
125        return (logX2Floor < 2 * logFloor + 1) ? logFloor : logFloor + 1;
126
127      default:
128        throw new AssertionError();
129    }
130  }

143  /**
144   * Returns the base-10 logarithm of {@code x}, rounded according to the specified rounding mode.
145   *
146   * @throws IllegalArgumentException if {@code x <= 0}
147   * @throws ArithmeticException if {@code mode} is {@link RoundingMode#UNNECESSARY} and {@code x}
148   *     is not a power of ten
149   */
150  @J2ktIncompatible
151  @GwtIncompatible // TODO
152  @SuppressWarnings("fallthrough")
153  public static int log10(BigInteger x, RoundingMode mode) {
154    checkPositive("x", x);
155    if (fitsInLong(x)) {
156      return LongMath.log10(x.longValue(), mode);
157    }
158
159    int approxLog10 = (int) (log2(x, FLOOR) * LN_2 / LN_10);
160    BigInteger approxPow = BigInteger.TEN.pow(approxLog10);
161    int approxCmp = approxPow.compareTo(x);
162
163    /*
164     * We adjust approxLog10 and approxPow until they're equal to floor(log10(x)) and
165     * 10^floor(log10(x)).
166     */
167
168    if (approxCmp > 0) {
169      /*
170       * The code is written so that even completely incorrect approximations will still yield the
171       * correct answer eventually, but in practice this branch should almost never be entered, and
172       * even then the loop should not run more than once.
173       */
174      do {
175        approxLog10--;
176        approxPow = approxPow.divide(BigInteger.TEN);
177        approxCmp = approxPow.compareTo(x);
178      } while (approxCmp > 0);
179    } else {
180      BigInteger nextPow = BigInteger.TEN.multiply(approxPow);
181      int nextCmp = nextPow.compareTo(x);
182      while (nextCmp <= 0) {
183        approxLog10++;
184        approxPow = nextPow;
185        approxCmp = nextCmp;
186        nextPow = BigInteger.TEN.multiply(approxPow);
187        nextCmp = nextPow.compareTo(x);
188      }
189    }
190
191    int floorLog = approxLog10;
192    BigInteger floorPow = approxPow;
193    int floorCmp = approxCmp;
194
195    switch (mode) {
196      case UNNECESSARY:
197        checkRoundingUnnecessary(floorCmp == 0);
198        // fall through
199      case FLOOR:
200      case DOWN:
201        return floorLog;
202
203      case CEILING:
204      case UP:
205        return floorPow.equals(x) ? floorLog : floorLog + 1;
206
207      case HALF_DOWN:
208      case HALF_UP:
209      case HALF_EVEN:
210        // Since sqrt(10) is irrational, log10(x) - floorLog can never be exactly 0.5
211        BigInteger x2 = x.pow(2);
212        BigInteger halfPowerSquared = floorPow.pow(2).multiply(BigInteger.TEN);
213        return (x2.compareTo(halfPowerSquared) <= 0) ? floorLog : floorLog + 1;
214      default:
215        throw new AssertionError();
216    }
217  }

220  private static final double LN_2 = Math.log(2);

222  /**
223   * Returns the square root of {@code x}, rounded with the specified rounding mode.
224   *
225   * @throws IllegalArgumentException if {@code x < 0}
226   * @throws ArithmeticException if {@code mode} is {@link RoundingMode#UNNECESSARY} and {@code
227   *     sqrt(x)} is not an integer
228   */
229  @J2ktIncompatible
230  @GwtIncompatible // TODO
231  @SuppressWarnings("fallthrough")
232  public static BigInteger sqrt(BigInteger x, RoundingMode mode) {
233    checkNonNegative("x", x);
234    if (fitsInLong(x)) {
235      return BigInteger.valueOf(LongMath.sqrt(x.longValue(), mode));
236    }
237    BigInteger sqrtFloor = sqrtFloor(x);
238    switch (mode) {
239      case UNNECESSARY:
240        checkRoundingUnnecessary(sqrtFloor.pow(2).equals(x)); // fall through
241      case FLOOR:
242      case DOWN:
243        return sqrtFloor;
244      case CEILING:
245      case UP:
246        int sqrtFloorInt = sqrtFloor.intValue();
247        boolean sqrtFloorIsExact =
248            (sqrtFloorInt * sqrtFloorInt == x.intValue()) // fast check mod 2^32
249                && sqrtFloor.pow(2).equals(x); // slow exact check
250        return sqrtFloorIsExact ? sqrtFloor : sqrtFloor.add(BigInteger.ONE);
251      case HALF_DOWN:
252      case HALF_UP:
253      case HALF_EVEN:
254        BigInteger halfSquare = sqrtFloor.pow(2).add(sqrtFloor);
255        /*
256         * We wish to test whether or not x <= (sqrtFloor + 0.5)^2 = halfSquare + 0.25. Since both x
257         * and halfSquare are integers, this is equivalent to testing whether or not x <=
258         * halfSquare.
259         */
260        return (halfSquare.compareTo(x) >= 0) ? sqrtFloor : sqrtFloor.add(BigInteger.ONE);
261      default:
262        throw new AssertionError();
263    }
264  }

266  @J2ktIncompatible
267  @GwtIncompatible // TODO
268  private static BigInteger sqrtFloor(BigInteger x) {
269    /*
270     * Adapted from Hacker's Delight, Figure 11-1.
271     *
272     * Using DoubleUtils.bigToDouble, getting a double approximation of x is extremely fast, and
273     * then we can get a double approximation of the square root. Then, we iteratively improve this
274     * guess with an application of Newton's method, which sets guess := (guess + (x / guess)) / 2.
275     * This iteration has the following two properties:
276     *
277     * a) every iteration (except potentially the first) has guess >= floor(sqrt(x)). This is
278     * because guess' is the arithmetic mean of guess and x / guess, sqrt(x) is the geometric mean,
279     * and the arithmetic mean is always higher than the geometric mean.
280     *
281     * b) this iteration converges to floor(sqrt(x)). In fact, the number of correct digits doubles
282     * with each iteration, so this algorithm takes O(log(digits)) iterations.
283     *
284     * We start out with a double-precision approximation, which may be higher or lower than the
285     * true value. Therefore, we perform at least one Newton iteration to get a guess that's
286     * definitely >= floor(sqrt(x)), and then continue the iteration until we reach a fixed point.
287     */
288    BigInteger sqrt0;
289    int log2 = log2(x, FLOOR);
290    if (log2 < Double.MAX_EXPONENT) {
291      sqrt0 = sqrtApproxWithDoubles(x);
292    } else {
293      int shift = (log2 - DoubleUtils.SIGNIFICAND_BITS) & ~1; // even!
294      /*
295       * We have that x / 2^shift < 2^54. Our initial approximation to sqrtFloor(x) will be
296       * 2^(shift/2) * sqrtApproxWithDoubles(x / 2^shift).
297       */
298      sqrt0 = sqrtApproxWithDoubles(x.shiftRight(shift)).shiftLeft(shift >> 1);
299    }
300    BigInteger sqrt1 = sqrt0.add(x.divide(sqrt0)).shiftRight(1);
301    if (sqrt0.equals(sqrt1)) {
302      return sqrt0;
303    }
304    do {
305      sqrt0 = sqrt1;
306      sqrt1 = sqrt0.add(x.divide(sqrt0)).shiftRight(1);
307    } while (sqrt1.compareTo(sqrt0) < 0);
308    return sqrt0;
309  }

317  /**
318   * Returns {@code x}, rounded to a {@code double} with the specified rounding mode. If {@code x}
319   * is precisely representable as a {@code double}, its {@code double} value will be returned;
320   * otherwise, the rounding will choose between the two nearest representable values with {@code
321   * mode}.
322   *
323   * <p>For the case of {@link RoundingMode#HALF_DOWN}, {@code HALF_UP}, and {@code HALF_EVEN},
324   * infinite {@code double} values are considered infinitely far away. For example, 2^2000 is not
325   * representable as a double, but {@code roundToDouble(BigInteger.valueOf(2).pow(2000), HALF_UP)}
326   * will return {@code Double.MAX_VALUE}, not {@code Double.POSITIVE_INFINITY}.
327   *
328   * <p>For the case of {@link RoundingMode#HALF_EVEN}, this implementation uses the IEEE 754
329   * default rounding mode: if the two nearest representable values are equally near, the one with
330   * the least significant bit zero is chosen. (In such cases, both of the nearest representable
331   * values are even integers; this method returns the one that is a multiple of a greater power of
332   * two.)
333   *
334   * @throws ArithmeticException if {@code mode} is {@link RoundingMode#UNNECESSARY} and {@code x}
335   *     is not precisely representable as a {@code double}
336   * @since 30.0
337   */
338  @J2ktIncompatible
339  @GwtIncompatible
340  public static double roundToDouble(BigInteger x, RoundingMode mode) {
341    return BigIntegerToDoubleRounder.INSTANCE.roundToDouble(x, mode);
342  }

387  /**
388   * Returns {@code n!}, that is, the product of the first {@code n} positive integers, or {@code 1}
389   * if {@code n == 0}.
390   *
391   * <p><b>Warning:</b> the result takes <i>O(n log n)</i> space, so use cautiously.
392   *
393   * <p>This uses an efficient binary recursive algorithm to compute the factorial with balanced
394   * multiplies. It also removes all the 2s from the intermediate products (shifting them back in at
395   * the end).
396   *
397   * @throws IllegalArgumentException if {@code n < 0}
398   */
399  public static BigInteger factorial(int n) {
400    checkNonNegative("n", n);
401
402    // If the factorial is small enough, just use LongMath to do it.
403    if (n < LongMath.factorials.length) {
404      return BigInteger.valueOf(LongMath.factorials[n]);
405    }
406
407    // Pre-allocate space for our list of intermediate BigIntegers.
408    int approxSize = IntMath.divide(n * IntMath.log2(n, CEILING), Long.SIZE, CEILING);
409    ArrayList<BigInteger> bignums = new ArrayList<>(approxSize);
410
411    // Start from the pre-computed maximum long factorial.
412    int startingNumber = LongMath.factorials.length;
413    long product = LongMath.factorials[startingNumber - 1];
414    // Strip off 2s from this value.
415    int shift = Long.numberOfTrailingZeros(product);
416    product >>= shift;
417
418    // Use floor(log2(num)) + 1 to prevent overflow of multiplication.
419    int productBits = LongMath.log2(product, FLOOR) + 1;
420    int bits = LongMath.log2(startingNumber, FLOOR) + 1;
421    // Check for the next power of two boundary, to save us a CLZ operation.
422    int nextPowerOfTwo = 1 << (bits - 1);
423
424    // Iteratively multiply the longs as big as they can go.
425    for (long num = startingNumber; num <= n; num++) {
426      // Check to see if the floor(log2(num)) + 1 has changed.
427      if ((num & nextPowerOfTwo) != 0) {
428        nextPowerOfTwo <<= 1;
429        bits++;
430      }
431      // Get rid of the 2s in num.
432      int tz = Long.numberOfTrailingZeros(num);
433      long normalizedNum = num >> tz;
434      shift += tz;
435      // Adjust floor(log2(num)) + 1.
436      int normalizedBits = bits - tz;
437      // If it won't fit in a long, then we store off the intermediate product.
438      if (normalizedBits + productBits >= Long.SIZE) {
439        bignums.add(BigInteger.valueOf(product));
440        product = 1;
441        productBits = 0;
442      }
443      product *= normalizedNum;
444      productBits = LongMath.log2(product, FLOOR) + 1;
445    }
446    // Check for leftovers.
447    if (product > 1) {
448      bignums.add(BigInteger.valueOf(product));
449    }
450    // Efficiently multiply all the intermediate products together.
451    return listProduct(bignums).shiftLeft(shift);
452  }
453
454  static BigInteger listProduct(List<BigInteger> nums) {
455    return listProduct(nums, 0, nums.size());
456  }
457
458  static BigInteger listProduct(List<BigInteger> nums, int start, int end) {
459    switch (end - start) {
460      case 0:
461        return BigInteger.ONE;
462      case 1:
463        return nums.get(start);
464      case 2:
465        return nums.get(start).multiply(nums.get(start + 1));
466      case 3:
467        return nums.get(start).multiply(nums.get(start + 1)).multiply(nums.get(start + 2));
468      default:
469        // Otherwise, split the list in half and recursively do this.
470        int m = (end + start) >>> 1;
471        return listProduct(nums, start, m).multiply(listProduct(nums, m, end));
472    }
473  }

475  /**
476   * Returns {@code n} choose {@code k}, also known as the binomial coefficient of {@code n} and
477   * {@code k}, that is, {@code n! / (k! (n - k)!)}.
478   *
479   * <p><b>Warning:</b> the result can take as much as <i>O(k log n)</i> space.
480   *
481   * @throws IllegalArgumentException if {@code n < 0}, {@code k < 0}, or {@code k > n}
482   */
483  public static BigInteger binomial(int n, int k) {
484    checkNonNegative("n", n);
485    checkNonNegative("k", k);
486    checkArgument(k <= n, "k (%s) > n (%s)", k, n);
487    if (k > (n >> 1)) {
488      k = n - k;
489    }
490    if (k < LongMath.biggestBinomials.length && n <= LongMath.biggestBinomials[k]) {
491      return BigInteger.valueOf(LongMath.binomial(n, k));
492    }
493
494    BigInteger accum = BigInteger.ONE;
495
496    long numeratorAccum = n;
497    long denominatorAccum = 1;
498
499    int bits = LongMath.log2(n, CEILING);
500
501    int numeratorBits = bits;
502
503    for (int i = 1; i < k; i++) {
504      int p = n - i;
505      int q = i + 1;
506
507      // log2(p) >= bits - 1, because p >= n/2
508
509      if (numeratorBits + bits >= Long.SIZE - 1) {
510        // The numerator is as big as it can get without risking overflow.
511        // Multiply numeratorAccum / denominatorAccum into accum.
512        accum =
513            accum
514                .multiply(BigInteger.valueOf(numeratorAccum))
515                .divide(BigInteger.valueOf(denominatorAccum));
516        numeratorAccum = p;
517        denominatorAccum = q;
518        numeratorBits = bits;
519      } else {
520        // We can definitely multiply into the long accumulators without overflowing them.
521        numeratorAccum *= p;
522        denominatorAccum *= q;
523        numeratorBits += bits;
524      }
525    }
526    return accum
527        .multiply(BigInteger.valueOf(numeratorAccum))
528        .divide(BigInteger.valueOf(denominatorAccum));
529  }
