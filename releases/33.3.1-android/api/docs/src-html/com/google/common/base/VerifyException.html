015package com.google.common.base;
016
017import com.google.common.annotations.GwtCompatible;
018import javax.annotation.CheckForNull;
019
020/**
021 * Exception thrown upon the failure of a <a
022 * href="https://github.com/google/guava/wiki/ConditionalFailuresExplained">verification check</a>,
023 * including those performed by the convenience methods of the {@link Verify} class.
024 *
025 * @since 17.0
026 */
027@GwtCompatible
028@ElementTypesAreNonnullByDefault
029public class VerifyException extends RuntimeException {
030  /** Constructs a {@code VerifyException} with no message. */
031  public VerifyException() {}
032
033  /** Constructs a {@code VerifyException} with the message {@code message}. */
034  public VerifyException(@CheckForNull String message) {
035    super(message);
036  }
037
038  /**
039   * Constructs a {@code VerifyException} with the cause {@code cause} and a message that is {@code
040   * null} if {@code cause} is null, and {@code cause.toString()} otherwise.
041   *
042   * @since 19.0
043   */
044  public VerifyException(@CheckForNull Throwable cause) {
045    super(cause);
046  }
047
048  /**
049   * Constructs a {@code VerifyException} with the message {@code message} and the cause {@code
050   * cause}.
051   *
052   * @since 19.0
053   */
054  public VerifyException(@CheckForNull String message, @CheckForNull Throwable cause) {
055    super(message, cause);
056  }
057}
